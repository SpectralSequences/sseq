1//! Computes the suspension map between different unstable Ext groups.
2//!
3//! Given an unstable Steenrod module $M$, compute the unstable Ext groups of $\Sigma^k M$ for all
4//! $k$ up till the stable range. Each result is printed in the form
5//! ```
6//! n s min_degree: num_gens - matrix
7//! ```
8//! The entries are to be interpreted as follows:
9//!  - `n` is the stem, which is defined to be `t - s - min_degree`
10//!  - `s` is the Adams filtration
11//!  - `min_degree` is the minimum degre of $\Sigma^k M$.
12//!  - `num_gens` is the number of generators in this Ext group
13//!  - `matrix` is the matrix representing the suspension map from $\Sigma^k M$. This is omitted if
14//!    the source or target of the suspension map is trivial, or if they have the same dimension
15//!    and the matrix is the identity matrix.
16//!
17//! The output is best read after sorting with `sort -n -k 1 -k 2 -k 3`.
18
19use std::{path::PathBuf, sync::Arc};
20
21use algebra::{
22    Algebra,
23    module::{Module, SuspensionModule},
24};
25use ext::{
26    chain_complex::{FiniteChainComplex, FreeChainComplex},
27    resolution::UnstableResolution,
28};
29use sseq::{
30    charting::{Backend, Orientation, TikzBackend},
31    coordinates::Bidegree,
32};
33
34fn main() -> anyhow::Result<()> {
35    ext::utils::init_logging()?;
36
37    let module = Arc::new(ext::utils::query_unstable_module_only()?);
38    let save_dir = {
39        let base = query::optional("Module save directory", |x| {
40            core::result::Result::<PathBuf, std::convert::Infallible>::Ok(PathBuf::from(x))
41        });
42        move |shift| {
43            base.as_ref().cloned().map(|mut x| {
44                x.push(format!("suspension{shift}"));
45                x
46            })
47        }
48    };
49
50    let max = Bidegree::n_s(
51        query::raw("Max n", str::parse),
52        query::raw("Max s", str::parse),
53    );
54
55    let disp_template: String = query::raw(
56        "LaTeX name template (replace % with min degree)",
57        str::parse,
58    );
59
60    let products = module.algebra().default_filtration_one_products();
61
62    for shift_t in 0..max.n() - module.min_degree() + 3 {
63        let shift = Bidegree::s_t(0, shift_t);
64        let res: Arc<UnstableResolution<FiniteChainComplex<_>>> =
65            Arc::new(UnstableResolution::new_with_save(
66                Arc::new(FiniteChainComplex::ccdz(Arc::new(SuspensionModule::new(
67                    Arc::clone(&module),
68                    shift.t(),
69                )))),
70                save_dir(shift.t()),
71            )?);
72
73        res.compute_through_stem(max + shift);
74
75        println!("\\begin{{figure}}[p]\\centering");
76
77        let sseq = res.to_sseq();
78        let products = products
79            .iter()
80            .map(|(name, op_deg, op_idx)| {
81                (name.clone(), res.filtration_one_products(*op_deg, *op_idx))
82            })
83            .collect::<Vec<_>>();
84
85        sseq.write_to_graph(
86            TikzBackend::new(std::io::stdout()),
87            2,
88            false,
89            products.iter(),
90            |g| {
91                g.text(
92                    1,
93                    max.s() as i32 - 1,
94                    disp_template.replace('%', &format!("{shift_t}")),
95                    Orientation::Right,
96                )
97            },
98        )?;
99
100        println!("\\end{{figure}}");
101    }
102    Ok(())
103}
