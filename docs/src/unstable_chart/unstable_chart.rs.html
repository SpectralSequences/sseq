1//! Computes the suspension map between different unstable Ext groups.
2//!
3//! Given an unstable Steenrod module $M$, compute the unstable Ext groups of $\Sigma^k M$ for all
4//! $k$ up till the stable range. Each result is printed in the form
5//! ```
6//! n s min_degree: num_gens - matrix
7//! ```
8//! The entries are to be interpreted as follows:
9//!  - `n` is the stem, which is defined to be `t - s - min_degree`
10//!  - `s` is the Adams filtration
11//!  - `min_degree` is the minimum degre of $\Sigma^k M$.
12//!  - `num_gens` is the number of generators in this Ext group
13//!  - `matrix` is the matrix representing the suspension map from $\Sigma^k M$. This is omitted if
14//!    the source or target of the suspension map is trivial, or if they have the same dimension
15//!    and the matrix is the identity matrix.
16//!
17//! The output is best read after sorting with `sort -n -k 1 -k 2 -k 3`.
18
19use std::{path::PathBuf, sync::Arc};
20
21use algebra::{
22    module::{Module, SuspensionModule},
23    Algebra,
24};
25use chart::{Backend, Orientation, TikzBackend};
26use ext::{
27    chain_complex::{FiniteChainComplex, FreeChainComplex},
28    resolution::UnstableResolution,
29};
30use sseq::coordinates::Bidegree;
31
32fn main() -> anyhow::Result<()> {
33    ext::utils::init_logging()?;
34
35    let module = Arc::new(ext::utils::query_unstable_module_only()?);
36    let save_dir = {
37        let base = query::optional("Module save directory", |x| {
38            core::result::Result::<PathBuf, std::convert::Infallible>::Ok(PathBuf::from(x))
39        });
40        move |shift| {
41            base.as_ref().cloned().map(|mut x| {
42                x.push(format!("suspension{shift}"));
43                x
44            })
45        }
46    };
47
48    let max = Bidegree::n_s(
49        query::raw("Max n", str::parse),
50        query::raw("Max s", str::parse),
51    );
52
53    let disp_template: String = query::raw(
54        "LaTeX name template (replace % with min degree)",
55        str::parse,
56    );
57
58    let products = module.algebra().default_filtration_one_products();
59
60    for shift_t in 0..max.n() - module.min_degree() + 3 {
61        let shift = Bidegree::s_t(0, shift_t);
62        let res: Arc<UnstableResolution<FiniteChainComplex<_>>> =
63            Arc::new(UnstableResolution::new_with_save(
64                Arc::new(FiniteChainComplex::ccdz(Arc::new(SuspensionModule::new(
65                    Arc::clone(&module),
66                    shift.t(),
67                )))),
68                save_dir(shift.t()),
69            )?);
70
71        res.compute_through_stem(max + shift);
72
73        println!("\\begin{{figure}}[p]\\centering");
74
75        let sseq = res.to_sseq();
76        let products = products
77            .iter()
78            .map(|(name, op_deg, op_idx)| {
79                (name.clone(), res.filtration_one_products(*op_deg, *op_idx))
80            })
81            .collect::<Vec<_>>();
82
83        sseq.write_to_graph(
84            TikzBackend::new(std::io::stdout()),
85            2,
86            false,
87            products.iter(),
88            |g| {
89                g.text(
90                    1,
91                    max.s() as i32 - 1,
92                    disp_template.replace('%', &format!("{shift_t}")),
93                    Orientation::Right,
94                )
95            },
96        )?;
97
98        println!("\\end{{figure}}");
99    }
100    Ok(())
101}
