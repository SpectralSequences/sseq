8fn main() -> anyhow::Result<()> {
9    ext::utils::init_logging()?;
10
11    eprintln!("This script computes Ext(M, N)");
12    let res = ext::utils::query_module_only("Module M", None, false)?;
13    let module_spec = query::raw("Module N", ext::utils::parse_module_name);
14    #[cfg(not(feature = "nassau"))]
15    let module = algebra::module::steenrod_module::from_json(res.algebra(), &module_spec)?;
16
17    #[cfg(feature = "nassau")]
18    let module = algebra::module::FDModule::from_json(res.algebra(), &module_spec)?;
19
20    let max = Bidegree::n_s(
21        query::raw("Max n", str::parse),
22        query::raw("Max s", str::parse),
23    );
24
25    res.compute_through_stem(max + Bidegree::n_s(module.max_degree().unwrap(), 1));
26    res.algebra()
27        .compute_basis(max.t() + module.max_degree().unwrap() + 2);
28
29    let hom_cc = HomCochainComplex::new(Arc::new(res), Arc::new(module));
30    hom_cc.compute_through_stem(max);
31
32    // FreeChainComplex::graded_dimension_string
33    let mut result = String::new();
34    for s in (0..=max.s()).rev() {
35        for n in hom_cc.min_degree()..=max.n() {
36            let b = Bidegree::n_s(n, s);
37            result.push(ext::utils::unicode_num(hom_cc.homology_dimension(b)));
38            result.push(' ');
39        }
40        result.push('\n');
41        // If it is empty so far, don't print anything
42        if result.trim_start().is_empty() {
43            result.clear()
44        }
45    }
46    print!("{result}");
47
48    Ok(())
49}

51mod hom_cochain_complex {
52    use std::sync::Arc;
53
54    use algebra::module::{
55        homomorphism::{HomPullback, ModuleHomomorphism},
56        HomModule, Module,
57    };
58    use ext::chain_complex::FreeChainComplex;
59    use fp::matrix::Subquotient;
60    use once::OnceBiVec;
61    use sseq::coordinates::Bidegree;
62
63    pub struct HomCochainComplex<CC: FreeChainComplex, M: Module<Algebra = CC::Algebra>> {
64        source: Arc<CC>,
65        target: Arc<M>,
66        modules: OnceBiVec<Arc<HomModule<M>>>,
67        differentials: OnceBiVec<Arc<HomPullback<M>>>,
68    }
69
70    impl<CC: FreeChainComplex, M: Module<Algebra = CC::Algebra>> HomCochainComplex<CC, M> {
71        pub fn new(source: Arc<CC>, target: Arc<M>) -> Self {
72            Self {
73                source,
74                target,
75                modules: OnceBiVec::new(0),
76                differentials: OnceBiVec::new(0),
77            }
78        }
79
80        pub fn min_degree(&self) -> i32 {
81            self.modules[0].min_degree()
82        }
83
84        pub fn compute_through_stem(&self, max: Bidegree) {
85            self.modules.extend(max.s() + 1, |s| {
86                Arc::new(HomModule::new(
87                    self.source.module(s),
88                    Arc::clone(&self.target),
89                ))
90            });
91            self.differentials.extend(max.s(), |s| {
92                Arc::new(HomPullback::new(
93                    Arc::clone(&self.modules[s]),
94                    Arc::clone(&self.modules[s + 1]),
95                    self.source.differential(s + 1),
96                ))
97            });
98            for (s, module) in self.modules.iter().enumerate() {
99                module.compute_basis(max.n() + s as i32 + 1);
100            }
101            for (s, d) in self.differentials.iter().enumerate() {
102                d.compute_auxiliary_data_through_degree(max.n() + s as i32 + 1);
103            }
104        }
105
106        pub fn homology_dimension(&self, b: Bidegree) -> usize {
107            if b.s() == 0 {
108                self.differentials[b.s()].kernel(b.t()).unwrap().dimension()
109            } else {
110                Subquotient::from_parts(
111                    self.differentials[b.s()].kernel(b.t()).cloned().unwrap(),
112                    self.differentials[b.s() - 1].image(b.t()).cloned().unwrap(),
113                )
114                .dimension()
115            }
116        }
117    }
118}
