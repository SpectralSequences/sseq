18pub fn get_gens() -> anyhow::Result<BiVec<Vec<String>>> {
19    ext::utils::init_logging()?;
20
21    // Query for generators
22    eprintln!("Input generators. Press return to finish.");
23    stderr().flush()?;
24
25    let mut gens: BiVec<Vec<_>> = BiVec::new(0);
26    loop {
27        let gen_deg: Option<i32> = query::optional("Generator degree", str::parse);
28        if gen_deg.is_none() {
29            eprintln!("This is the list of generators and degrees:");
30            for (i, deg_i_gens) in gens.iter_enum() {
31                for gen in deg_i_gens.iter() {
32                    eprint!("({i}, {gen}) ");
33                }
34            }
35            eprintln!();
36            if query::yes_no("Is it okay?") {
37                break;
38            } else {
39                if query::yes_no("Start over?") {
40                    gens = BiVec::new(0);
41                }
42                continue;
43            }
44        }
45
46        let gen_deg = gen_deg.unwrap();
47
48        gens.extend_negative(gen_deg, Vec::new());
49        gens.extend_with(gen_deg, |_| Vec::new());
50
51        let gen_name = query::with_default(
52            "Generator name",
53            &format!("x{gen_deg}{}", gens[gen_deg].len()).replace('-', "_"),
54            |x| {
55                match x.chars().next() {
56                    Some(a) => {
57                        if !a.is_alphabetic() {
58                            return Err("variable name must start with a letter".to_string());
59                        }
60                    }
61                    None => return Err("Variable name cannot be empty".to_string()),
62                };
63                for c in x.chars() {
64                    if !c.is_alphanumeric() && c != '_' {
65                        return Err(format!(
66                            "Variable name cannot contain {c}. Should be alphanumeric and '_'"
67                        ));
68                    }
69                }
70                Ok(x.to_string())
71            },
72        );
73        gens[gen_deg].push(gen_name);
74    }
75    Ok(gens)
76}

79    let mut gens_json = json!({});
80    for (i, deg_i_gens) in gens.iter_enum() {
81        for gen in deg_i_gens {

83        }
84    }
85    gens_json

88pub fn interactive_module_define_fdmodule(
89    output_json: &mut Value,
90    p: ValidPrime,
91) -> anyhow::Result<()> {
92    output_json["p"] = Value::from(p.as_u32());
93    let algebra = Arc::new(AdemAlgebra::new(p, false));
94
95    let gens = get_gens()?;
96    let min_degree = gens.min_degree();
97    let max_degree = gens.len();
98
99    algebra.compute_basis(max_degree - min_degree);
100
101    let mut graded_dim = BiVec::with_capacity(min_degree, max_degree);
102    for i in gens.iter().map(Vec::len) {
103        graded_dim.push(i);
104    }
105
106    let mut module = FDModule::new(Arc::clone(&algebra), String::new(), graded_dim);
107
108    for (i, deg_i_gens) in gens.iter_enum() {
109        for (j, gen) in deg_i_gens.iter().enumerate() {
110            module.set_basis_element_name(i, j, gen.to_string());
111        }
112    }
113
114    eprintln!(
115        "Input actions. Write the value of the action in the form 'a x0 + b x1 + ...' where a, b \
116         are non-negative integers and x0, x1 are names of the generators. The coefficient can be \
117         omitted if it is 1"
118    );
119
120    let len = gens.len();
121    for input_deg in gens.range().rev() {
122        for output_deg in (input_deg + 1)..len {
123            let op_deg = output_deg - input_deg;
124            if gens[output_deg].is_empty() {
125                continue;
126            }
127            for op_idx in algebra.generators(op_deg) {
128                for input_idx in 0..gens[input_deg].len() {
129                    let output = query::raw(
130                        &format!(
131                            "{} {}",
132                            algebra.basis_element_to_string(op_deg, op_idx),
133                            gens[input_deg][input_idx]
134                        ),
135                        |expr| {
136                            let mut result = vec![0; gens[output_deg].len()];
137                            if expr == "0" {
138                                return Ok(result);
139                            }
140                            for term in expr.split('+') {
141                                let term = term.trim();
142                                let (coef, gen) = match term.split_once(' ') {
143                                    Some((coef, gen)) => (str::parse::<u32>(coef)?, gen),
144                                    None => (1, term),
145                                };
146
147                                if let Some(gen_idx) =
148                                    gens[output_deg].iter().position(|d| d == gen)
149                                {
150                                    result[gen_idx] += coef;
151                                } else {
152                                    return Err(anyhow!(
153                                        "No generator {gen} in degree {output_deg}"
154                                    ));
155                                }
156                            }
157
158                            Ok(result)
159                        },
160                    );
161
162                    module.set_action(op_deg, op_idx, input_deg, input_idx, &output);
163                }
164            }
165            module.extend_actions(input_deg, output_deg);
166            module.check_validity(input_deg, output_deg)?;
167        }
168    }
169
170    module.to_json(output_json);
171    Ok(())
172}

174/// Given a string representation of an element in an algebra together with a generator, multiply
175/// each term on the right with the generator.
176fn replace(algebra_elt: &str, gen: &str) -> String {
177    algebra_elt.replace('+', &format!("{gen} +")) + " " + gen
178}

180pub fn interactive_module_define_fpmodule(
181    output_json: &mut Value,
182    p: ValidPrime,
183) -> anyhow::Result<()> {
184    let gens = get_gens()?;
185    let min_degree = gens.min_degree();
186    let max_degree = gens.len();
187
188    let ev = SteenrodEvaluator::new(p);
189
190    let mut graded_dim = BiVec::with_capacity(min_degree, max_degree);
191    for i in gens.iter().map(Vec::len) {
192        graded_dim.push(i);
193    }
194
195    eprintln!("Input relations");
196    match p.as_u32() {
197        2 => eprintln!("Write relations in the form 'Sq6 * Sq2 * x + Sq7 * y'"),
198        _ => eprintln!(
199            "Write relations in the form 'Q5 * P(5) * x + 2 * P(1, 3) * Q2 * y', where P(...) and \
200             Qi are Milnor basis elements."
201        ),
202    }
203
204    let mut degree_lookup = HashMap::default();
205    for (i, deg_i_gens) in gens.iter_enum() {
206        for gen in deg_i_gens.iter() {
207            degree_lookup.insert(gen.clone(), i);
208        }
209    }
210
211    let mut adem_relations = Vec::new();
212    let mut milnor_relations = Vec::new();
213
214    loop {
215        let relation = query::raw("Enter relation", |rel| {
216            let result = ev.evaluate_module_adem(rel)?;
217
218            if result.is_empty() {
219                return Ok(result);
220            }
221
222            // Check that the generators exist and the terms all have the same degree
223            let degrees = result
224                .iter()
225                .map(|(gen, (op_deg, _))| {
226                    degree_lookup
227                        .get(gen)
228                        .map(|d| d + op_deg)
229                        .ok_or_else(|| anyhow!("Unknown generator: {gen}"))
230                })
231                .collect::<Result<Vec<_>, _>>()?;
232            for window in degrees.windows(2) {
233                if window[0] != window[1] {
234                    return Err(anyhow!(
235                        "Relation terms have different degrees: {} and {}",
236                        window[0],
237                        window[1],
238                    ));
239                }
240            }
241
242            Ok(result)
243        });
244
245        if relation.is_empty() {
246            break;
247        }
248
249        let mut adem_relation = String::new();
250        let mut milnor_relation = String::new();
251
252        let mut milnor_op = FpVector::new(p, 0);
253        for (gen, (op_deg, adem_op)) in relation.iter() {
254            if adem_op.is_zero() {
255                continue;
256            }
257            if !adem_relation.is_empty() {
258                adem_relation += " + ";
259                milnor_relation += " + ";
260            }
261            milnor_op.set_scratch_vector_size(adem_op.len());
262            ev.adem_to_milnor(&mut milnor_op, 1, *op_deg, adem_op);
263
264            adem_relation += &replace(&ev.adem.element_to_string(*op_deg, adem_op.as_slice()), gen);
265            milnor_relation += &replace(
266                &ev.milnor.element_to_string(*op_deg, milnor_op.as_slice()),
267                gen,
268            );
269        }
270        if !adem_relation.is_empty() {
271            adem_relations.push(Value::String(adem_relation));
272            milnor_relations.push(Value::String(milnor_relation));
273        }
274    }
275
276    output_json["p"] = Value::from(p.as_u32());
277    output_json["type"] = Value::String("finitely presented module".to_owned());
278    for (i, deg_i_gens) in gens.iter_enum() {
279        for gen in deg_i_gens {
280            output_json["gens"][gen] = Value::from(i);
281        }
282    }
283    output_json["adem_relations"] = Value::Array(adem_relations);
284    output_json["milnor_relations"] = Value::Array(milnor_relations);
285    Ok(())
286}

288fn main() -> anyhow::Result<()> {
289    let module_type = query::with_default(
290        "Input module type (default 'finite dimensional module'):\n (fd) - finite dimensional \
291         module \n (fp) - finitely presented module\n",
292        "fd",
293        |x| match x {
294            "fd" | "fp" => Ok(x.to_string()),
295            _ => Err(format!("Invalid type '{x}'. Type must be 'fd' or 'fp'")),
296        },
297    );
298
299    let p: ValidPrime = query::with_default("p", "2", str::parse);
300    let mut output_json = json!({});
301
302    eprintln!("module_type: {module_type}");
303    match &*module_type {
304        "fd" => interactive_module_define_fdmodule(&mut output_json, p)?,
305        "fp" => interactive_module_define_fpmodule(&mut output_json, p)?,
306        _ => unreachable!(),
307    }
308    println!("{output_json}");
309    Ok(())
