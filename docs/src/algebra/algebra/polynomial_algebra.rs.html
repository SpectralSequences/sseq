62pub trait PolynomialAlgebra: std::fmt::Display + Sized + Send + Sync + 'static {
63    fn prime(&self) -> ValidPrime;
64
65    fn polynomial_monomials(&self) -> &TruncatedPolynomialMonomialBasis;
66    fn exterior_monomials(&self) -> &TruncatedPolynomialMonomialBasis;
67
68    fn min_degree(&self) -> i32 {
69        0
70    }
71
72    fn polynomial_generators_in_degree(&self, degree: i32) -> usize;
73    fn exterior_generators_in_degree(&self, degree: i32) -> usize;
74    fn repr_poly_generator(&self, degree: i32, _index: usize) -> (String, u32);
75    fn repr_ext_generator(&self, degree: i32, _index: usize) -> String;
76
77    fn basis_table(&self) -> &OnceVec<PolynomialAlgebraTableEntry>;
78
79    fn frobenius_on_generator(&self, degree: i32, index: usize) -> Option<usize>;
80    fn compute_generating_set(&self, degree: i32);
81
82    fn compute_basis_step(&self, degree: i32) {
83        assert!(degree as usize == self.basis_table().len());
84        let num_poly_gens = self.polynomial_generators_in_degree(degree);
85        let num_ext_gens = self.exterior_generators_in_degree(degree);
86        let poly_parts = self.polynomial_monomials();
87        let ext_parts = self.exterior_monomials();
88        if degree > 0 {
89            poly_parts.add_gens_and_calculate_parts(degree, num_poly_gens);
90            ext_parts.add_gens_and_calculate_parts(degree, num_ext_gens);
91        }
92        let mut table = PolynomialAlgebraTableEntry::new();
93        for poly_deg in 0..=degree {
94            let ext_deg = degree - poly_deg;
95            for p in poly_parts.parts(poly_deg) {
96                for e in ext_parts.parts(ext_deg) {
97                    let index = table.index_to_monomial.len();
98                    let mut m = PolynomialAlgebraMonomial {
99                        degree,
100                        poly: p.clone(),
101                        ext: e.clone(),
102                        valid: true,
103                    };
104                    self.set_monomial_degree(&mut m, degree);
105                    table.monomial_to_index.insert(m.clone(), index);
106                    table.index_to_monomial.push(m);
107                }
108            }
109        }
110        self.basis_table().push(table);
111    }
112
113    fn monomial_to_index(&self, monomial: &PolynomialAlgebraMonomial) -> usize {
114        self.basis_table()[monomial.degree as usize]
115            .monomial_to_index
116            .get(monomial)
117            .copied()
118            .unwrap_or_else(|| panic!("Didn't find monomial: {monomial}"))
119    }
120
121    fn index_to_monomial(&self, degree: i32, index: usize) -> &PolynomialAlgebraMonomial {
122        &self.basis_table()[degree as usize].index_to_monomial[index]
123    }
124
125    fn frobenius_monomial(&self, target: &mut FpVector, source: &FpVector) {
126        let p = self.prime().as_i32();
127        for (i, c) in source.iter_nonzero() {
128            let (gen_degree, gen_index) = self.polynomial_monomials().internal_idx_to_gen_deg(i);
129            let frob = self.frobenius_on_generator(gen_degree, gen_index);
130            if let Some(e) = frob {
131                let out_idx = self
132                    .polynomial_monomials()
133                    .gen_deg_idx_to_internal_idx(p * gen_degree, e);
134                target.add_basis_element(out_idx, c);
135            }
136        }
137    }
138
139    fn multiply_monomials(
140        &self,
141        target: &mut PolynomialAlgebraMonomial,
142        source: &PolynomialAlgebraMonomial,
143    ) -> Option<u32> {
144        let minus_one = self.prime() - 1;
145        self.set_monomial_degree(target, target.degree + source.degree);
146        let mut temp_source_ext = source.ext.clone();
147        temp_source_ext.set_scratch_vector_size(target.ext.len());
148        // If we made sign_rule handle vectors of different lengths, we could avoid cloning ext here.
149        let coeff = if target.ext.sign_rule(&temp_source_ext) {
150            minus_one
151        } else {
152            1
153        };
154        target.ext.add_truncate(&temp_source_ext, 1)?;
155
156        let mut carry_vec = [FpVector::new(self.prime(), target.poly.len())];
157        let mut source_vec = source.poly.clone();
158        source_vec.set_scratch_vector_size(target.poly.len());
159        let mut carry_q = true;
160        while carry_q {
161            carry_q = target.poly.add_carry(&source_vec, 1, &mut carry_vec);
162            if carry_q {
163                source_vec.set_to_zero();
164                self.frobenius_monomial(&mut source_vec, &carry_vec[0]);
165                carry_vec[0].set_to_zero();
166            }
167        }
168        Some(coeff)
169    }
170
171    fn multiply_polynomials(
172        &self,
173        target: &mut FpVector,
174        coeff: u32,
175        left_degree: i32,
176        left: &FpVector,
177        right_degree: i32,
178        right: &FpVector,
179    ) {
180        let p = self.prime();
181        target.set_scratch_vector_size(self.dimension(left_degree + right_degree));
182        for (left_idx, left_entry) in left.iter_nonzero() {
183            for (right_idx, right_entry) in right.iter_nonzero() {
184                let mut target_mono = self.index_to_monomial(left_degree, left_idx).clone();
185                let source_mono = self.index_to_monomial(right_degree, right_idx);
186                let nonzero_result = self.multiply_monomials(&mut target_mono, source_mono);
187                if let Some(c) = nonzero_result {
188                    let idx = self.monomial_to_index(&target_mono);
189                    target.add_basis_element(idx, (left_entry * right_entry * c * coeff) % p);
190                }
191            }
192        }
193    }
194
195    fn multiply_polynomial_by_monomial(
196        &self,
197        target: &mut FpVector,
198        coeff: u32,
199        left_degree: i32,
200        left: &FpVector,
201        right_mono: &PolynomialAlgebraMonomial,
202    ) {
203        let p = self.prime();
204        target.extend_len(self.dimension(left_degree + right_mono.degree));
205        for (left_idx, left_entry) in left.iter_nonzero() {
206            let mut target_mono = self.index_to_monomial(left_degree, left_idx).clone(); // Could reduce cloning a bit but probably best not to worry.
207            let nonzero_result = self.multiply_monomials(&mut target_mono, right_mono);
208            if let Some(c) = nonzero_result {
209                let idx = self.monomial_to_index(&target_mono);
210                target.add_basis_element(idx, (left_entry * c * coeff) % p);
211            }
212        }
213    }

215    // At p=2 this is redundant but at odd primes one must worry about signs.
216    fn multiply_monomial_by_polynomial(
217        &self,
218        target: &mut FpVector,
219        coeff: u32,
220        left_mono: &PolynomialAlgebraMonomial,
221        right_degree: i32,
222        right: &FpVector,
223    ) {
224        let p = self.prime();
225        target.extend_len(self.dimension(right_degree + left_mono.degree));
226        for (right_idx, right_entry) in right.iter_nonzero() {
227            let mut target_mono = left_mono.clone(); // Could reduce cloning a bit but probably best not to worry.
228            let right_mono = self.index_to_monomial(right_degree, right_idx);
229            let nonzero_result = self.multiply_monomials(&mut target_mono, right_mono);
230            if let Some(c) = nonzero_result {
231                let idx = self.monomial_to_index(&target_mono);
232                target.add_basis_element(idx, (right_entry * c * coeff) % p);
233            }
234        }
235    }

237    fn set_monomial_degree(&self, mono: &mut PolynomialAlgebraMonomial, degree: i32) {
238        mono.degree = degree;
239        mono.ext.set_scratch_vector_size(
240            self.exterior_monomials()
241                .generators_up_to_degree(mono.degree),
242        );
243        mono.poly.set_scratch_vector_size(
244            self.polynomial_monomials()
245                .generators_up_to_degree(mono.degree),
246        );
247    }

249    fn max_computed_degree(&self) -> i32 {
250        self.basis_table().len() as i32 - 1
251    }

254impl<A: PolynomialAlgebra> Algebra for A {
255    fn prime(&self) -> ValidPrime {
256        self.prime()
257    }
258
259    fn compute_basis(&self, degree: i32) {
260        self.compute_generating_set(degree);
261        for i in self.max_computed_degree() + 1..=degree {
262            self.compute_basis_step(i);
263        }
264    }
265
266    fn dimension(&self, degree: i32) -> usize {
267        if degree < 0 {
268            0
269        } else {
270            self.basis_table()[degree as usize].index_to_monomial.len()
271        }
272    }
273
274    fn basis_element_to_string(&self, degree: i32, index: usize) -> String {
275        let mono = self.index_to_monomial(degree, index);
276        let mut exp_map = HashMap::default();
277        for (i, e) in mono.poly.iter_nonzero() {
278            let (gen_deg, gen_idx) = self.polynomial_monomials().internal_idx_to_gen_deg(i);
279            let (var, var_exp) = self.repr_poly_generator(gen_deg, gen_idx);
280            let entry = exp_map.entry(var).or_insert((0, gen_deg / var_exp as i32));
281            entry.0 += (e * var_exp) as i32;
282        }
283        let result = exp_map
284            .iter()
285            .sorted_by_key(|(_, &(_, gen_deg))| gen_deg)
286            .map(|(var, &(var_exp, gen_deg))| {
287                let pow = if var_exp > 1 {
288                    format!("^{{{var_exp}}}")
289                } else {
290                    "".to_string()
291                };
292                let s = format!("{var}{pow}");
293                (s, gen_deg)
294            })
295            .merge_by(
296                mono.ext.iter_nonzero().map(|(i, _)| {
297                    let (gen_deg, gen_idx) = self.exterior_monomials().internal_idx_to_gen_deg(i);
298                    let var = self.repr_ext_generator(gen_deg, gen_idx);
299                    (var, gen_deg)
300                }),
301                |x, y| x.1 < y.1,
302            )
303            .map(|(v, _gen_deg)| v)
304            .join(" ");
305        if result.is_empty() {
306            "1".to_string()
307        } else {
308            result
309        }
310    }

316    fn multiply_basis_elements(
317        &self,
318        mut result: FpSliceMut,
319        coeff: u32,
320        left_degree: i32,
321        left_idx: usize,
322        right_degree: i32,
323        right_idx: usize,
324    ) {
325        if coeff == 0 {
326            return;
327        }
328        let mut target = self.index_to_monomial(left_degree, left_idx).clone();
329        let source = self.index_to_monomial(right_degree, right_idx);
330        if self.multiply_monomials(&mut target, source).is_some() {
331            let idx = self.monomial_to_index(&target);
332            result.add_basis_element(idx, coeff);
333        }
334    }
