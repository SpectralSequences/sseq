32pub struct MilnorProfile {
33    /// If `true`, unspecified p_part entries will be 0. Otherwise they will be infinity.
34    pub truncated: bool,
35    /// A bitmask indicating which of the Q_k we want to include (1 = include). Defaults to `!0`.
36    /// This is only relevant at odd primes.
37    #[serde(default = "q_part_default")]
38    pub q_part: u32,
39    /// The profile function for the Q part.
40    #[serde(default)]
41    pub p_part: PPart,
42}
43
44impl MilnorProfile {
45    pub fn is_trivial(&self) -> bool {
46        !self.truncated && self.q_part == !0 && self.p_part.is_empty()
47    }
48
49    pub fn get_p_part(&self, i: usize) -> PPartEntry {
50        self.p_part
51            .get(i)
52            .copied()
53            .unwrap_or(if self.truncated { 0 } else { PPartEntry::MAX })
54    }
55
56    /// Checks whether the profile function is valid
57    pub fn is_valid(&self) -> bool {
58        for (i, &hi) in self.p_part.iter().enumerate() {
59            for (j, &hj) in self.p_part.iter().enumerate().skip(i + 1) {
60                if hi > (j - i) as PPartEntry + hj && self.p_part[j - i] > hj {
61                    return false;
62                }
63            }
64        }
65        if self.truncated {
66            let len = self.p_part.len();
67            for (i, &hi) in self.p_part.iter().enumerate() {
68                if hi > (len - i) as PPartEntry {
69                    return false;
70                }
71            }
72        }
73        if self.q_part != !0 {
74            for i in BitflagIterator::set_bit_iterator(!self.q_part as u64) {
75                for j in 0..i {
76                    if (self.q_part >> j) & 1 == 1 && self.get_p_part(i - j - 1) > j as PPartEntry {
77                        return false;
78                    }
79                }
80            }
81        }
82        true
83    }
84
85    /// Whether the profile is that of A(n). This is relevant since A(n) is generated by P(p^n) and
86    /// β but a general subalgebra is not.
87    pub fn is_an(&self, generic: bool) -> bool {
88        if *self == Self::default() {
89            return true;
90        }
91        if !self.truncated {
92            return false;
93        }
94        if self.p_part.len() != self.p_part[0] as usize {
95            return false;
96        }
97        if generic && self.q_part != (1 << (self.p_part.len() + 1)) - 1 {
98            return false;
99        }
100        true
101    }
102}

124impl MilnorBasisElement {
125    fn from_p(p_part: PPart, degree: i32) -> Self {
126        Self {
127            q_part: 0,
128            p_part,
129            degree,
130        }
131    }
132
133    fn excess(&self, p: ValidPrime) -> u32 {
134        if p == 2 {
135            self.p_part.iter().sum::<PPartEntry>()
136        } else {
137            self.q_part.count_ones() + 2 * self.p_part.iter().sum::<PPartEntry>()
138        }
139    }
140
141    pub fn clone_into(&self, other: &mut Self) {
142        other.q_part = self.q_part;
143        other.degree = self.degree;
144        other.p_part.clear();
145        other.p_part.extend_from_slice(&self.p_part);
146    }
147
148    /// Update the degree component to the correct degree
149    pub fn compute_degree(&mut self, p: ValidPrime) {
150        let q = if p == 2 { 1 } else { 2 * (p.as_i32() - 1) };
151        let xi_degrees = combinatorics::xi_degrees(p);
152        let tau_degrees = combinatorics::tau_degrees(p);
153
154        self.degree = q * std::iter::zip(xi_degrees, &self.p_part)
155            .map(|(&a, &b)| a * b as i32)
156            .sum::<i32>()
157            + BitflagIterator::set_bit_iterator(self.q_part as u64)
158                .map(|k| tau_degrees[k])
159                .sum::<i32>();
160    }
161}

198            }
199            write!(f, "P({})", self.p_part.iter().format(", "))?;
200        }
201        Ok(())

225#[cfg(not(feature = "odd-primes"))]
226impl<V> MilnorHashMap<V> {
227    /// Encode a [`MilnorBasisElement`] of a known degree into a `u64`. This is achieved by packing
228    /// the PPart into a single `u64`, where we omit the first entry since it can be derived from
229    /// the degree. This currently supports elements up to degree 2^9 * 3 = 1536.
230    fn code(x: &MilnorBasisElement) -> u64 {
231        let mut counter = 0;
232        let mut shift = 0;
233        for (idx, &entry) in x.p_part.iter().skip(1).enumerate() {
234            counter += (entry as u64) << shift;
235            shift += 9 - idx;
236        }
237        counter
238    }
239
240    fn reserve(&mut self, additional: usize) {
241        self.inner.reserve(additional);
242    }
243
244    fn insert(&mut self, k: MilnorBasisElement, v: V) {
245        if self.degree == -1 {
246            self.degree = k.degree;
247        }
248        assert_eq!(k.degree, self.degree);
249        assert!(
250            self.inner.insert(Self::code(&k), v).is_none(),
251            "Duplicate entry for {k}"
252        );
253    }
254
255    fn get(&self, k: &MilnorBasisElement) -> Option<&V> {
256        assert_eq!(k.degree, self.degree);
257        self.inner.get(&Self::code(k))
258    }
259}

378    fn default_filtration_one_products(&self) -> Vec<(String, i32, usize)> {
379        let mut products = Vec::with_capacity(4);
380        let max_degree;
381        if self.generic() {
382            if self.profile.q_part & 1 != 0 {
383                products.push((
384                    "a_0".to_string(),
385                    MilnorBasisElement {
386                        degree: 1,
387                        q_part: 1,
388                        p_part: vec![],
389                    },
390                ));
391            }
392            if (self.profile.p_part.is_empty() && !self.profile.truncated)
393                || (!self.profile.p_part.is_empty() && self.profile.p_part[0] > 0)
394            {
395                products.push((
396                    "h_0".to_string(),
397                    MilnorBasisElement {
398                        degree: (2 * self.prime() - 2) as i32,
399                        q_part: 0,
400                        p_part: vec![1],
401                    },
402                ));
403            }
404            max_degree = (2 * self.prime() - 2) as i32;
405        } else {
406            let mut max = 4;
407            if !self.profile.p_part.is_empty() {
408                max = std::cmp::min(4, self.profile.p_part[0]);
409            } else if self.profile.truncated {
410                max = 0;
411            }
412            for i in 0..max {
413                let degree = 1 << i; // degree is 2^hi
414                products.push((
415                    format!("h_{i}"),
416                    MilnorBasisElement {
417                        degree,
418                        q_part: 0,
419                        p_part: vec![1 << i],
420                    },
421                ));
422            }
423            max_degree = 1 << 3;
424        }
425        self.compute_basis(max_degree + 1);
426
427        products
428            .into_iter()
429            .map(|(name, b)| (name, b.degree, self.basis_element_to_index(&b)))
430            .collect()
431    }

433    fn compute_basis(&self, max_degree: i32) {
434        self.compute_ppart(max_degree);
435
436        if self.generic() {
437            self.generate_basis_generic(max_degree);
438        } else {
439            self.generate_basis_2(max_degree);
440        }
441
442        // Populate hash map
443        self.basis_element_to_index_map
444            .extend(max_degree as usize, |d| {
445                let basis = &self.basis_table[d];
446                let mut map = MilnorHashMap::default();
447                map.reserve(basis.len());
448                for (i, b) in basis.iter().enumerate() {
449                    map.insert(b.clone(), i);
450                }
451                map
452            });
453
454        #[cfg(feature = "cache-multiplication")]
455        {
456            self.multiplication_table
457                .extend(max_degree as usize, |_| OnceVec::new());
458
459            for d in 0..=max_degree as usize {
460                self.multiplication_table[d].extend(max_degree as usize - d, |e| {
461                    (0..self.dimension(d as i32))
462                        .map(|i| {
463                            (0..self.dimension(e as i32))
464                                .map(|j| {
465                                    let mut res =
466                                        FpVector::new(self.prime(), self.dimension((d + e) as i32));
467                                    self.multiply(
468                                        res.as_slice_mut(),
469                                        1,
470                                        &self.basis_table[d][i],
471                                        &self.basis_table[e][j],
472                                    );
473                                    res
474                                })
475                                .collect::<Vec<_>>()
476                        })
477                        .collect::<Vec<_>>()
478                });
479            }
480        }
481
482        if self.unstable_enabled {
483            self.generate_excess_table(max_degree);
484        }
485    }
486
487    fn dimension(&self, degree: i32) -> usize {
488        if degree < 0 {
489            return 0;
490        }
491        self.basis_table[degree as usize].len()

583    fn basis_element_from_string(&self, elt: &str) -> Option<(i32, usize)> {
584        use nom::{
585            branch::alt,
586            bytes::complete::tag,
587            character::complete::char,
588            combinator::{map, opt},
589            multi::{many0, separated_list1},
590            sequence::{preceded, tuple},
591        };
592
593        use crate::steenrod_parser::{brackets, digits, p_or_sq};
594
595        let p = self.prime();
596
597        let mut parser = alt((
598            map(char('1'), |_| (0, 0)),
599            map(char('b'), |_| (1, 0)),
600            map(preceded(p_or_sq, digits), |i| self.beps_pn(0, i)),
601            map(
602                tuple((tag("P^"), digits, char('_'), digits)),
603                |(_, s, _, t)| {
604                    let entry = p.pow(s);
605                    let degree = entry as i32 * self.q() * combinatorics::xi_degrees(p)[t];
606                    let mut elt = MilnorBasisElement {
607                        degree,
608                        q_part: 0,
609                        p_part: vec![0; t],
610                    };
611                    elt.p_part[t - 1] = entry as PPartEntry;
612                    self.compute_basis(degree);
613                    (degree, self.basis_element_to_index(&elt))
614                },
615            ),
616            map(
617                tuple((
618                    many0(preceded(tag("Q_"), digits::<u32>)),
619                    opt(preceded(
620                        char('P'),
621                        brackets(separated_list1(char(','), digits)),
622                    )),
623                )),
624                |(q_list, p_list)| {
625                    let q_part = q_list.into_iter().fold(0, |acc, q| acc + (1 << q));
626                    let mut elt = MilnorBasisElement {
627                        degree: 0,
628                        q_part,
629                        p_part: p_list.unwrap_or_default(),
630                    };
631                    elt.compute_degree(p);
632                    self.compute_basis(elt.degree);
633
634                    (elt.degree, self.basis_element_to_index(&elt))
635                },
636            ),
637        ));
638
639        if let Ok(("", res)) = parser(elt) {
640            Some(res)
641        } else {
642            None
643        }
644    }

676impl GeneratedAlgebra for MilnorAlgebra {
677    fn generator_to_string(&self, degree: i32, idx: usize) -> String {
678        if self.generic() {
679            if degree == 1 {
680                return "b".to_string();
681            }
682            let elt = self.basis_element_from_index(degree, idx);
683            let len = elt.p_part.len();
684            if elt.q_part != 0 {
685                elt.to_string()
686            } else if len == 1 {
687                format!("P{}", degree / self.q())
688            } else {
689                format!(
690                    "P^{}_{}",
691                    degree / (self.q() * combinatorics::xi_degrees(self.prime())[len - 1]),
692                    len
693                )
694            }
695        } else {
696            let elt = self.basis_element_from_index(degree, idx);
697            let len = elt.p_part.len();
698            if len == 1 {
699                format!("Sq{degree}")
700            } else {
701                format!(
702                    "P^{}_{}",
703                    degree / (combinatorics::xi_degrees(self.prime())[len - 1]),
704                    len
705                )
706            }
707        }
708    }
709
710    fn generators(&self, degree: i32) -> Vec<usize> {
711        if degree == 0 {
712            return vec![];
713        } else if degree == 1 {
714            return vec![0]; // Q_0
715        }
716
717        let p = self.prime();
718
719        // Check for the Q_k
720        if self.generic() && degree % 2 == 1 {
721            if self.profile.is_an(true) {
722                return vec![];
723            }
724
725            // If this is 2p^k - 1, then return Q_k
726            if let (k, 2) = factor_pk(p, degree as u32 + 1) {
727                let q_part = 1 << k;
728                if self.profile.q_part & q_part != 0 {
729                    return vec![self.basis_element_to_index(&MilnorBasisElement {
730                        degree,
731                        q_part,
732                        p_part: vec![],
733                    })];
734                }
735            }
736            return vec![];
737        }
738
739        if self.profile.is_an(self.generic()) {
740            // Look for P(p^k), which has degree p^k q.
741            let q = self.q() as u32;
742            if !(degree as u32).is_multiple_of(q) {
743                return vec![];
744            }
745            if let (k, 1) = factor_pk(p, degree as u32 / q) {
746                if (k as PPartEntry) < self.profile.get_p_part(0) {
747                    return vec![self.basis_element_to_index(&MilnorBasisElement {
748                        degree,
749                        q_part: 0,
750                        p_part: vec![(degree as u32 / q) as PPartEntry],
751                    })];
752                }
753            }
754            vec![]
755        } else {
756            // Look for P(0, ..., 0, p^k), which has degree (2p^j - 2) p^k.
757            let (k, rem) = factor_pk(p, degree as u32);
758
759            let reduced = if self.generic() {
760                // rem must be even because degree is even
761                (rem + 2) / 2
762            } else {
763                rem + 1
764            };
765
766            if let (j, 1) = factor_pk(p, reduced) {
767                if self.profile.get_p_part(j as usize - 1) <= k as PPartEntry {
768                    return vec![];
769                }
770                let mut p_part = vec![0; j as usize];
771                p_part[j as usize - 1] = p.pow(k) as PPartEntry;
772                return vec![self.basis_element_to_index(&MilnorBasisElement {
773                    degree,
774                    q_part: 0,
775                    p_part,
776                })];
777            }
778            vec![]
779        }
780    }

782    fn decompose_basis_element(
783        &self,
784        degree: i32,
785        idx: usize,
786    ) -> Vec<(u32, (i32, usize), (i32, usize))> {
787        let basis = self.basis_element_from_index(degree, idx);
788        // If qpart = 0, return self
789        if basis.q_part == 0 {
790            self.decompose_basis_element_ppart(degree, idx)
791        } else {
792            self.decompose_basis_element_qpart(degree, idx)
793        }
794    }

796    fn generating_relations(&self, degree: i32) -> Vec<Vec<(u32, (i32, usize), (i32, usize))>> {
797        if self.generic() && degree == 2 {
798            // beta^2 = 0 is an edge case
799            return vec![vec![(1, (1, 0), (1, 0))]];
800        }
801        let p = self.prime();
802        let inadmissible_pairs = combinatorics::inadmissible_pairs(p, self.generic(), degree);
803        let mut result = Vec::new();
804        for (x, b, y) in inadmissible_pairs {
805            let mut relation = Vec::new();
806            // Adem relation. Sometimes these don't exist because of profiles. Then just ignore it.
807            (|| {
808                let (first_degree, first_index) = self.try_beps_pn(0, x as PPartEntry)?;
809                let (second_degree, second_index) = self.try_beps_pn(b, y as PPartEntry)?;
810                relation.push((
811                    p - 1,
812                    (first_degree, first_index),
813                    (second_degree, second_index),
814                ));
815                for e1 in 0..=b {
816                    let e2 = b - e1;
817                    // e1 and e2 determine where a bockstein shows up.
818                    // e1 determines whether a bockstein shows up in front
819                    // e2 determines whether a bockstein shows up in middle
820                    // So our output term looks like b^{e1} P^{x+y-j} b^{e2} P^{j}
821                    for j in 0..=x / p {
822                        let c = combinatorics::adem_relation_coefficient(p, x, y, j, e1, e2);
823                        if c == 0 {
824                            continue;
825                        }
826                        if j == 0 {
827                            relation.push((
828                                c,
829                                self.try_beps_pn(e1, (x + y) as PPartEntry)?,
830                                (e2 as i32, 0),
831                            ));
832                            continue;
833                        }
834                        let first_sq = self.try_beps_pn(e1, (x + y - j) as PPartEntry)?;
835                        let second_sq = self.try_beps_pn(e2, j as PPartEntry)?;
836                        relation.push((c, first_sq, second_sq));
837                    }
838                }
839                result.push(relation);
840                Some(())
841            })();
842        }
843        result
844    }
845}

847// Compute basis functions
848impl MilnorAlgebra {
849    fn compute_ppart(&self, max_degree: i32) {
850        self.ppart_table.extend(0, |_| vec![Vec::new()]);
851
852        let p = self.prime().as_i32();
853        let q = if p == 2 { 1 } else { 2 * p - 2 };
854        let new_deg = max_degree / q;
855
856        let xi_degrees = combinatorics::xi_degrees(self.prime());
857        let mut profile_list = Vec::with_capacity(xi_degrees.len());
858        for i in 0..xi_degrees.len() {
859            if i < self.profile.p_part.len() {
860                profile_list.push((self.prime().pow(self.profile.p_part[i]) - 1) as PPartEntry);
861            } else if self.profile.truncated {
862                profile_list.push(0);
863            } else {
864                profile_list.push(PPartEntry::MAX);
865            }
866        }
867
868        self.ppart_table.extend(new_deg as usize, |d| {
869            let d = d as i32;
870            let mut new_row = Vec::new(); // Improve this
871            for i in 0..xi_degrees.len() {
872                if xi_degrees[i] > d {
873                    break;
874                }
875                if profile_list[i] == 0 {
876                    continue;
877                }
878
879                let rem = (d - xi_degrees[i]) as usize;
880                for old in &self.ppart_table[rem] {
881                    // ppart_table[rem] is arranged in increasing order of highest
882                    // xi_i. If we get something too large, we may abort;
883                    if old.len() > i + 1 {
884                        break;
885                    }
886                    if old.len() == i + 1 && old[i] == profile_list[i] {
887                        continue;
888                    }
889                    let mut new = old.clone();
890                    if new.len() < i + 1 {
891                        new.resize(i + 1, 0);
892                    }
893                    new[i] += 1;
894                    new_row.push(new);
895                }
896            }
897            new_row
898        });
899    }
900
901    fn generate_basis_generic(&self, max_degree: i32) {
902        let q = 2 * self.prime() - 2;
903        let tau_degrees = combinatorics::tau_degrees(self.prime());
904
905        self.basis_table.extend(max_degree as usize, |d| {
906            let mut table = Vec::new();
907            let residue = d as u32 % q;
908
909            for q_part in 0u32.. {
910                if q_part.count_ones() % q != residue {
911                    continue;
912                }
913
914                let mut q_degree = 0;
915                let mut bs = q_part;
916                for &entry in tau_degrees {
917                    q_degree += entry * (bs & 1) as i32;
918                    bs >>= 1;
919                    if bs == 0 {
920                        break;
921                    }
922                }
923
924                if q_degree > d as i32 {
925                    break;
926                }
927
928                if q_part & !self.profile.q_part != 0 {
929                    continue;
930                }
931
932                table.extend(
933                    self.ppart_table[(d - q_degree as usize) / q as usize]
934                        .iter()
935                        .map(|p_part| MilnorBasisElement {
936                            p_part: p_part.clone(),
937                            q_part,
938                            degree: d as i32,
939                        }),
940                );
941            }
942            if self.unstable_enabled {
943                table.sort_by_cached_key(|e| e.excess(self.p));
944            }
945            table
946        });
947    }
948
949    fn generate_basis_2(&self, max_degree: i32) {
950        self.basis_table.extend(max_degree as usize, |d| {
951            let mut table: Vec<_> = self.ppart_table[d]
952                .iter()
953                .map(|p| MilnorBasisElement::from_p(p.clone(), d as i32))
954                .collect();
955            if self.unstable_enabled {
956                table.sort_by_cached_key(|e| e.excess(fp::prime::TWO));
957            }
958            table
959        });
960    }
961
962    fn generate_excess_table(&self, max_degree: i32) {
963        let p = self.prime();
964        self.excess_table.extend(max_degree as usize, |n| {
965            let mut new_entry = Vec::with_capacity(n);
966            let mut cur_excess = 0;
967            for (i, elt) in self.basis_table[n].iter().enumerate() {
968                let excess = elt.excess(p);
969                for _ in cur_excess..excess {
970                    new_entry.push(i);
971                }
972                cur_excess = excess;
973            }
974            let dim = self.dimension(n as i32);
975            for _ in cur_excess..n as u32 {
976                new_entry.push(dim);
977            }
978            new_entry
979        });
980    }

983// Multiplication logic
984impl MilnorAlgebra {
985    fn try_beps_pn(&self, e: u32, x: PPartEntry) -> Option<(i32, usize)> {
986        let q = self.q() as u32;
987        let degree = (q * x + e) as i32;
988        self.compute_basis(degree);
989        self.try_basis_element_to_index(&MilnorBasisElement {
990            degree,
991            q_part: e,
992            p_part: vec![x as PPartEntry],
993        })
994        .map(|index| (degree, index))
995    }
996
997    /// Return the degree and index of $Q_1^e P(x)$.
998    pub fn beps_pn(&self, e: u32, x: PPartEntry) -> (i32, usize) {
999        self.try_beps_pn(e, x).unwrap()
1000    }
1001
1002    fn multiply_qpart(&self, m1: &MilnorBasisElement, f: u32) -> Vec<(u32, MilnorBasisElement)> {
1003        let mut new_result: Vec<(u32, MilnorBasisElement)> = vec![(1, m1.clone())];
1004        let mut old_result: Vec<(u32, MilnorBasisElement)> = Vec::new();
1005
1006        for k in BitflagIterator::set_bit_iterator(f as u64) {
1007            let k = k as u32;
1008            let pk = self.p.pow(k) as PPartEntry;
1009            std::mem::swap(&mut new_result, &mut old_result);
1010            new_result.clear();
1011
1012            // We implement the formula
1013            // P(R) Q_k = Q_k P^R + Q_{k+1} P(R - p^k e_1) + Q_{k+2} P(R - p^k e_2) +
1014            // ... + Q_{k + i} P(R - p^k e_i) + ...
1015            // where e_i is the vector with value 1 in entry i and 0 otherwise (in the above
1016            // formula, the first xi is xi_1, hence the offset below). If R - p^k e_i has a
1017            // negative entry, the term is 0.
1018            //
1019            // We also use the fact that Q_k Q_j = -Q_j Q_k
1020            for (coef, term) in &old_result {
1021                for i in 0..=term.p_part.len() {
1022                    // If there is already Q_{k+i} on the other side, the result is 0
1023                    if term.q_part & (1 << (k + i as u32)) != 0 {
1024                        continue;
1025                    }
1026                    // Check if R - p^k e_i < 0. Only do this from the first term onwards.
1027                    if i > 0 && term.p_part[i - 1] < pk {
1028                        continue;
1029                    }
1030
1031                    let mut new_p = term.p_part.clone();
1032                    if i > 0 {
1033                        new_p[i - 1] -= pk;
1034                    }
1035
1036                    // Now calculate the number of Q's we are moving past
1037                    let larger_q = (term.q_part >> (k + i as u32 + 1)).count_ones();
1038
1039                    // If new_p ends with 0, drop them
1040                    while let Some(0) = new_p.last() {
1041                        new_p.pop();
1042                    }
1043                    // Now put everything together
1044                    let m = MilnorBasisElement {
1045                        p_part: new_p,
1046                        q_part: term.q_part | (1 << (k + i as u32)),
1047                        degree: 0, // we don't really care about the degree here. The final degree of the whole calculation is known a priori
1048                    };
1049                    let c = if larger_q.is_multiple_of(2) {
1050                        *coef
1051                    } else {
1052                        *coef * (self.prime() - 1)
1053                    };
1054
1055                    new_result.push((c, m));
1056                }
1057            }
1058        }
1059        new_result
1060    }

1062    pub fn multiply(
1063        &self,
1064        res: FpSliceMut,
1065        coef: u32,
1066        m1: &MilnorBasisElement,
1067        m2: &MilnorBasisElement,
1068    ) {
1069        PPartAllocation::with_local(|allocation| {
1070            self.multiply_with_allocation(res, coef, m1, m2, i32::MAX, allocation)
1071        });
1072    }
1073
1074    pub fn multiply_with_allocation(
1075        &self,
1076        mut res: FpSliceMut,
1077        coef: u32,
1078        m1: &MilnorBasisElement,
1079        m2: &MilnorBasisElement,
1080        excess: i32,
1081        mut allocation: PPartAllocation,
1082    ) -> PPartAllocation {
1083        let target_deg = m1.degree + m2.degree;
1084        if self.generic() {
1085            let m1f = self.multiply_qpart(m1, m2.q_part);
1086            for (cc, basis) in m1f {
1087                let mut multiplier = PPartMultiplier::<false>::new_from_allocation(
1088                    self.prime(),
1089                    &basis.p_part,
1090                    &m2.p_part,
1091                    allocation,
1092                    basis.q_part,
1093                    target_deg,
1094                );
1095
1096                while let Some(c) = multiplier.next() {
1097                    let idx = self.basis_element_to_index(&multiplier.ans);
1098                    if idx < self.dimension_unstable(target_deg, excess) {
1099                        res.add_basis_element(idx, c * cc * coef);
1100                    }
1101                }
1102                allocation = multiplier.into_allocation()
1103            }
1104        } else {
1105            let mut multiplier = PPartMultiplier::<false>::new_from_allocation(
1106                self.prime(),
1107                &m1.p_part,
1108                &m2.p_part,
1109                allocation,
1110                0,
1111                target_deg,
1112            );
1113
1114            while let Some(c) = multiplier.next() {
1115                let idx = self.basis_element_to_index(&multiplier.ans);
1116                if idx < self.dimension_unstable(target_deg, excess) {
1117                    res.add_basis_element(idx, c * coef);
1118                }
1119            }
1120            allocation = multiplier.into_allocation()
1121        }
1122        allocation
1123    }

1125    pub fn multiply_basis_by_element(
1126        &self,
1127        res: FpSliceMut,
1128        coef: u32,
1129        m1: &MilnorBasisElement,
1130        s_deg: i32,
1131        s: FpSlice,
1132    ) {
1133        PPartAllocation::with_local(|allocation| {
1134            self.multiply_basis_by_element_with_allocation(res, coef, m1, s_deg, s, allocation)
1135        });
1136    }

1138    fn multiply_basis_by_element_with_allocation(
1139        &self,
1140        mut res: FpSliceMut,

1142        m1: &MilnorBasisElement,
1143        s_deg: i32,

1145        mut allocation: PPartAllocation,

1149                res.copy(),
1150                coef * c,
1151                m1,
1152                self.basis_element_from_index(s_deg, i),
1153                i32::MAX,
1154                allocation,
1155            );
1156        }
1157        allocation
1158    }

1292        }

1294        #[allow(clippy::manual_memcpy)]

1296            M[0][k] = s[k - 1];
1297        }
1298
1299        let ans = MilnorBasisElement {
1300            q_part,
1301            p_part: allocation.p_part,

1327    /// This compute the first l > k such that (sum + l) choose l != 0 mod p, stopping if we reach
1328    /// max + 1. This is useful for incrementing the matrix.
1329    ///
1330    /// TODO: Improve odd prime performance
1331    fn next_val(&self, sum: PPartEntry, k: PPartEntry, max: PPartEntry) -> PPartEntry {
1332        match self.prime().as_u32() {
1333            2 => {
1334                if MOD4 {
1335                    // x.count_ones() + y.count_ones() - (x + y).count_ones() is the number of
1336                    // carries when adding x to y.
1337                    //
1338                    // The p-adic valuation of (n + r) choose r is the number of carries when
1339                    // adding r to n in base p.
1340                    (k + 1..max + 1)
1341                        .find(|&l| {
1342                            sum & l == 0
1343                                || (sum.count_ones() + l.count_ones()) - (sum + l).count_ones() == 1
1344                        })
1345                        .unwrap_or(max + 1)
1346                } else {
1347                    ((k | sum) + 1) & !sum
1348                }
1349            }
1350            _ => (k + 1..max + 1)
1351                .find(|&l| !PPartEntry::binomial_odd_is_zero(self.prime(), sum + l, l))
1352                .unwrap_or(max + 1),
1353        }
1354    }
1355
1356    /// We have a matrix of the form
1357    ///    | s₁  s₂  s₃ ...
1358    /// --------------------
1359    /// r₁ |
1360    /// r₂ |     x_{ij}
1361    /// r₃ |
1362    ///
1363    /// We think of ourselves as modifiying the center pieces x_{ij}, while the r_i's and s_j's are
1364    /// only there to ensure the x_{ij}'s don't get too big. The idea is to sweep through the
1365    /// matrix row by row, from top-to-bottom, and left-to-right. In each pass, we find the first
1366    /// entry that can be incremented. We then increment it and zero out all the entries that
1367    /// appear before it. This will give us all valid entries.
1368    fn update(&mut self) -> bool {
1369        for i in 1..self.rows {
1370            // total is sum x_{ij} p^j up to the jth column
1371            let mut total = self.M[i][0];
1372            let mut p_to_the_j = 1;
1373            for j in 1..self.cols {
1374                p_to_the_j *= self.prime().as_u32() as PPartEntry;
1375                if total < p_to_the_j {
1376                    // We don't have enough weight left in the entries above this one in the column to increment this cell.
1377                    // Add the weight from this cell to the total, we can use it to increment a cell lower down.
1378                    total += self.M[i][j] * p_to_the_j;
1379                    continue;
1380                }
1381                let col_sum: PPartEntry = (0..i).map(|k| self.M[k][j]).sum();
1382                if col_sum == 0 {
1383                    total += self.M[i][j] * p_to_the_j;
1384                    continue;
1385                }
1386
1387                let max_inc = std::cmp::min(col_sum, total / p_to_the_j);
1388
1389                // Compute the sum of entries along the diagonal to the bottom-left
1390                let mut sum = 0;
1391                for c in (i + j + 1).saturating_sub(self.rows)..j {
1392                    sum += self.M[i + j - c][c];
1393                }
1394
1395                // Find the next possible value we can increment M[i][j] to without setting the
1396                // coefficient to 0. The coefficient is the multinomial coefficient of the
1397                // diagonal, and if the multinomial coefficient of any subset is zero, so is the
1398                // coefficient of the whole diagonal.
1399                let next_val = self.next_val(sum, self.M[i][j], max_inc + self.M[i][j]);
1400                let inc = next_val - self.M[i][j];
1401
1402                // The remaining obstacle to incrementing this entry is the column sum condition.
1403                // For this, we only need a non-zero entry in the column j above row i.
1404                if inc <= max_inc {
1405                    // If so, we found our next matrix.
1406                    for row in 1..i {
1407                        self.M[row][0] = self.r[row - 1];
1408                        for col in 1..self.cols {
1409                            self.M[0][col] += self.M[row][col];
1410                            self.M[row][col] = 0;
1411                        }
1412                    }
1413                    for col in 1..j {
1414                        self.M[0][col] += self.M[i][col];
1415                        self.M[i][col] = 0;
1416                    }
1417                    self.M[0][j] -= inc;
1418                    self.M[i][j] += inc;
1419                    self.M[i][0] = total - p_to_the_j * inc;
1420                    return true;
1421                }
1422                // All the cells above this one are zero so we didn't find our next matrix.
1423                // Add the weight from this cell to the total, we can use it to increment a cell lower down.
1424                total += self.M[i][j] * p_to_the_j;
1425            }
1426        }
1427        false
1428    }
1429}

1431impl<const MOD4: bool> Iterator for PPartMultiplier<'_, MOD4> {
1432    type Item = u32;

1434    fn next(&mut self) -> Option<u32> {
1435        let p = self.prime().as_u32() as PPartEntry;
1436        'outer: loop {
1437            self.ans.p_part.clear();
1438            let mut coef = 1;
1439
1440            if self.init {
1441                self.init = false;
1442                for i in 1..std::cmp::min(self.cols, self.rows) {
1443                    if MOD4 {
1444                        coef *= PPartEntry::binomial4(self.M[i][0] + self.M[0][i], self.M[0][i]);
1445                        coef %= 4;
1446                    } else {
1447                        coef *= PPartEntry::binomial(
1448                            self.prime(),
1449                            self.M[i][0] + self.M[0][i],
1450                            self.M[0][i],
1451                        );
1452                        coef %= p;
1453                    }
1454                    if coef == 0 {
1455                        continue 'outer;
1456                    }
1457                }
1458                self.ans
1459                    .p_part
1460                    .reserve(std::cmp::max(self.cols, self.rows) - 1);
1461                self.ans.p_part.extend(&self.M[0][1..self.cols]);
1462
1463                if self.rows > self.cols {
1464                    self.ans.p_part.resize(self.r.len(), 0);
1465                }
1466                self.ans
1467                    .p_part
1468                    .iter_mut()
1469                    .zip(self.r.iter())
1470                    .for_each(|(l, r)| *l += r);
1471
1472                // If new_p ends with 0, drop them
1473                while let Some(0) = self.ans.p_part.last() {
1474                    self.ans.p_part.pop();
1475                }
1476                return Some(coef);
1477            } else if self.update() {
1478                self.ans.p_part.reserve(self.diag_num);
1479                for diag_idx in 1..=self.diag_num {
1480                    let i_min = (diag_idx + 1).saturating_sub(self.cols);
1481                    let i_max = std::cmp::min(diag_idx + 1, self.rows);
1482                    let mut sum = 0;
1483
1484                    if self.prime() == 2 {
1485                        if MOD4 {
1486                            for i in i_min..i_max {
1487                                let entry = self.M[i][diag_idx - i];
1488                                sum += entry;
1489                                if coef.is_multiple_of(2) {
1490                                    coef *= PPartEntry::binomial2(sum, entry);
1491                                } else {
1492                                    coef *= PPartEntry::binomial4(sum, entry);
1493                                }
1494                                coef %= 4;
1495                                if coef == 0 {
1496                                    continue 'outer;
1497                                }
1498                            }
1499                        } else {
1500                            let mut or = 0;
1501                            for i in i_min..i_max {
1502                                sum += self.M[i][diag_idx - i];
1503                                or |= self.M[i][diag_idx - i];
1504                            }
1505                            if sum != or {
1506                                continue 'outer;
1507                            }
1508                        }
1509                    } else {
1510                        #[cfg(feature = "odd-primes")]
1511                        {
1512                            self.diagonal.clear();
1513                            for i in i_min..i_max {
1514                                self.diagonal.push(self.M[i][diag_idx - i]);
1515                                sum += self.M[i][diag_idx - i];
1516                            }
1517
1518                            coef *= PPartEntry::multinomial_odd(self.prime(), &mut self.diagonal);
1519                            coef %= p;
1520                            if coef == 0 {
1521                                continue 'outer;
1522                            }
1523                        }
1524                    }
1525                    self.ans.p_part.push(sum);
1526                }
1527                // If new_p ends with 0, drop them
1528                while let Some(0) = self.ans.p_part.last() {
1529                    self.ans.p_part.pop();
1530                }
1531
1532                return Some(coef);
1533            } else {
1534                return None;
1535            }
1536        }
1537    }

1540impl MilnorAlgebra {
1541    fn decompose_basis_element_qpart(
1542        &self,
1543        degree: i32,
1544        idx: usize,
1545    ) -> Vec<(u32, (i32, usize), (i32, usize))> {
1546        let basis = self.basis_element_from_index(degree, idx);
1547        // Look for left-most non-zero qpart
1548        let i = basis.q_part.trailing_zeros();
1549        // If the basis element is just Q_{k+1}, we decompose Q_{k+1} = P(p^k) Q_k - Q_k P(p^k).
1550        if basis.q_part == 1 << i && basis.p_part.is_empty() {
1551            let ppow = self.prime().pow(i - 1);
1552
1553            let q_degree = (2 * ppow - 1) as i32;
1554            let p_degree = (ppow * (2 * self.prime() - 2)) as i32;
1555
1556            let p_idx = self
1557                .basis_element_to_index(&MilnorBasisElement::from_p(
1558                    vec![ppow as PPartEntry],
1559                    p_degree,
1560                ))
1561                .to_owned();
1562
1563            let q_idx = self
1564                .basis_element_to_index(&MilnorBasisElement {
1565                    q_part: 1 << (i - 1),
1566                    p_part: Vec::new(),
1567                    degree: q_degree,
1568                })
1569                .to_owned();
1570
1571            vec![
1572                (1, (p_degree, p_idx), (q_degree, q_idx)),
1573                (self.prime() - 1, (q_degree, q_idx), (p_degree, p_idx)),
1574            ]
1575        } else {
1576            // Otherwise, separate out the first Q_k.
1577            let first_degree = combinatorics::tau_degrees(self.prime())[i as usize];
1578            let second_degree = degree - first_degree;
1579
1580            let first_idx = self.basis_element_to_index(&MilnorBasisElement {
1581                q_part: 1 << i,
1582                p_part: Vec::new(),
1583                degree: first_degree,
1584            });
1585
1586            let second_idx = self.basis_element_to_index(&MilnorBasisElement {
1587                q_part: basis.q_part ^ (1 << i),
1588                p_part: basis.p_part.clone(),
1589                degree: second_degree,
1590            });
1591
1592            vec![(1, (first_degree, first_idx), (second_degree, second_idx))]
1593        }
1594    }

1596    fn decompose_basis_element_ppart(
1597        &self,
1598        degree: i32,
1599        idx: usize,
1600    ) -> Vec<(u32, (i32, usize), (i32, usize))> {
1601        let p = self.prime();
1602
1603        // We define an ordering on the p parts as follows: we order each entry in reverse, and
1604        // then impose the reverse lexicographic ordering. Then for each non-generator P(R), `init`
1605        // is a partial decomposition such that the non-zero terms in the `init` product are all
1606        // greater than or equal to P(R) (and P(R) has non-zero coefficient in `init`). We can then
1607        // apply this algorithm recursively to decompose an element.
1608
1609        // result is the products we have added so far
1610        let mut result = Vec::new();
1611        // buffer is the products we are adding in the current iteration
1612        let mut buffer = Vec::new();
1613
1614        // out_vec is the remaining items we have to kill. We are done when this hits zero.
1615        let mut out_vec = FpVector::new(p, self.dimension(degree));
1616        out_vec.set_entry(idx, p - 1);
1617
1618        while let Some((idx, c)) = out_vec.iter_nonzero().next() {
1619            let b = self.basis_element_from_index(degree, idx);
1620            let len = b.p_part.len();
1621
1622            if b.p_part[0..len - 1].iter().all(|&x| x == 0) {
1623                // There is only one entry
1624                let entry = b.p_part[len - 1];
1625                let (k, m) = factor_pk(p, entry);
1626
1627                // This is a power of p
1628                if m == 1 {
1629                    if len == 1 || !self.profile.is_an(self.generic()) {
1630                        buffer.extend([(p - c, (degree, idx), (0, 0))]);
1631                    } else {
1632                        // Write this as [P(p^(len + k - 1)), P(0, .., 0, P^k)] plus higher order
1633                        // terms.
1634                        let l_entry = p.pow(len as u32 + k - 1) as PPartEntry;
1635                        let r_entry = p.pow(k) as PPartEntry;
1636
1637                        let l_degree = l_entry as i32 * self.q();
1638                        let l_index = self.basis_element_to_index(&MilnorBasisElement {
1639                            q_part: 0,
1640                            p_part: vec![l_entry],
1641                            degree: l_degree,
1642                        });
1643
1644                        let mut r_p_part = vec![0; len - 1];
1645                        r_p_part[len - 2] = r_entry;
1646                        let r_degree =
1647                            r_entry as i32 * combinatorics::xi_degrees(p)[len - 2] * self.q();
1648
1649                        let r_index = self.basis_element_to_index(&MilnorBasisElement {
1650                            q_part: 0,
1651                            p_part: r_p_part,
1652                            degree: r_degree,
1653                        });
1654                        buffer.extend(vec![
1655                            (p - c, (l_degree, l_index), (r_degree, r_index)),
1656                            (c, (r_degree, r_index), (l_degree, l_index)),
1657                        ])
1658                    }
1659                } else {
1660                    // This is not a power of p. Just subtract the lowest power of p.
1661                    let pk = p.pow(k) as PPartEntry;
1662                    let rem_entry = entry - pk;
1663
1664                    let entry_deg = combinatorics::xi_degrees(p)[len - 1] * self.q();
1665
1666                    let mut elt = MilnorBasisElement {
1667                        q_part: 0,
1668                        degree: 0,
1669                        p_part: vec![0; len],
1670                    };
1671
1672                    elt.p_part[len - 1] = pk;
1673                    elt.degree = entry_deg * elt.p_part[len - 1] as i32;
1674                    let first = (elt.degree, self.basis_element_to_index(&elt));
1675
1676                    elt.p_part[len - 1] = rem_entry;
1677                    elt.degree = entry_deg * elt.p_part[len - 1] as i32;
1678                    let second = (elt.degree, self.basis_element_to_index(&elt));
1679
1680                    let coef =
1681                        p - fp::prime::inverse(p, PPartEntry::binomial(p, pk + rem_entry, pk));
1682                    buffer.extend([(coef, first, second)])
1683                }
1684            } else {
1685                // There is more than one entry. Just separate out the last entry.
1686                let last_entry = b.p_part[len - 1];
1687                let last_deg = combinatorics::xi_degrees(p)[len - 1] * self.q() * last_entry as i32;
1688                let mut elt = MilnorBasisElement {
1689                    q_part: 0,
1690                    p_part: vec![0; len],
1691                    degree: last_deg,
1692                };
1693                elt.p_part[len - 1] = last_entry;
1694                let first = (elt.degree, self.basis_element_to_index(&elt));
1695
1696                elt.degree = degree - last_deg;
1697                elt.p_part.clear();
1698                elt.p_part.extend_from_slice(&b.p_part[0..len - 1]);
1699                while let Some(0) = elt.p_part.last() {
1700                    elt.p_part.pop();
1701                }
1702                let second = (elt.degree, self.basis_element_to_index(&elt));
1703                buffer.extend([(p - c, first, second)]);
1704            };
1705            for (c, first, second) in &buffer {
1706                self.multiply_basis_elements(
1707                    out_vec.as_slice_mut(),
1708                    *c,
1709                    first.0,
1710                    first.1,
1711                    second.0,
1712                    second.1,
1713                );
1714            }
1715            result.extend(&buffer);
1716            buffer.clear();
1717        }
1718        result
1719    }
1720}

1737    fn coproduct(&self, op_deg: i32, op_idx: usize) -> Vec<(i32, usize, i32, usize)> {
1738        assert_eq!(self.prime(), 2, "Coproduct at odd primes not supported");
1739        if op_deg == 0 {
1740            return vec![(0, 0, 0, 0)];
1741        }
1742        let xi_degrees = combinatorics::xi_degrees(self.prime());
1743
1744        let mut len = 1;
1745        let p_part = &self.basis_element_from_index(op_deg, op_idx).p_part;
1746
1747        for i in p_part.iter() {
1748            len *= i + 1;
1749        }
1750        let len = len as usize;
1751        let mut result = Vec::with_capacity(len);
1752
1753        let mut cur_ppart: PPart = vec![0; p_part.len()];
1754        loop {
1755            let mut left_degree: i32 = 0;
1756            for i in 0..cur_ppart.len() {
1757                left_degree += cur_ppart[i] as i32 * xi_degrees[i];
1758            }
1759            let right_degree: i32 = op_deg - left_degree;
1760
1761            let mut left_ppart = cur_ppart.clone();
1762            while let Some(0) = left_ppart.last() {
1763                left_ppart.pop();
1764            }
1765
1766            let mut right_ppart = cur_ppart
1767                .iter()
1768                .enumerate()
1769                .map(|(i, v)| p_part[i] - *v)
1770                .collect::<Vec<_>>();
1771            while let Some(0) = right_ppart.last() {
1772                right_ppart.pop();
1773            }
1774
1775            let left_idx = self.basis_element_to_index(&MilnorBasisElement {
1776                degree: left_degree,
1777                q_part: 0,
1778                p_part: left_ppart,
1779            });
1780            let right_idx = self.basis_element_to_index(&MilnorBasisElement {
1781                degree: right_degree,
1782                q_part: 0,
1783                p_part: right_ppart,
1784            });
1785
1786            result.push((left_degree, left_idx, right_degree, right_idx));
1787            if Self::increment_p_part(&mut cur_ppart, p_part) {
1788                break;
1789            }
1790        }
1791        result
1792    }
