1//! Finite fields over a prime.
2
3use fp::{
4    prime::ValidPrime,
5    vector::{FpSlice, FpSliceMut},
6};
7
8use crate::algebra::{Algebra, Bialgebra};
9
10/// $\mathbb{F}_p$, viewed as an [`Algebra`] over itself.
11///
12/// As an [`Algebra`], a field is one-dimensional, with basis element `1`.
13/// It is also trivially a coalgebra via the trivial diagonal comultiplication,
14/// and thus a [`Bialgebra`].
15pub struct Field {
16    prime: ValidPrime,
17}
18
19impl Field {
20    /// Returns a new `Field` over the given prime `p`.
21    pub fn new(p: ValidPrime) -> Self {
22        Self { prime: p }
23    }
24}
25
26impl std::fmt::Display for Field {
27    fn fmt(&self, f: &mut std::fmt::Formatter) -> std::fmt::Result {
28        write!(f, "F_{}", self.prime)
29    }
30}
31
32impl Algebra for Field {
33    fn prime(&self) -> ValidPrime {
34        self.prime
35    }
36
37    fn compute_basis(&self, _degree: i32) {}
38
39    fn dimension(&self, degree: i32) -> usize {
40        usize::from(degree == 0)
41    }
42
43    fn multiply_basis_elements(
44        &self,
45        mut result: FpSliceMut,
46        coeff: u32,
47        _r_degree: i32,
48        _r_idx: usize,
49        _s_degree: i32,
50        _s_idx: usize,
51    ) {
52        result.add_basis_element(0, coeff)
53    }
54
55    fn default_filtration_one_products(&self) -> Vec<(String, i32, usize)> {
56        vec![]
57    }
58
59    fn basis_element_to_string(&self, degree: i32, _idx: usize) -> String {
60        assert!(degree == 0);
61        "1".to_string()
62    }
63
64    fn element_to_string(&self, degree: i32, element: FpSlice) -> String {
65        assert!(degree == 0);
66        format!("{}", element.entry(0))
67    }
68
69    fn basis_element_from_string(&self, _elt: &str) -> Option<(i32, usize)> {
70        Some((0, 0))
71    }
72}
73
74impl Bialgebra for Field {
75    fn coproduct(&self, _op_deg: i32, _op_idx: usize) -> Vec<(i32, usize, i32, usize)> {
76        vec![(1, 0, 1, 0)]
77    }
78
79    fn decompose(&self, _op_deg: i32, _op_idx: usize) -> Vec<(i32, usize)> {
80        vec![(1, 0)]
81    }
82}
