19// This is here so that the Python bindings can use modules defined aor SteenrodAlgebraT with their own algebra enum.
20// In order for things to work SteenrodAlgebraT cannot implement Algebra.
21// Otherwise, the algebra enum for our bindings will see an implementation clash.
22pub trait SteenrodAlgebraT: Send + Sync + Algebra {

161impl SteenrodAlgebra {
162    pub fn from_json(
163        json: &Value,
164        mut algebra_type: AlgebraType,
165        unstable: bool,
166    ) -> anyhow::Result<Self> {
167        let spec: AlgebraSpec = AlgebraSpec::deserialize(json)?;
168
169        if let Some(list) = spec.algebra {
170            let algebra_name = &algebra_type.to_string();
171            if !list.iter().any(|x| x == algebra_name) {
172                println!("Module does not support algebra {algebra_name}");
173                println!("Using {} instead", list[0]);
174                algebra_type = list[0].parse()?;
175            }
176        }
177
178        Ok(match algebra_type {
179            AlgebraType::Adem => Self::AdemAlgebra(AdemAlgebra::new(spec.p, unstable)),
180            AlgebraType::Milnor => Self::MilnorAlgebra(MilnorAlgebra::new_with_profile(
181                spec.p,
182                spec.profile.unwrap_or_default(),
183                unstable,
184            )),
185        })
186    }
187}

189macro_rules! dispatch_steenrod {
190    () => {};
191    ($vis:vis fn $method:ident$(<$($lt:lifetime),+>)?(&$($lt2:lifetime)?self$(, $arg:ident: $ty:ty )*$(,)?) $(-> $ret:ty)?; $($tail:tt)*) => {
192        $vis fn $method$(<$($lt),+>)?(&$($lt2)?self, $($arg: $ty),* ) $(-> $ret)* {
193            match self {
194                SteenrodAlgebra::AdemAlgebra(a) => a.$method($($arg),*),
195                SteenrodAlgebra::MilnorAlgebra(a) => a.$method($($arg),*),
196            }
197        }
198        dispatch_steenrod!{$($tail)*}
199    };
200}
