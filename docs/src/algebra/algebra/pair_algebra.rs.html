1//! This implements the notion of a split pair algebra in the sense of
2//! <https://arxiv.org/abs/2105.07628v1>, whose notation we will use throughout.
3//!
4//! The Steenrod algebra admits a lift to a split pair algebra, which can be used to compute d2
5//! differentials algorithmically.
6//!
7//! To keep the pair algebra business contained, we put the implementation of the Milnor algebra as
8//! a pair algebra in this file instead of `milnor_algebra.rs`.

22/// A lift of an algebra to a split pair algebra. See module introduction for more.
23pub trait PairAlgebra: Algebra {
24    /// An element in the cohomological degree zero part of the pair algebra. This tends to not be
25    /// a ring over Fp, so we let the algebra specify how it wants to represent the elements.
26    type Element: Send + Sync;
27
28    fn element_is_zero(elt: &Self::Element) -> bool;
29
30    /// Assert that `elt` is in the image of the differential. Drop the data recording the
31    /// complement of the image of the differential.
32    fn finalize_element(_elt: &mut Self::Element) {}
33
34    /// Create a new zero element in the given degree.
35    fn new_pair_element(&self, degree: i32) -> Self::Element;
36
37    /// Given $r, s \in \pi_0(A)$, compute $\sigma(r) \sigma(s)$ and add the result to
38    /// `result`.
39    fn sigma_multiply_basis(
40        &self,
41        result: &mut Self::Element,
42        coeff: u32,
43        r_degree: i32,
44        r_idx: usize,
45        s_degree: i32,
46        s_idx: usize,
47    );
48
49    /// Same as [`PairAlgebra::sigma_multiply_basis`] but with non-basis elements.
50    fn sigma_multiply(
51        &self,
52        result: &mut Self::Element,
53        coeff: u32,
54        r_degree: i32,
55        r: FpSlice,
56        s_degree: i32,
57        s: FpSlice,
58    ) {
59        if coeff == 0 {
60            return;
61        }
62        for (r_idx, c) in r.iter_nonzero() {
63            for (s_idx, d) in s.iter_nonzero() {
64                self.sigma_multiply_basis(result, c * d * coeff, r_degree, r_idx, s_degree, s_idx);
65            }
66        }
67    }
68
69    /// Compute $A(r, s)$ and write the result to `result`.
70    fn a_multiply(
71        &self,
72        result: FpSliceMut,
73        coeff: u32,
74        r_degree: i32,
75        r: FpSlice,
76        s_degree: i32,
77        s: &Self::Element,
78    );
79
80    /// The element p is classified by a filtration on element in Ext of the underlying algebra,
81    /// which is represented by an indecomposable in degree 1. This returns the index of said
82    /// indecomposable.
83    fn p_tilde(&self) -> usize;
84
85    fn element_to_bytes(&self, elt: &Self::Element, buffer: &mut impl io::Write) -> io::Result<()>;
86    fn element_from_bytes(
87        &self,
88        degree: i32,
89        buffer: &mut impl io::Read,
90    ) -> io::Result<Self::Element>;
91}

100macro_rules! sub {
101    ($elt:ident, $k:expr, $n:expr) => {
102        if $k > 0 {
103            if $elt.p_part[$k - 1] < (1 << $n) {
104                continue;
105            }
106            $elt.p_part[$k - 1] -= 1 << $n;
107            $elt.degree -= combinatorics::xi_degrees(TWO)[$k - 1] * (1 << $n);
108        }
109    };
110}
111macro_rules! unsub {
112    ($elt:ident, $k:expr, $n:expr) => {
113        if $k > 0 {
114            $elt.p_part[$k - 1] += 1 << $n;
115            $elt.degree += combinatorics::xi_degrees(TWO)[$k - 1] * (1 << $n);
116        }
117    };
118}

120pub struct MilnorPairElement {
121    ones: FpVector,
122    twos: FpVector,
123    #[cfg(debug_assertions)]
124    degree: i32,
125    ys: Vec<Vec<FpVector>>,
126}
127
128impl PairAlgebra for MilnorAlgebra {
129    type Element = MilnorPairElement;
130
131    fn new_pair_element(&self, degree: i32) -> Self::Element {
132        let p = self.prime();
133        assert_eq!(p, TWO);
134
135        let max_k = if degree == 0 {
136            0
137        } else {
138            fp::prime::log2(degree as usize) + 1
139        };
140        let mut ys = Vec::with_capacity(max_k);
141        for k in 0..max_k {
142            let rem_degree = degree as usize + 1 - (1 << k);
143            let max_l = fp::prime::log2(rem_degree) + 1;
144            let mut row = Vec::with_capacity(max_l);
145            for l in 0..max_l {
146                row.push(FpVector::new(
147                    p,
148                    self.dimension((rem_degree - (1 << l)) as i32),
149                ));
150            }
151            ys.push(row);
152        }
153
154        MilnorPairElement {
155            ones: FpVector::new(p, self.dimension(degree)),
156            twos: FpVector::new(p, self.dimension(degree)),
157            ys,
158            #[cfg(debug_assertions)]
159            degree,
160        }
161    }
162
163    fn element_is_zero(elt: &Self::Element) -> bool {
164        elt.ones.is_zero()
165            && elt.twos.is_zero()
166            && elt.ys.iter().all(|v| v.iter().all(|x| x.is_zero()))
167    }
168
169    fn p_tilde(&self) -> usize {
170        0
171    }
172
173    /// Assert that `elt` is in the image of the differential. Drop the data recording the
174    /// complement of the image of the differential.
175    fn finalize_element(elt: &mut Self::Element) {
176        assert!(elt.ones.is_zero());
177        elt.ones = FpVector::new(elt.twos.prime(), 0);
178    }
179
180    fn sigma_multiply_basis(
181        &self,
182        result: &mut Self::Element,
183        coeff: u32,
184        r_degree: i32,
185        r_idx: usize,
186        s_degree: i32,
187        s_idx: usize,
188    ) {
189        #[cfg(debug_assertions)]
190        assert_eq!(r_degree + s_degree, result.degree);
191
192        // First write the Y terms
193        let mut r = self.basis_element_from_index(r_degree, r_idx).clone();
194        let mut s = self.basis_element_from_index(s_degree, s_idx).clone();
195
196        PPartAllocation::with_local(|mut allocation| {
197            for k in 0..s.p_part.len() {
198                sub!(s, k + 1, 0);
199                for n in 1..r.p_part.len() + 1 {
200                    sub!(r, n, k);
201                    for m in 0..n {
202                        sub!(r, m, k);
203                        allocation = self.multiply_with_allocation(
204                            result.ys[m + k][n + k].as_slice_mut(),
205                            coeff,
206                            &r,
207                            &s,
208                            i32::MAX,
209                            allocation,
210                        );
211                        unsub!(r, m, k);
212                    }
213                    unsub!(r, n, k);
214                }
215                unsub!(s, k + 1, 0);
216            }
217
218            // Now the product terms
219            let mut multiplier = PPartMultiplier::<true>::new_from_allocation(
220                TWO,
221                &r.p_part,
222                &s.p_part,
223                allocation,
224                0,
225                r.degree + s.degree,
226            );
227
228            // coeff should always be 1, so no need to optimize the even case.
229            while let Some(c) = multiplier.next() {
230                let idx = self.basis_element_to_index(&multiplier.ans);
231                let c = c * coeff;
232                // TODO: optimize
233                if c == 2 {
234                    result.twos.add_basis_element(idx, 1);
235                } else {
236                    // c = 1 or 3
237                    let existing = result.ones.entry(idx);
238                    let c = c + existing;
239                    result.ones.set_entry(idx, c & 1);
240                    result.twos.add_basis_element(idx, (c >> 1) & 1);
241                }
242            }
243            multiplier.into_allocation()
244        });
245    }
246
247    fn a_multiply(
248        &self,
249        mut result: FpSliceMut,
250        coeff: u32,
251        r_degree: i32,
252        r: FpSlice<'_>,
253        s_degree: i32,
254        s: &Self::Element,
255    ) {
256        assert!(s.ones.is_zero());
257
258        if r_degree == 0 {
259            return;
260        }
261
262        // The twos terms
263        for (r_idx, c) in r.iter_nonzero() {
264            let mut r = self.basis_element_from_index(r_degree, r_idx).clone();
265            sub!(r, 1, 0);
266            self.multiply_basis_by_element(
267                result.copy(),
268                coeff * c,
269                &r,
270                s_degree,
271                s.twos.as_slice(),
272            );
273            unsub!(r, 1, 0);
274        }
275
276        // The Y terms
277        for (k, row) in s.ys.iter().enumerate() {
278            for (l, vec) in row.iter().enumerate() {
279                if vec.is_zero() {
280                    continue;
281                }
282                let degree_shift = (1 << k) + (1 << l) - 1;
283                for (r_idx, c) in r.iter_nonzero() {
284                    let r = self.basis_element_from_index(r_degree, r_idx);
285                    let a_degree = r_degree + degree_shift - 1;
286                    a_y_cached(self, r, k, l, |v| {
287                        self.multiply_element_by_element(
288                            result.copy(),
289                            coeff * c,
290                            a_degree,
291                            v.as_slice(),
292                            s_degree - degree_shift,
293                            vec.as_slice(),
294                        )
295                    });
296                }
297            }
298        }
299    }
300
301    fn element_to_bytes(&self, elt: &Self::Element, buffer: &mut impl io::Write) -> io::Result<()> {
302        elt.twos.to_bytes(buffer)?;
303        for row in &elt.ys {
304            for v in row {
305                v.to_bytes(buffer)?;
306            }
307        }
308        Ok(())
309    }
310
311    fn element_from_bytes(
312        &self,
313        degree: i32,
314        buffer: &mut impl io::Read,
315    ) -> io::Result<Self::Element> {
316        let p = self.prime();
317        assert_eq!(p, TWO);
318
319        let max_k = if degree == 0 {
320            0
321        } else {
322            fp::prime::log2(degree as usize) + 1
323        };
324
325        let twos = FpVector::from_bytes(p, self.dimension(degree), buffer)?;
326
327        let mut ys = Vec::with_capacity(max_k);
328        for k in 0..max_k {
329            let rem_degree = degree as usize + 1 - (1 << k);
330            let max_l = fp::prime::log2(rem_degree) + 1;
331            let mut row = Vec::with_capacity(max_l);
332            for l in 0..max_l {
333                row.push(FpVector::from_bytes(
334                    p,
335                    self.dimension((rem_degree - (1 << l)) as i32),
336                    buffer,
337                )?);
338            }
339            ys.push(row);
340        }
341
342        Ok(MilnorPairElement {
343            ones: FpVector::new(p, 0),
344            twos,
345            ys,
346            #[cfg(debug_assertions)]
347            degree,
348        })
349    }
350}

352// Use thread-local storage to memoize a_y computation. Since the possible values of k, l grow as
353// log n, in practice it is going to be at most, say, 64, and the memory usage here should be
354// dwarfed by that of storing a single quasi-inverse
355
356use std::hash::{BuildHasher, Hash, Hasher};
357
358thread_local! {
359    static AY_CACHE: RefCell<HashMap<(MilnorElt, (usize, usize)), FpVector>> = RefCell::new(HashMap::default());
360}
361
362/// Compute $A(Sq(R), Y_{k, l})$ where $a = Sq(R)$. This queries the cache and computes it using
363/// [`a_y_inner`] if not available.
364fn a_y_cached(
365    algebra: &MilnorAlgebra,
366    a: &MilnorElt,
367    k: usize,
368    l: usize,
369    f: impl FnOnce(&FpVector),
370) {
371    AY_CACHE.with(|cache| {
372        let cache = &mut *cache.try_borrow_mut().unwrap();
373        let mut hasher = cache.hasher().build_hasher();
374        a.hash(&mut hasher);
375        (k, l).hash(&mut hasher);
376
377        let raw_entry = cache.raw_entry();
378        let result = raw_entry
379            .from_hash(hasher.finish(), |v| &v.0 == a && v.1 == (k, l))
380            .map(|(_, y)| y);
381
382        match result {
383            Some(v) => f(v),
384            None => {
385                let v = a_y_inner(algebra, a, k, l);
386                f(&v);
387                cache.insert((a.clone(), (k, l)), v);
388            }
389        }
390    })
391}
392
393/// Actually computes $A(a, Y_{k, l})$ and returns the result.
394fn a_y_inner(algebra: &MilnorAlgebra, a: &MilnorElt, k: usize, l: usize) -> FpVector {
395    let mut a = a.clone();
396    let mut result = FpVector::new(TWO, algebra.dimension(a.degree + (1 << k) + (1 << l) - 2));
397    let mut t = MilnorElt {
398        q_part: 0,
399        p_part: vec![],
400        degree: 0,
401    };
402
403    for i in 0..=a.p_part.len() {
404        if i + k < l {
405            continue;
406        }
407
408        sub!(a, i, k);
409        for j in 0..=std::cmp::min(i + k - l, a.p_part.len()) {
410            sub!(a, j, l);
411
412            t.p_part.clear();
413            t.p_part.resize(k + i, 0);
414
415            t.p_part[k + i - 1] += 1;
416            t.p_part[l + j - 1] += 1;
417
418            t.degree = (1 << (k + i)) + (1 << (l + j)) - 2;
419
420            // We can just read off the value of the product instead of passing through the
421            // algorithm, but this is cached so problem for another day...
422            algebra.multiply(result.as_slice_mut(), 1, &t, &a);
423
424            unsub!(a, j, l);
425        }
426        unsub!(a, i, k);
427    }
428    result
429}

438    fn from_p_part(p_part: &[PPartEntry]) -> MilnorElt {
439        let degree = p_part
440            .iter()
441            .enumerate()
442            .map(|(i, &n)| combinatorics::xi_degrees(TWO)[i] * (n as i32))
443            .sum();
444
445        MilnorElt {
446            q_part: 0,
447            p_part: p_part.into(),
448            degree,
449        }
450    }

452    #[test]
453    fn test_a_y() {
454        let algebra = MilnorAlgebra::new(TWO, false);
455
456        let mut result = FpVector::new(TWO, 0);
457
458        let mut check = |p_part: &[PPartEntry], k, l, ans: Expect| {
459            let a = from_p_part(p_part);
460
461            let target_deg = a.degree + (1 << k) + (1 << l) - 2;
462            algebra.compute_basis(target_deg + 1);
463            result.set_scratch_vector_size(algebra.dimension(target_deg));
464            a_y_cached(&algebra, &a, k, l, |v| result.add(v, 1));
465            ans.assert_eq(&algebra.element_to_string(target_deg, result.as_slice()));
466        };
467
468        check(&[1], 0, 1, expect![["P(2)"]]);
469        check(&[1], 1, 2, expect![["0"]]);
470        check(&[0, 1], 0, 1, expect![["P(1, 1)"]]);
471        check(&[0, 2], 1, 3, expect![["P(0, 0, 2)"]]);
472        check(&[1, 2], 0, 1, expect![["P(2, 2)"]]);
473    }
