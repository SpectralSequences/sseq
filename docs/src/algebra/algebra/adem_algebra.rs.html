40/// An Adem basis element for the Steenrod algebra.
41///
42/// To encode an element
43/// $$
44///     \beta^{\varepsilon_0} P^{i_0}
45///     \beta^{\varepsilon_1} P^{i_1}
46///     \cdots
47///     \beta^{\varepsilon_n} P^{i_n}
48///     \beta^{\varepsilon_{n+1}},
49/// $$
50/// we set
51/// $$ \begin{aligned}
52/// \mathtt{ps}         &= [i_0, i_1, \ldots, i_n] \\\\
53/// \mathtt{bocksteins} &= 000\cdots0\varepsilon_{n+1} \varepsilon_n \cdots \varepsilon_0
54/// \end{aligned} $$
55#[derive(Debug, Clone)]
56pub struct AdemBasisElement {
57    /// The degree of the element.
58    pub degree: i32,
59    /// A bitset of which $\beta$ Bocksteins are in the element's decomposition.
60    pub bocksteins: u32,
61    /// A list of which Steenrod powers are in the element's decompositions.
62    pub ps: Vec<u32>,
63    /// Whether to denote the generators as powers $P^i$ or squares $Sq^i$ when using string formatting;
64    /// `true` denotes the former.
65    pub p_or_sq: bool,
66}
67
68/// A Steenrod power $P^i$, or a Bockstein $\beta^\varepsilon$.
69#[derive(Debug)]
70pub enum PorBockstein {
71    P(u32),
72    Bockstein(bool),
73}
74
75impl AdemBasisElement {
76    /// Returns an iterator over the element's decomposition.
77    ///
78    /// This returns alternating Bocksteins and Ps, but skipping any
79    /// `\beta^0` factors.
80    fn iter_filtered(&self) -> impl Iterator<Item = PorBockstein> + '_ {
81        BitflagIterator::new(self.bocksteins as u64)
82            .map(PorBockstein::Bockstein)
83            .interleave(self.ps.iter().map(|b| PorBockstein::P(*b)))
84            .filter(|b| !matches!(b, PorBockstein::Bockstein(false)))
85    }
86
87    fn excess(&self, p: ValidPrime) -> u32 {
88        if self.degree < 2 {
89            self.degree as u32 // special case these so that self.ps[0] is defined
90        } else if p == 2 {
91            2 * self.ps[0] - self.ps.iter().sum::<u32>()
92        } else {
93            2 * (p * self.ps[0] - (p - 1) * self.ps.iter().sum::<u32>()) + 2 * (self.bocksteins & 1)
94                - self.bocksteins.count_ones()
95        }
96    }
97}

99impl PartialEq for AdemBasisElement {
100    fn eq(&self, other: &Self) -> bool {
101        self.ps == other.ps && self.bocksteins == other.bocksteins
102    }
103}

114impl fmt::Display for AdemBasisElement {
115    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
116        let p_or_sq = if self.p_or_sq { "P" } else { "Sq" };
117        let result = self
118            .iter_filtered()
119            .map(|e| match e {
120                PorBockstein::P(exp) => format!("{p_or_sq}{exp}"),
121                PorBockstein::Bockstein(_) => "b".to_string(),
122            })
123            .format(" ");
124
125        write!(f, "{result}")?;
126        Ok(())
127    }

130/// Shifts a `Vec`'s elements back by `offset`.
131///
132/// # Safety
133///
134/// This function leaves `v` in an invalid state, since `ptr` does not point to
135/// the beginning of an allocation after this function completes. Operations
136/// which may trigger resizing *must* be avoided, since the allocator will
137/// otherwise get extremely confused.
138///
139/// The sum of `offset`s to calls to this function for a particular `v` must
140/// always be positive (i.e., to avoid pointing before the allocation), less
141/// than the `v`'s capacity (to avoid pointing after), and zero when `v` is
142/// destroyed or resized (for the reasons described above)
143unsafe fn shift_vec<T>(v: &mut Vec<T>, offset: isize) {
144    let ptr = v.as_ptr();
145    let len = v.len();
146    let cap = v.capacity();
147    let w = std::mem::replace(
148        v,
149        Vec::from_raw_parts(
150            (ptr as *mut T).offset(offset),
151            (len as isize - offset) as usize,
152            (cap as isize - offset) as usize,
153        ),
154    );
155    std::mem::forget(w);
156}

185    fn magic(&self) -> u32 {
186        if self.unstable_enabled {
187            1 + (self.prime() << 16)
188        } else {

190        }
191    }

193    fn prime(&self) -> ValidPrime {
194        self.p
195    }

197    fn default_filtration_one_products(&self) -> Vec<(String, i32, usize)> {
198        let mut products = Vec::with_capacity(4);
199        let max_degree = if self.generic {
200            products.push((
201                "a_0".to_string(),
202                AdemBasisElement {
203                    degree: 1,
204                    bocksteins: 1,
205                    ps: vec![],
206                    p_or_sq: self.prime() != 2,
207                },
208            ));
209            products.push((
210                "h_0".to_string(),
211                AdemBasisElement {
212                    degree: (2 * self.prime() - 2) as i32,
213                    bocksteins: 0,
214                    ps: vec![1],
215                    p_or_sq: self.prime() != 2,
216                },
217            ));
218            (2 * self.prime() - 2) as i32
219        } else {
220            for i in 0..4 {
221                let degree = 1 << i; // degree is 2^hi
222                let ps = vec![degree as u32];
223                products.push((
224                    format!("h_{i}"),
225                    AdemBasisElement {
226                        degree,
227                        bocksteins: 0,
228                        ps,
229                        p_or_sq: self.prime() != 2,
230                    },
231                ));
232            }
233            1 << 3
234        };
235
236        self.compute_basis(max_degree);
237        products
238            .into_iter()
239            .map(|(name, b)| (name, b.degree, self.basis_element_to_index(&b)))
240            .collect()
241    }

243    fn compute_basis(&self, max_degree: i32) {
244        if self.generic {
245            self.generate_basis_generic(max_degree);
246            self.generate_basis_element_to_index_map(max_degree);
247            self.generate_multiplication_table_generic(max_degree);
248        } else {
249            self.generate_basis2(max_degree);
250            self.generate_basis_element_to_index_map(max_degree);
251            self.generate_multiplication_table_2(max_degree);
252        }
253
254        if self.unstable_enabled {
255            self.generate_excess_table(max_degree);
256        }
257    }

259    fn dimension(&self, degree: i32) -> usize {
260        if degree < 0 {
261            0
262        } else {
263            self.basis_table[degree as usize].len()
264        }
265    }

267    fn multiply_basis_elements(
268        &self,
269        result: FpSliceMut,

271        r_degree: i32,
272        r_index: usize,
273        s_degree: i32,

276        self.multiply_inner(
277            result,
278            coeff,
279            r_degree,
280            r_index,
281            s_degree,
282            s_index,
283            i32::MAX,
284        );
285    }

287    fn basis_element_to_string(&self, degree: i32, idx: usize) -> String {
288        format!("{}", self.basis_element_from_index(degree, idx))
289    }
290
291    fn basis_element_from_string(&self, mut elt: &str) -> Option<(i32, usize)> {
292        use nom::{sequence::preceded, Parser};
293
294        use crate::steenrod_parser::{digits, p_or_sq};
295
296        let q = self.q();
297
298        let mut bocksteins = 0;
299        let mut ps = Vec::new();
300        let mut degree = 0;
301        let mut bockstein_count = 0;
302
303        loop {
304            if elt.starts_with('b') {
305                elt = &elt[std::cmp::min(2, elt.len())..];
306                degree += 1;
307                bocksteins += 1 << bockstein_count;
308            }
309            if elt.is_empty() {
310                break;
311            }
312            let (rem, sqn) = preceded(p_or_sq, digits).parse(elt).ok()?;
313            elt = rem;
314
315            ps.push(sqn);
316            degree += sqn as i32 * q;
317
318            bockstein_count += 1;
319        }
320
321        self.compute_basis(degree);
322        Some((
323            degree,
324            self.basis_element_to_index(&AdemBasisElement {
325                ps,
326                bocksteins,
327                degree,
328                p_or_sq: self.generic,
329            }),
330        ))
331    }
332}

334impl GeneratedAlgebra for AdemAlgebra {
335    fn generator_to_string(&self, degree: i32, _idx: usize) -> String {
336        if self.generic {
337            if degree == 1 {
338                "b".to_string()
339            } else {
340                format!("P{}", degree as u32 / (2 * self.prime() - 2))
341            }
342        } else {
343            format!("Sq{degree}")
344        }
345    }
346
347    fn generators(&self, degree: i32) -> Vec<usize> {
348        let p = self.prime();
349        if degree == 0 {
350            return vec![];
351        }
352        if self.generic {
353            if degree == 1 {
354                return vec![0];
355            }
356            // Test if degree is q*p^k.
357            let mut temp_degree = degree as u32;
358            if !temp_degree.is_multiple_of(2 * (p - 1)) {
359                return vec![];
360            }
361            temp_degree /= 2 * (p - 1);
362            while temp_degree.is_multiple_of(p.as_u32()) {
363                temp_degree /= p;
364            }
365            if temp_degree != 1 {
366                return vec![];
367            }
368            let idx = self.basis_element_to_index(&AdemBasisElement {
369                degree,
370                bocksteins: 0,
371                ps: vec![degree as u32 / (2 * p - 2)],
372                p_or_sq: self.prime() != 2,
373            });
374            vec![idx]
375        } else {
376            // I guess we're assuming here that not generic ==> p == 2. There's probably tons of places we assume that though.
377            if degree.count_ones() != 1 {
378                return vec![];
379            }
380            let idx = self.basis_element_to_index(&AdemBasisElement {
381                degree,
382                bocksteins: 0,
383                ps: vec![degree as u32],
384                p_or_sq: self.prime() != 2,
385            });
386            vec![idx]
387        }
388    }
389
390    fn decompose_basis_element(
391        &self,
392        degree: i32,
393        idx: usize,
394    ) -> Vec<(u32, (i32, usize), (i32, usize))> {
395        if self.generic {
396            self.decompose_basis_element_generic(degree, idx)
397        } else {
398            self.decompose_basis_element_2(degree, idx)
399        }
400    }
401
402    /// We return Adem relations $b^2 = 0$, $P^i P^j = \cdots$ for $i < pj$, and $P^i b P^j = \cdots$ for $i < pj + 1$. It suffices to check these because
403    /// they generate all relations.
404    fn generating_relations(&self, degree: i32) -> Vec<Vec<(u32, (i32, usize), (i32, usize))>> {
405        if self.generic && degree == 2 {
406            // beta^2 = 0 is an edge case
407            return vec![vec![(1, (1, 0), (1, 0))]];
408        }
409
410        let p = self.prime();
411        let inadmissible_pairs = combinatorics::inadmissible_pairs(p, self.generic, degree);
412        let mut result = Vec::new();
413
414        for (x, b, y) in inadmissible_pairs {
415            let mut relation = Vec::new();
416            // Adem relation
417            let first_sq = self.beps_pn(0, x);
418            let second_sq = self.beps_pn(b, y);
419            relation.push((p - 1, first_sq, second_sq));
420            for e1 in 0..=b {
421                let e2 = b - e1;
422                // e1 and e2 determine where a bockstein shows up.
423                // e1 determines if a bockstein shows up in front
424                // e2 determines if a bockstein shows up in middle
425                // So our output term looks like b^{e1} P^{x+y-j} b^{e2} P^{j}
426                for j in 0..=x / p {
427                    let c = combinatorics::adem_relation_coefficient(p, x, y, j, e1, e2);
428                    if c == 0 {
429                        continue;
430                    }
431                    let idx = self.basis_element_to_index(&AdemBasisElement {
432                        degree,
433                        ps: if j == 0 {
434                            vec![x + y]
435                        } else {
436                            vec![x + y - j, j]
437                        },
438                        bocksteins: e1 | (e2 << 1),
439                        p_or_sq: self.prime() != 2,
440                    });
441                    relation.push((c, (degree, idx), (0, 0)));
442                }
443            }
444            result.push(relation);
445        }
446        result
447    }
448}

484impl AdemAlgebra {
485    /// Constructs a new [`AdemAlgebra`].
486    pub fn new(p: ValidPrime, unstable_enabled: bool) -> Self {
487        let even_basis_table = OnceVec::new();
488        let basis_table = OnceVec::new();
489        let basis_element_to_index_map = OnceVec::new();
490        let multiplication_table = OnceVec::new();
491        let excess_table = OnceVec::new();
492        Self {
493            p,
494            generic: p != 2,
495            unstable_enabled,
496            even_basis_table,
497            basis_table,
498            basis_element_to_index_map,
499            multiplication_table,
500            excess_table,
501        }
502    }
503
504    pub fn generic(&self) -> bool {
505        self.generic
506    }
507
508    pub fn q(&self) -> i32 {
509        if self.generic {
510            2 * (self.prime().as_i32() - 1)
511        } else {
512            1
513        }
514    }
515
516    fn generate_basis_even(&self, max_degree: i32) {
517        let p = self.prime();
518
519        self.even_basis_table.extend(0, |_| {
520            vec![AdemBasisElement {
521                degree: 0,
522                bocksteins: if self.generic { u32::MAX << 1 } else { 0 },
523                ps: vec![],
524                p_or_sq: p != 2,
525            }]
526        });
527
528        self.even_basis_table.extend(max_degree as usize, |n| {
529            let n = n as i32;
530
531            let mut basis = Vec::new();
532            // Put Sqn into the list.
533            basis.push(AdemBasisElement {
534                degree: n,
535                bocksteins: if self.generic { u32::MAX << 2 } else { 0 },
536                ps: vec![n as u32],
537                p_or_sq: self.prime() != 2,
538            });
539
540            // last = last term. We append (last,) to the end of
541            // elements of degree n - last whose own last square is
542            // at least p * last.
543            // In order for this to be possible, this means that p last <= n - last,
544            // or (p+1) * last <= n or last <= n/(p+1). We order the squares in decreasing
545            // order of their last element so that as we walk over the previous basis
546            // when we find a square whose end is too small, we can break.
547            for last in (1..=n / (p.as_i32() + 1)).rev() {
548                let previous_basis = &self.even_basis_table[(n - last) as usize];
549                for prev_elt in previous_basis {
550                    let prev_elt_p_len = prev_elt.ps.len();
551                    let old_last_sq = prev_elt.ps[prev_elt_p_len - 1] as i32;
552                    if old_last_sq < p.as_i32() * last {
553                        break;
554                    }
555                    // Write new basis element to basis element buffer
556
557                    let degree = prev_elt.degree + last;
558                    // We're using bocksteins as a bit mask:
559                    // A bit in bocksteins shall be set if it's illegal for a bockstein to occur there.
560                    let mut bocksteins = prev_elt.bocksteins;
561                    if self.generic {
562                        bocksteins |= if old_last_sq == p.as_i32() * last {
563                            1 << prev_elt_p_len
564                        } else {
565                            0
566                        };
567                        bocksteins &= !(1 << (prev_elt_p_len + 1));
568                    }
569                    let mut ps: Vec<u32> = Vec::with_capacity(prev_elt_p_len + 1);
570                    ps.extend_from_slice(&prev_elt.ps);
571                    ps.push(last as u32);
572                    basis.push(AdemBasisElement {
573                        degree,
574                        bocksteins,
575                        ps,
576                        p_or_sq: p != 2,
577                    });
578                }
579            }
580            basis
581        });
582    }
583
584    fn generate_basis2(&self, max_degree: i32) {
585        self.generate_basis_even(max_degree);
586        self.basis_table.extend(max_degree as usize, |n| {
587            let mut table = self.even_basis_table[n].clone();
588            if self.unstable_enabled {
589                table.sort_by_cached_key(|e| e.excess(fp::prime::TWO));
590            }
591            table
592        });
593    }
594
595    // Our approach is to pick the bocksteins and the P's separately and merge.
596    fn generate_basis_generic(&self, max_degree: i32) {
597        self.generate_basis_even(max_degree);
598
599        let p = self.prime();
600        let q = 2 * (p.as_i32() - 1);
601        self.basis_table.extend(max_degree as usize, |n| {
602            let n = n as i32;
603            let residue = n % q;
604
605            let mut basis: Vec<AdemBasisElement> = Vec::new();
606            // First we need to know how many bocksteins we'll use so we know how much degree
607            // to assign to the Ps. The Ps all have degree divisible by q=2p-2, so num_bs needs to
608            // be congruent to degree mod q.
609            let num_bs_bound = std::cmp::min(MAX_XI_TAU, (n + 1) as usize);
610            for num_bs in (residue as usize..num_bs_bound).step_by(q as usize) {
611                let even_basis = &self.even_basis_table[(n as usize - num_bs) / q as usize];
612
613                for elt in even_basis.iter().rev() {
614                    if elt.ps.len() + 1 < num_bs {
615                        // Not enough space to fit the bs.
616                        // Ps ordered in descending length, so none of the later ones will have space either
617                        continue;
618                    }
619                    for bocksteins in BinomialIterator::new(num_bs) {
620                        if 32 - bocksteins.leading_zeros() > elt.ps.len() as u32 + 1 {
621                            // Too large of a bockstein.
622                            break;
623                        }
624                        // elt.bocksteins contains 1 in locations where the sequence is "just
625                        // barely admissible" and so adding a bockstein would make it inadmissible.
626                        if bocksteins & elt.bocksteins != 0 {
627                            continue;
628                        }
629                        basis.push(AdemBasisElement {
630                            degree: n,
631                            bocksteins,
632                            ps: elt.ps.clone(),
633                            p_or_sq: p != 2,
634                        });
635                        // BinomialIterator does not have a stopping condition, so we need to stop
636                        // manually here
637                        if num_bs == 0 {
638                            break;
639                        }
640                    }
641                }
642            }
643            if self.unstable_enabled {
644                basis.sort_by_cached_key(|e| e.excess(p));
645            }
646            basis
647        });
648    }
649
650    fn generate_basis_element_to_index_map(&self, max_degree: i32) {
651        self.basis_element_to_index_map
652            .extend(max_degree as usize, |n| {
653                let basis = &self.basis_table[n];
654                let mut map = HashMap::default();
655                map.reserve(basis.len());
656                for (i, basis) in basis.iter().enumerate() {
657                    map.insert(basis.clone(), i);
658                }
659                map
660            });
661    }

678    fn tail_of_basis_element_to_index(
679        &self,
680        elt: &mut AdemBasisElement,
681        idx: u32,
682        q: u32,
683    ) -> usize {
684        let degree = elt.degree;
685        let bocksteins = elt.bocksteins;
686        for i in 0..idx as usize {
687            elt.degree -= (q * elt.ps[i] + (elt.bocksteins & 1)) as i32;
688            elt.bocksteins >>= 1;
689        }
690        unsafe {
691            shift_vec(&mut elt.ps, idx as isize);
692        }
693        let result = self.basis_element_to_index(elt);
694        unsafe {
695            shift_vec(&mut elt.ps, -(idx as isize));
696        }
697        elt.degree = degree;
698        elt.bocksteins = bocksteins;
699        result
700    }

702    fn generate_multiplication_table_2(&self, max_degree: i32) {
703        self.multiplication_table.extend(0, |_| Vec::new());
704
705        // degree -> first_square -> admissibile sequence idx -> result vector
706        self.multiplication_table.extend(max_degree as usize, |n| {
707            let mut table: Vec<Vec<FpVector>> = Vec::with_capacity(n + 1);
708            let n = n as i32;
709
710            table.push(Vec::with_capacity(0));
711            for x in 1..=n {
712                let dimension = self.dimension(n - x);
713                table.push(Vec::with_capacity(dimension));
714            }
715            for x in (1..=n).rev() {
716                for idx in 0..self.dimension(n - x) {
717                    let res = self.generate_multiplication_table_2_step(&table, n, x, idx);
718                    table[x as usize].push(res);
719                }
720            }
721            table
722        });
723    }

725    fn generate_multiplication_table_2_step(
726        &self,
727        table: &[Vec<FpVector>],
728        n: i32,
729        x: i32,
730        idx: usize,
731    ) -> FpVector {
732        let output_dimension = self.dimension(n);
733        let mut result = FpVector::new(self.prime(), output_dimension);
734        let cur_basis_elt = self.basis_element_from_index(n - x, idx);
735        let x = x as u32;
736        let mut working_elt = cur_basis_elt.clone();
737
738        // Be careful to deal with the case that cur_basis_elt has length 0
739        // If the length is 0 or the sequence is already admissible, we can just write a 1 in the answer
740        // and continue.
741        if cur_basis_elt.ps.is_empty() || x >= 2 * cur_basis_elt.ps[0] {
742            working_elt.ps.insert(0, x);
743            working_elt.degree = n;
744            let out_idx = self.basis_element_to_index(&working_elt);
745            result.add_basis_element(out_idx, 1);
746            return result;
747        }
748
749        // We now want to decompose Sq^x Sq^y = \sum_j *coef* Sq^{x + y - j} Sq^j.
750        let y = working_elt.ps[0];
751
752        let tail_idx = self.tail_of_basis_element_to_index(&mut working_elt, 1, 1);
753
754        for j in 0..=x / 2 {
755            if combinatorics::adem_relation_coefficient(fp::prime::TWO, x, y, j, 0, 0) == 0 {
756                continue;
757            }
758            if j == 0 {
759                working_elt.ps[0] = x + y;
760                working_elt.degree += x as i32;
761                // In this case the result is guaranteed to be admissible so we can immediately add it to result
762                let out_idx = self.basis_element_to_index(&working_elt);
763                result.add_basis_element(out_idx, 1);
764                continue;
765            }
766            // Now we need to reduce Sqj * (rest of Sqs)
767            // The answer to this is in the table we're currently making.
768            // total degree -> first sq -> idx of rest of squares
769            let rest_reduced =
770                &self.multiplication_table[(n as u32 - (x + y) + j) as usize][j as usize][tail_idx];
771            for (i, _coeff) in rest_reduced.iter_nonzero() {
772                // Reduce Sq^{x+y-j} * whatever square using the table in the same degree, larger index
773                // Since we're doing the first squares in decreasing order and x + y - j > x,
774                // we already calculated this.
775                let source = &table[(x + y - j) as usize][i];
776                result.add(source, 1);
777            }
778        }
779        result
780    }

782    fn generate_multiplication_table_generic(&self, max_degree: i32) {
783        self.multiplication_table.extend(0, |_| Vec::new());
784
785        let q = 2 * self.prime().as_i32() - 2;
786        self.multiplication_table.extend(max_degree as usize, |n| {
787            let n = n as i32;
788            let mut table: Vec<Vec<FpVector>> = Vec::with_capacity(2 * (n / q + 1) as usize);
789            for i in 0..=n / q {
790                for b in 0..=1 {
791                    // This corresponds to x = 2i + b
792                    let dimension = self.dimension(n - q * i - b);
793                    table.push(Vec::with_capacity(dimension));
794                }
795            }
796            for i in (0..=n / q).rev() {
797                for idx in 0..self.dimension(n - q * i - 1) {
798                    let res =
799                        self.generate_multiplication_table_generic_step(&table, n, 2 * i + 1, idx);
800                    table[1 + 2 * i as usize].push(res);
801                }
802                if i != 0 {
803                    for idx in 0..self.dimension(n - q * i) {
804                        let res =
805                            self.generate_multiplication_table_generic_step(&table, n, 2 * i, idx);
806                        table[2 * i as usize].push(res);
807                    }
808                }
809            }
810            table
811        });
812    }

814    /// This function expresses $Sq^x$ (current) in terms of the admissible basis and returns
815    /// the result as an FpVector, where (current) is the admissible monomial of degree $n - qx$
816    /// (so that $Sq^x)$ (current) has degree $n$) and index `idx`.
817    ///
818    /// Here $Sq^x$ means $P^{x/2}$ if $x$ is even and $\beta P^{(x-1)/2}$ if $x$ is odd.
819    ///
820    /// Note that x is always positive.
821    fn generate_multiplication_table_generic_step(
822        &self,
823        table: &[Vec<FpVector>],
824        n: i32,
825        x: i32,
826        idx: usize,
827    ) -> FpVector {
828        let p: i32 = self.prime().as_i32(); // we use p for the i32 version and self.p for the u32 version
829        let q: i32 = 2 * p - 2;
830
831        let x: u32 = x as u32;
832
833        let output_dimension = self.dimension(n);
834        let mut result = FpVector::new(self.prime(), output_dimension);
835
836        // If x is just \beta, this is super easy.
837        if x == 1 {
838            let mut elt = self.basis_element_from_index(n - 1, idx).clone();
839            if elt.bocksteins & 1 == 0 {
840                elt.bocksteins |= 1;
841                elt.degree += 1;
842                let index = self.basis_element_to_index(&elt);
843                result.add_basis_element(index, 1);
844            }
845            return result;
846        }
847
848        // If x is \beta P^i, it is also easy.
849        if x & 1 != 0 {
850            let rest_reduced = &self.multiplication_table[n as usize - 1][x as usize - 1][idx];
851            for (id, coef) in rest_reduced.iter().enumerate() {
852                let mut elt = self.basis_element_from_index(n - 1, id).clone();
853                // We dispose of all terms with a leading Bockstein
854                if elt.bocksteins & 1 == 0 {
855                    elt.bocksteins |= 1;
856                    elt.degree += 1;
857                    let index = self.basis_element_to_index(&elt);
858                    result.add_basis_element(index, coef);
859                }
860            }
861            return result;
862        }
863
864        // Now there is no Bockstein. We first check if the result is already admissible.
865        let i: u32 = x / 2;
866        let mut working_elt = self
867            .basis_element_from_index(n - (q * i as i32), idx)
868            .clone();
869
870        let b: u32 = working_elt.bocksteins & 1;
871        if working_elt.ps.is_empty() || i >= self.prime() * working_elt.ps[0] + b {
872            working_elt.ps.insert(0, i);
873            working_elt.bocksteins <<= 1;
874            working_elt.degree = n;
875
876            let out_idx = self.basis_element_to_index(&working_elt);
877            result.add_basis_element(out_idx, 1);
878            return result;
879        }
880
881        // In other cases, use the Adem relations.
882        let j: u32 = working_elt.ps[0];
883
884        let tail_idx = self.tail_of_basis_element_to_index(&mut working_elt, 1, q as u32);
885
886        if b == 0 {
887            // We use P^i P^j = \sum ... P^{i + j - k} P^k
888            for k in 0..=i / self.prime() {
889                let c = combinatorics::adem_relation_coefficient(self.prime(), i, j, k, 0, 0);
890                if c == 0 {
891                    continue;
892                }
893                if k == 0 {
894                    // We will never need working_elt in the future. We can leave it messed up
895                    working_elt.ps[0] = i + j;
896                    working_elt.degree = n;
897                    let new_index = self.basis_element_to_index(&working_elt);
898                    result.add_basis_element(new_index, c);
899                    continue;
900                }
901
902                let rest_reduced = &self.multiplication_table
903                    [(n - q * (i + j - k) as i32) as usize][2 * k as usize][tail_idx];
904                for (id, coeff) in rest_reduced.iter().enumerate() {
905                    let source = &table[2 * (i + j - k) as usize][id];
906                    result.add(source, (c * coeff) % self.prime());
907                }
908            }
909        } else {
910            // First treat the k = 0 case.
911            // \beta P^{i + j - k} P^i
912            let c = combinatorics::adem_relation_coefficient(self.prime(), i, j, 0, 1, 0);
913            working_elt.ps[0] = i + j;
914            working_elt.degree = n;
915            let index = self.basis_element_to_index(&working_elt);
916            result.add_basis_element(index, c);
917
918            // P^{i + j - k} \beta P^k. Check if there is \beta following P^k
919            if working_elt.bocksteins & 2 == 0 {
920                let c = combinatorics::adem_relation_coefficient(self.prime(), i, j, 0, 0, 1);
921                working_elt.bocksteins ^= 3; // flip the first two bits, so that it now ends with 10
922                let index = self.basis_element_to_index(&working_elt);
923                result.add_basis_element(index, c);
924            }
925
926            for k in 1..=i / self.prime() {
927                // \beta P^{i + j - k} P^k
928                let c = combinatorics::adem_relation_coefficient(self.prime(), i, j, k, 1, 0);
929                if c != 0 {
930                    let rest_reduced = &self.multiplication_table
931                        [(n - q * (i + j - k) as i32 - 1) as usize][2 * k as usize][tail_idx];
932                    for (id, coeff) in rest_reduced.iter().enumerate() {
933                        let source = &table[1 + 2 * (i + j - k) as usize][id];
934                        result.add(source, (c * coeff) % self.prime());
935                    }
936                }
937
938                // P^{i + j - k} \beta P^k
939                let c = combinatorics::adem_relation_coefficient(self.prime(), i, j, k, 0, 1);
940                if c != 0 {
941                    let rest_reduced = &self.multiplication_table
942                        [(n - q * (i + j - k) as i32) as usize][1 + 2 * k as usize][tail_idx];
943                    for (id, coeff) in rest_reduced.iter().enumerate() {
944                        let source = &table[2 * (i + j - k) as usize][id];
945                        result.add(source, (c * coeff) % self.prime());
946                    }
947                }
948            }
949        }
950        result
951    }

953    pub fn multiply_inner(
954        &self,
955        mut result: FpSliceMut,
956        coeff: u32,
957        r_degree: i32,
958        r_index: usize,
959        s_degree: i32,
960        s_index: usize,
961        excess: i32,
962    ) {
963        if coeff == 0 {
964            return;
965        }
966        assert!(r_index < self.dimension(r_degree));
967        assert!(s_index < self.dimension_unstable(s_degree, excess));
968
969        if s_degree == 0 {
970            if r_index >= self.dimension_unstable(r_degree, excess) {
971                return;
972            }
973            result.add_basis_element(r_index, coeff);
974            return;
975        }
976        let r = self.basis_element_from_index(r_degree, r_index);
977        let s = self.basis_element_from_index(s_degree, s_index);
978        let mut monomial = AdemBasisElement {
979            degree: r.degree + s.degree,
980            bocksteins: 0,
981            ps: Vec::with_capacity(r.ps.len() + s.ps.len()),
982            p_or_sq: self.prime() != 2,
983        };
984        if self.generic && (r.bocksteins >> r.ps.len()) & s.bocksteins & 1 == 1 {
985            // If there is a bockstein at the end of r and one at the beginning of s, these run into each other
986            // and the output is 0.
987            return;
988        } else if self.generic {
989            monomial.bocksteins = r.bocksteins;
990            monomial.bocksteins |= s.bocksteins << (r.ps.len());
991        }
992
993        monomial.ps.extend_from_slice(&r.ps);
994        monomial.ps.extend_from_slice(&s.ps);
995
996        let stop_early = true;
997        let index_to_check_for_admissibility = r.ps.len() as i32 - 1;
998        if self.generic {
999            // If r ends in a bockstein, we need to move it over because we consider
1000            // the monomial from right to left in chunks like bP^i. The b from the end of r gets donated
1001            // to the P from the beginning of s.
1002            let leading_degree = r.degree - ((r.bocksteins >> r.ps.len()) & 1) as i32;
1003            self.make_mono_admissible_generic(
1004                result,
1005                coeff,
1006                &mut monomial,
1007                index_to_check_for_admissibility,
1008                leading_degree,
1009                excess,
1010                stop_early,
1011            );
1012        } else {
1013            let leading_degree = r.degree;
1014            self.make_mono_admissible_2(
1015                result,
1016                &mut monomial,
1017                index_to_check_for_admissibility,
1018                leading_degree,
1019                excess,
1020                stop_early,
1021            );
1022        }
1023    }

1025    pub fn make_mono_admissible(
1026        &self,
1027        result: FpSliceMut,
1028        coeff: u32,
1029        monomial: &mut AdemBasisElement,
1030        excess: i32,
1031    ) {
1032        let q = if self.generic {
1033            2 * self.prime() - 2
1034        } else {
1035            1
1036        };
1037        let mut leading_degree = monomial.degree - (q * monomial.ps[monomial.ps.len() - 1]) as i32;
1038        let idx = monomial.ps.len() as i32 - 2;
1039        let stop_early = false;
1040        if self.generic {
1041            leading_degree -= ((monomial.bocksteins >> (monomial.ps.len() - 1)) & 1) as i32;
1042            self.make_mono_admissible_generic(
1043                result,
1044                coeff,
1045                monomial,
1046                idx,
1047                leading_degree,
1048                excess,
1049                stop_early,
1050            );
1051        } else {
1052            self.make_mono_admissible_2(result, monomial, idx, leading_degree, excess, stop_early);
1053        }
1054    }

1056    /// Reduce a Steenrod monomial at the prime 2.
1057    /// # Arguments:
1058    ///  * `algebra` - an Adem algebra. This would be a method of class AdemAlgebra.
1059    ///  * `result`  - Where we put the result
1060    ///  * `monomial` - a not necessarily admissible Steenrod monomial which we will reduce.
1061    ///    We destroy monomial->Ps.
1062    ///  * `idx` - the only index to check for inadmissibility in the input (we assume that we've gotten
1063    ///    our input as a product of two admissible sequences.)
1064    ///  * `leading_degree` - the degree of the squares between 0 and idx (so of length idx + 1)
1065    fn make_mono_admissible_2(
1066        &self,
1067        mut result: FpSliceMut,
1068        monomial: &mut AdemBasisElement,
1069        mut idx: i32,
1070        mut leading_degree: i32,
1071        excess: i32,
1072        stop_early: bool,
1073    ) {
1074        while idx < 0
1075            || idx as usize == monomial.ps.len() - 1
1076            || monomial.ps[idx as usize] >= 2 * monomial.ps[idx as usize + 1]
1077        {
1078            if idx < 0 || stop_early {
1079                // Admissible so write monomial to result.
1080                let idx = self.basis_element_to_index(monomial);
1081                // If excess is too large, quit. It's faster to check this by comparing idx to dimension
1082                // than to use fromIndex because fromIndex dereferences a hash map.
1083                if idx >= self.dimension_unstable(monomial.degree, excess) {
1084                    return;
1085                }
1086                result.add_basis_element(idx, 1);
1087                return;
1088            }
1089            leading_degree -= monomial.ps[idx as usize] as i32;
1090            idx -= 1;
1091        }
1092        let idx = idx as usize;
1093        let adm_idx = self.tail_of_basis_element_to_index(monomial, idx as u32 + 1, 1);
1094        let x = monomial.ps[idx] as i32;
1095        let tail_degree = monomial.degree - leading_degree + x;
1096        let reduced_tail = &self.multiplication_table[tail_degree as usize][x as usize][adm_idx];
1097
1098        let mut new_monomial = AdemBasisElement {
1099            degree: monomial.degree,
1100            bocksteins: 0,
1101            ps: monomial.ps[0..idx].to_vec(),
1102            p_or_sq: self.prime() != 2,
1103        };
1104
1105        for (it_idx, _value) in reduced_tail.iter_nonzero() {
1106            let cur_tail_basis_elt = self.basis_element_from_index(tail_degree, it_idx);
1107            new_monomial.ps.truncate(idx);
1108            new_monomial.ps.extend_from_slice(&cur_tail_basis_elt.ps);
1109            self.make_mono_admissible_2(
1110                result.copy(),
1111                &mut new_monomial,
1112                idx as i32 - 1,
1113                leading_degree - x,
1114                excess,
1115                stop_early,
1116            );
1117        }
1118    }

1120    fn make_mono_admissible_generic(
1121        &self,
1122        mut result: FpSliceMut,
1123        coeff: u32,
1124        monomial: &mut AdemBasisElement,
1125        mut idx: i32,
1126        mut leading_degree: i32,
1127        excess: i32,
1128        stop_early: bool,
1129    ) {
1130        let p = self.prime();
1131        let q = 2 * p - 2;
1132        // Check for admissibility
1133        let b1 = if idx >= 0 {
1134            (monomial.bocksteins >> idx) & 1
1135        } else {
1136            0
1137        };
1138        let b2 = (monomial.bocksteins >> (idx + 1)) & 1;
1139        while idx < 0
1140            || idx == monomial.ps.len() as i32 - 1
1141            || monomial.ps[idx as usize] >= p * monomial.ps[idx as usize + 1] + b2
1142        {
1143            if idx < 0 || stop_early {
1144                // Admissible so write monomial to result.
1145                let idx = self.basis_element_to_index(monomial);
1146                if idx >= self.dimension_unstable(monomial.degree, excess) {
1147                    return;
1148                }
1149                result.add_basis_element(idx, coeff);
1150                return;
1151            }
1152            leading_degree -= (q * monomial.ps[idx as usize]) as i32;
1153            leading_degree -= ((monomial.bocksteins >> idx) & 1) as i32;
1154            idx -= 1;
1155        }
1156        let idx = idx as usize;
1157        let adm_idx = self.tail_of_basis_element_to_index(monomial, idx as u32 + 1, q);
1158        // Notice how much we avoid bockstein twiddling here. It's all hidden in multiplication_table =)
1159        let x = monomial.ps[idx];
1160        let bx = (x << 1) + b1;
1161        let tail_degree = monomial.degree - leading_degree + (q * x + b1) as i32;
1162        let reduced_tail = &self.multiplication_table[tail_degree as usize][bx as usize][adm_idx];
1163        let mut new_monomial = AdemBasisElement {
1164            degree: monomial.degree,
1165            bocksteins: 0,
1166            ps: monomial.ps[0..idx].to_vec(),
1167            p_or_sq: self.prime() != 2,
1168        };
1169
1170        for (it_idx, it_value) in reduced_tail.iter_nonzero() {
1171            let cur_tail_basis_elt = self.basis_element_from_index(tail_degree, it_idx);
1172            new_monomial.ps.truncate(idx);
1173            new_monomial.ps.extend_from_slice(&cur_tail_basis_elt.ps);
1174            new_monomial.bocksteins = monomial.bocksteins & ((1 << idx) - 1);
1175            new_monomial.bocksteins |= cur_tail_basis_elt.bocksteins << idx;
1176            let new_leading_degree = leading_degree - (q * x + b1) as i32;
1177            self.make_mono_admissible_generic(
1178                result.copy(),
1179                (coeff * it_value) % p,
1180                &mut new_monomial,
1181                idx as i32 - 1,
1182                new_leading_degree,
1183                excess,
1184                stop_early,
1185            );
1186        }
1187    }

1189    fn decompose_basis_element_2(
1190        &self,
1191        degree: i32,
1192        idx: usize,
1193    ) -> Vec<(u32, (i32, usize), (i32, usize))> {
1194        let b = self.basis_element_from_index(degree, idx);
1195        if b.ps.len() > 1 {
1196            let degree_first = b.ps[0] as i32;
1197            let degree_rest = b.degree - b.ps[0] as i32;
1198            let ps_rest = b.ps[1..].to_vec();
1199            let idx_first = self.basis_element_to_index(&AdemBasisElement {
1200                degree: degree_first,
1201                bocksteins: 0,
1202                ps: vec![b.ps[0]],
1203                p_or_sq: self.prime() != 2,
1204            });
1205            let idx_rest = self.basis_element_to_index(&AdemBasisElement {
1206                degree: degree_rest,
1207                bocksteins: 0,
1208                ps: ps_rest,
1209                p_or_sq: self.prime() != 2,
1210            });
1211            return vec![(1, (degree_first, idx_first), (degree_rest, idx_rest))];
1212        }
1213        let sq = b.ps[0];
1214        let tz = sq.trailing_zeros();
1215        let first_sq = 1 << tz;
1216        let second_sq = sq ^ first_sq;
1217        let first_degree = first_sq as i32;
1218        let second_degree = second_sq as i32;
1219        let first_idx = self.basis_element_to_index(&AdemBasisElement {
1220            degree: first_degree,
1221            bocksteins: 0,
1222            ps: vec![first_sq],
1223            p_or_sq: self.prime() != 2,
1224        });
1225        let second_idx = self.basis_element_to_index(&AdemBasisElement {
1226            degree: second_degree,
1227            bocksteins: 0,
1228            ps: vec![second_sq],
1229            p_or_sq: self.prime() != 2,
1230        });
1231        let mut out_vec = FpVector::new(fp::prime::TWO, self.dimension(degree));
1232        self.multiply_basis_elements(
1233            out_vec.as_slice_mut(),
1234            1,
1235            first_degree,
1236            first_idx,
1237            second_degree,
1238            second_idx,
1239        );
1240        out_vec.set_entry(idx, 0);
1241        let mut result = vec![(1, (first_degree, first_idx), (second_degree, second_idx))];
1242        for (i, _v) in out_vec.iter_nonzero() {
1243            result.extend(self.decompose_basis_element_2(degree, i));
1244        }
1245        result
1246    }
1247
1248    fn decompose_basis_element_generic(
1249        &self,
1250        degree: i32,
1251        idx: usize,
1252    ) -> Vec<(u32, (i32, usize), (i32, usize))> {
1253        let p = self.prime();
1254        let b = self.basis_element_from_index(degree, idx);
1255        let leading_bockstein_idx = 1; // << (b.ps.len());
1256        if b.bocksteins & leading_bockstein_idx != 0 {
1257            let mut b_new = b.clone();
1258            b_new.bocksteins ^= leading_bockstein_idx;
1259            b_new.degree -= 1;
1260            let first_degree = 1;
1261            let first_idx = 0;
1262            let rest_degree = b_new.degree;
1263            let rest_idx = self.basis_element_to_index(&b_new);
1264            return vec![(1, (first_degree, first_idx), (rest_degree, rest_idx))];
1265        }
1266        if b.bocksteins != 0 || b.ps.len() != 1 {
1267            let first_degree = (b.ps[0] * 2 * (p - 1)) as i32;
1268            let rest_degree = b.degree - first_degree;
1269            let ps_first = vec![b.ps[0]];
1270            let ps_rest = b.ps[1..].to_vec();
1271            let first = AdemBasisElement {
1272                degree: first_degree,
1273                bocksteins: 0,
1274                ps: ps_first,
1275                p_or_sq: self.prime() != 2,
1276            };
1277            let rest = AdemBasisElement {
1278                degree: rest_degree,
1279                bocksteins: b.bocksteins >> 1,
1280                ps: ps_rest,
1281                p_or_sq: self.prime() != 2,
1282            };
1283            let first_idx = self.basis_element_to_index(&first);
1284            let rest_idx = self.basis_element_to_index(&rest);
1285            return vec![(1, (first_degree, first_idx), (rest_degree, rest_idx))];
1286        }
1287
1288        let sq = b.ps[0];
1289        let mut pow = 1;
1290        {
1291            let mut temp_sq = sq;
1292            while temp_sq.is_multiple_of(p.as_u32()) {
1293                temp_sq /= p;
1294                pow *= p;
1295            }
1296        }
1297
1298        let first_sq = pow;
1299        let second_sq = sq - first_sq;
1300        let first_degree = (first_sq * 2 * (p - 1)) as i32;
1301        let second_degree = (second_sq * 2 * (p - 1)) as i32;
1302        let first_idx = self.basis_element_to_index(&AdemBasisElement {
1303            degree: first_degree,
1304            bocksteins: 0,
1305            ps: vec![first_sq],
1306            p_or_sq: self.prime() != 2,
1307        });
1308        let second_idx = self.basis_element_to_index(&AdemBasisElement {
1309            degree: second_degree,
1310            bocksteins: 0,
1311            ps: vec![second_sq],
1312            p_or_sq: self.prime() != 2,
1313        });
1314        let mut out_vec = FpVector::new(p, self.dimension(degree));
1315        self.multiply_basis_elements(
1316            out_vec.as_slice_mut(),
1317            1,
1318            first_degree,
1319            first_idx,
1320            second_degree,
1321            second_idx,
1322        );
1323        let mut result = Vec::new();
1324        let c = out_vec.entry(idx);
1325        assert!(c != 0);
1326        let c_inv = fp::prime::inverse(p, p - c);
1327        result.push((
1328            ((p - 1) * c_inv) % p,
1329            (first_degree, first_idx),
1330            (second_degree, second_idx),
1331        ));
1332        out_vec.set_entry(idx, 0);
1333        for (i, v) in out_vec.iter_nonzero() {
1334            let (c, t1, t2) = self.decompose_basis_element_generic(degree, i)[0];
1335            result.push(((c_inv * c * v) % p, t1, t2));
1336        }
1337        result
1338    }

1340    pub fn beps_pn(&self, e: u32, x: u32) -> (i32, usize) {
1341        if x == 0 && e == 1 {
1342            return (1, 0);
1343        } else if x == 0 {
1344            return (0, 0);
1345        }
1346
1347        let p = self.prime();
1348        let q = if self.generic { 2 * p - 2 } else { 1 };
1349        let degree = (x * q + e) as i32;
1350        let index = self.basis_element_to_index(&AdemBasisElement {
1351            degree,
1352            bocksteins: e,
1353            ps: vec![x],
1354            p_or_sq: self.prime() != 2,
1355        });
1356        (degree, index)
1357    }
1358}

1360impl AdemAlgebra {
1361    fn generate_excess_table(&self, max_degree: i32) {
1362        let p = self.prime();
1363        self.excess_table.extend(max_degree as usize, |n| {
1364            let mut new_entry = Vec::with_capacity(n);
1365            let mut cur_excess = 0;
1366            for (i, elt) in self.basis_table[n].iter().enumerate() {
1367                let excess = elt.excess(p);
1368                for _ in cur_excess..excess {
1369                    new_entry.push(i);
1370                }
1371                cur_excess = excess;
1372            }
1373            let dim = self.dimension(n as i32);
1374            for _ in cur_excess..n as u32 {
1375                new_entry.push(dim);
1376            }
1377            new_entry
1378        });
1379    }
1380}

1382impl Bialgebra for AdemAlgebra {
1383    fn decompose(&self, op_deg: i32, op_idx: usize) -> Vec<(i32, usize)> {
1384        let elt = &self.basis_table[op_deg as usize][op_idx];
1385        if self.generic {
1386            let mut result: Vec<(i32, usize)> = Vec::with_capacity(elt.ps.len() * 2 + 1);
1387            let mut bockstein = elt.bocksteins;
1388            for item in &elt.ps {
1389                if bockstein & 1 == 1 {
1390                    result.push((1, 0));
1391                }
1392                bockstein >>= 1;
1393                result.push(self.beps_pn(0, *item));
1394            }
1395            if bockstein & 1 == 1 {
1396                result.push((1, 0));
1397            }
1398            result.reverse();
1399            result
1400        } else {
1401            elt.ps
1402                .iter()
1403                .rev()
1404                .map(|i| (*i as i32, 0))
1405                .collect::<Vec<_>>()
1406        }
1407    }
1408
1409    fn coproduct(&self, op_deg: i32, op_idx: usize) -> Vec<(i32, usize, i32, usize)> {
1410        if self.generic {
1411            if op_deg == 1 {
1412                vec![(1, 0, 0, 0), (0, 0, 1, 0)]
1413            } else {
1414                let q = self.prime() * 2 - 2;
1415                let op_deg = op_deg as u32;
1416                assert_eq!(op_deg % q, 0);
1417
1418                (0..=op_deg / q)
1419                    .map(|j| {
1420                        let first = self.beps_pn(0, j);
1421                        let last = self.beps_pn(0, op_deg / q - j);
1422                        (first.0, first.1, last.0, last.1)
1423                    })
1424                    .collect::<Vec<_>>()
1425            }
1426        } else {
1427            assert_eq!(op_idx, 0);
1428            (0..=op_deg)
1429                .map(|j| (j, 0, op_deg - j, 0))
1430                .collect::<Vec<_>>()
1431        }
1432    }
1433}
