18pub(crate) fn module_gens_from_json(
19    gens: &serde_json::Value,
20) -> (
21    bivec::BiVec<usize>,
22    bivec::BiVec<Vec<String>>,
23    impl for<'a> Fn(&'a str) -> anyhow::Result<(i32, usize)> + '_,
24) {
25    use anyhow::anyhow;
26    let gens = gens.as_object().unwrap();
27
28    let degrees = gens
29        .iter()
30        .map(|(_, x)| x.as_i64().unwrap() as i32)
31        .collect::<Vec<_>>();
32
33    let min_degree = degrees.iter().copied().min().unwrap_or(0);
34    let max_degree = degrees.iter().copied().max().unwrap_or(-1) + 1;
35
36    let mut gen_to_idx = rustc_hash::FxHashMap::default();
37    let mut graded_dimension = bivec::BiVec::with_capacity(min_degree, max_degree);
38    let mut gen_names = bivec::BiVec::with_capacity(min_degree, max_degree);
39
40    for _ in min_degree..max_degree {
41        graded_dimension.push(0);
42        gen_names.push(vec![]);
43    }
44
45    for (name, degree) in gens {
46        let degree = degree.as_i64().unwrap() as i32;
47        gen_names[degree].push(name.clone());
48        gen_to_idx.insert(name.clone(), (degree, graded_dimension[degree]));
49        graded_dimension[degree] += 1;
50    }
51    (graded_dimension, gen_names, move |g| {
52        gen_to_idx
53            .get(g)
54            .copied()
55            .ok_or_else(|| anyhow!("Invalid generator: {g}"))
56    })
57}
