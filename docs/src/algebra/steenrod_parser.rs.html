22#[derive(Debug, Clone)]
23pub enum AlgebraBasisElt {
24    AList(Vec<BocksteinOrSq>), // Admissible list.
25    PList(PPart),
26    P(u32),
27    Q(u32),
28}

30#[derive(Debug, Clone)]
31pub enum AlgebraNode {
32    Product(Box<Self>, Box<Self>),
33    Sum(Box<Self>, Box<Self>),
34    BasisElt(AlgebraBasisElt),
35    Scalar(i32),
36}

38pub type ModuleNode = Vec<(AlgebraNode, String)>;

40/// Pad both ends with whitespace
41pub(crate) fn space<'a, O, E: ParseError<&'a str>, F: Parser<&'a str, O, E>>(
42    f: F,
43) -> impl FnMut(&'a str) -> IResultBase<&'a str, O, E> {
44    delimited(space0, f, space0)
45}
46
47/// Surround with brackets
48pub(crate) fn brackets<'a, O, E: ParseError<&'a str>, F: Parser<&'a str, O, E>>(
49    f: F,
50) -> impl FnMut(&'a str) -> IResultBase<&'a str, O, E> {
51    delimited(char('('), f, char(')'))
52}

54pub(crate) fn digits<T: FromStr + Copy>(i: &str) -> IResult<&str, T> {
55    map_res(space(digit), FromStr::from_str)(i)
56}

58pub(crate) fn p_or_sq(i: &str) -> IResult<&str, &str> {
59    alt((tag("P"), tag("Sq")))(i)
60}

62fn fold_separated<I: Clone, OS, O, E>(
63    mut sep: impl Parser<I, OS, E>,
64    mut f: impl Parser<I, O, E>,
65    mut acc: impl FnMut(O, O) -> O,
66) -> impl FnMut(I) -> IResultBase<I, O, E> {
67    move |i: I| {
68        let (mut i, mut res) = f.parse(i)?;
69        loop {
70            match sep.parse(i.clone()) {
71                Err(nom::Err::Error(_)) => return Ok((i, res)),
72                Err(e) => return Err(e),
73                Ok((i1, _)) => match f.parse(i1.clone()) {
74                    Err(nom::Err::Error(_)) => return Ok((i, res)),
75                    Err(e) => return Err(e),
76                    Ok((i2, o)) => {
77                        i = i2;
78                        res = acc(res, o);
79                    }
80                },
81            }
82        }
83    }
84}

86#[derive(Clone, Debug, Copy)]
87pub enum BocksteinOrSq {
88    Bockstein,
89    Sq(u32),
90}
91
92impl BocksteinOrSq {
93    pub(crate) fn to_adem_basis_elt(self, q: i32) -> AdemBasisElement {
94        match self {
95            Self::Bockstein => {
96                if q == 1 {
97                    AdemBasisElement {
98                        degree: 1,
99                        bocksteins: 0,
100                        ps: vec![1],
101                        p_or_sq: false,
102                    }
103                } else {
104                    AdemBasisElement {
105                        degree: 1,
106                        bocksteins: 1,
107                        ps: vec![],
108                        p_or_sq: true,
109                    }
110                }
111            }
112            Self::Sq(x) => AdemBasisElement {
113                degree: x as i32 * q,
114                bocksteins: 0,
115                ps: vec![x],
116                p_or_sq: q != 1,
117            },
118        }
119    }
120}

122fn algebra_generator(i: &str) -> IResult<&str, AlgebraBasisElt> {
123    alt((
124        map(char('b'), |_| AlgebraBasisElt::Q(0)),
125        map(preceded(char('Q'), digits), AlgebraBasisElt::Q),
126        map(preceded(p_or_sq, digits), AlgebraBasisElt::P),
127        map(
128            alt((
129                preceded(p_or_sq, brackets(separated_list1(char(','), digits))),
130                preceded(char('M'), brackets(many0(digits))),
131            )),
132            AlgebraBasisElt::PList,
133        ),
134        map(
135            preceded(
136                char('A'),
137                brackets(many0(alt((
138                    map(char('b'), |_| BocksteinOrSq::Bockstein),
139                    map(digits, BocksteinOrSq::Sq),
140                )))),
141            ),
142            AlgebraBasisElt::AList,
143        ),
144    ))(i)
145}
146
147fn scalar(i: &str) -> IResult<&str, i32> {
148    alt((
149        digits,
150        preceded(char('+'), digits),
151        map(preceded(char('-'), digits), |x: i32| -x),
152    ))(i)
153}
154
155fn algebra_factor(i: &str) -> IResult<&str, AlgebraNode> {
156    space(alt((
157        map(algebra_generator, AlgebraNode::BasisElt),
158        map(scalar, AlgebraNode::Scalar),
159        brackets(algebra_expr),
160    )))(i)
161}
162
163fn algebra_term(i: &str) -> IResult<&str, AlgebraNode> {
164    let (i, sign) = opt(alt((char('+'), char('-'))))(i)?;
165
166    let (i, mut res) = fold_separated(char('*'), algebra_factor, |acc, val| {
167        AlgebraNode::Product(Box::new(acc), Box::new(val))
168    })(i)?;
169
170    if let Some('-') = sign {
171        res = AlgebraNode::Product(Box::new(AlgebraNode::Scalar(-1)), Box::new(res));
172    }
173    Ok((i, res))
174}
175
176fn algebra_expr(i: &str) -> IResult<&str, AlgebraNode> {
177    fold_separated(
178        peek(alt((char('+'), char('-')))),
179        space(algebra_term),
180        |acc, val| AlgebraNode::Sum(Box::new(acc), Box::new(val)),
181    )(i)
182}

184fn module_generator(i: &str) -> IResult<&str, String> {
185    let (rest, (a, more_str)) = pair(alpha1, alphanumeric0)(i)?;
186    if a.starts_with("Sq") || a.starts_with('P') || a.starts_with('Q') {
187        return Err(nom::Err::Failure(VerboseError {
188            errors: vec![(
189                &i[0..a.len()],
190                VerboseErrorKind::Context(
191                    "Module generators are not allowed to start with P, Q, or Sq",
192                ),
193            )],
194        }));
195    }
196    Ok((rest, a.to_string() + more_str))
197}
198
199fn module_term(i: &str) -> IResult<&str, ModuleNode> {
200    use AlgebraNode::*;
201
202    let (i, prefix) = opt(alt((
203        map(pair(space(algebra_term), char('*')), |(a, _)| a),
204        map(char('-'), |_| Scalar(-1)),
205        map(char('+'), |_| Scalar(1)),
206    )))(i)
207    .unwrap();
208
209    match space(module_generator)(i) {
210        Ok((i, gen)) => return Ok((i, vec![(prefix.unwrap_or(Scalar(1)), gen)])),
211        Err(nom::Err::Error(_)) => (),
212        Err(e) => return Err(e),
213    }
214
215    let (i, expr) = context("Parsing bracketed expression", space(brackets(module_expr)))(i)?;
216    Ok((
217        i,
218        match prefix {
219            Some(a) => expr
220                .into_iter()
221                .map(|(b, v)| (Product(Box::new(a.clone()), Box::new(b)), v))
222                .collect(),
223            None => expr,
224        },
225    ))
226}
227
228fn module_expr(i: &str) -> IResult<&str, ModuleNode> {
229    fold_separated(
230        peek(alt((char('+'), char('-')))),
231        module_term,
232        |mut a, b| {
233            a.extend_from_slice(&b);
234            a
235        },
236    )(i)
237}

239fn convert_error(i: &str) -> impl FnOnce(nom::Err<VerboseError<&str>>) -> anyhow::Error + '_ {
240    move |err| {
241        anyhow!(match err {
242            nom::Err::Error(e) | nom::Err::Failure(e) => nom::error::convert_error(i, e),
243            _ => format!("{err:#}"),
244        })
245    }
246}
247
248pub fn parse_algebra(i: &str) -> anyhow::Result<AlgebraNode> {
249    let (rest, parse_tree) = algebra_expr(i)
250        .map_err(convert_error(i))
251        .with_context(|| format!("Error when parsing algebra string {i}"))?;
252    if rest.is_empty() {
253        Ok(parse_tree)
254    } else {
255        Err(anyhow!(
256            "Failed to consume all of input. Remaining: '{rest}'"
257        ))
258    }
259}
260
261pub fn parse_module(i: &str) -> anyhow::Result<ModuleNode> {
262    let (rest, parse_tree) = module_expr(i)
263        .map_err(convert_error(i))
264        .with_context(|| format!("Error when parsing module string {i}"))?;
265    if rest.is_empty() {
266        Ok(parse_tree)
267    } else {
268        Err(anyhow!(
269            "Failed to consume all of input. Remaining: '{rest}'"
270        ))
271    }
272}
