13/// Execute a function on a range of bidegrees, possibly in parallel.
14///
15/// Given a function `f(s, t)`, compute it for every `s` in `[min_s, max_s]` and every `t` in
16/// `[min_t, max_t(s)]`.  Further, we only compute `f(s, t)` when `f(s - 1, t')` has been computed
17/// for all `t' < t`.
18///
19/// The function `f` should return a range starting from t and ending at the largest `T` such that
20/// `f(s, t')` has already been computed for every `t' < T`.
21///
22/// While `iter_s_t` could have had kept track of that data, it is usually the case that `f` would
23/// compute something and write it to a `OnceBiVec`, and
24/// [`OnceBiVec::push_ooo`](once::OnceBiVec::push_ooo) would return this range for us.
25///
26/// This uses [`maybe_rayon`] under the hood, and `f` should feel free to use further parallelism.
27///
28/// # Arguments:
29///  - `max_s`: This is exclusive
30///  - `max_t`: This is exclusive
31pub fn iter_s_t<T: Sync>(
32    f: &(impl Fn(Bidegree) -> std::ops::Range<i32> + Sync),
33    min: Bidegree,
34    max: BidegreeRange<T>,
35) {
36    // Track `tracing` spans correctly
37    let tracing_span = tracing::Span::current();
38    let f = &|b| {
39        let _tracing_guard = tracing_span.enter();
40        f(b)
41    };
42
43    maybe_rayon::scope(|scope| {
44        // Rust does not support recursive closures, so we have to pass everything along as
45        // arguments.
46        fn run<'a, S: Sync>(
47            scope: &maybe_rayon::Scope<'a>,
48            f: &'a (impl Fn(Bidegree) -> std::ops::Range<i32> + Sync + 'a),
49            max: BidegreeRange<'a, S>,
50            current: Bidegree,
51        ) {
52            let mut ret = f(current);
53            if current.s() + 1 < max.s() {
54                ret.start += 1;
55                ret.end = std::cmp::min(ret.end + 1, max.t(current.s() + 1));
56
57                if !ret.is_empty() {
58                    // We spawn a new scope to avoid recursion, which may blow the stack
59                    scope.spawn(move |scope| {
60                        ret.into_maybe_par_iter()
61                            .for_each(|t| run(scope, f, max, Bidegree::s_t(current.s() + 1, t)));
62                    });
63                }
64            }
65        }
66
67        maybe_rayon::join(
68            || {
69                (min.t()..max.t(min.s()))
70                    .into_maybe_par_iter()
71                    .for_each(|t| run(scope, f, max, Bidegree::s_t(min.s(), t)))
72            },
73            || {
74                (min.s() + 1..max.s())
75                    .into_maybe_par_iter()
76                    .for_each(|s| run(scope, f, max, Bidegree::s_t(s, min.t())))
77            },
78        );
79    });
80}

83mod tests {
84    use fp::{prime::ValidPrime, vector::FpVector};
85
86    use super::{Bidegree, BidegreeElement, BidegreeGenerator};

88    #[test]
89    fn test_bidegree_generator_try_from_element() {
90        let b = Bidegree::n_s(23, 9);
91        let mut vec = FpVector::new(ValidPrime::new(2), 2);
92        vec.set_entry(1, 1);
93        let h1_pd0 = BidegreeElement::new(b, vec.clone());
94        assert_eq!(Ok(BidegreeGenerator::new(b, 1)), h1_pd0.try_into());
95        vec.set_entry(0, 1);
96        let h0_squared_i = BidegreeElement::new(b, vec.clone());
97        assert_eq!(
98            Result::<BidegreeGenerator, ()>::Err(()),
99            h0_squared_i.try_into()
100        );
101    }
