12/// An element of a bigraded vector space. Most commonly used to index elements of spectral
13/// sequences.
14#[derive(Debug, Clone, PartialEq, Eq, Hash, Serialize, Deserialize)]
15pub struct BidegreeElement {
16    /// Bidegree of the element
17    degree: Bidegree,
18    /// Representing vector
19    vec: FpVector,
20}
21
22impl BidegreeElement {
23    pub fn new(degree: Bidegree, vec: FpVector) -> Self {
24        Self { degree, vec }
25    }
26
27    pub fn s(&self) -> i32 {
28        self.degree.s()
29    }
30
31    pub fn t(&self) -> i32 {
32        self.degree.t()
33    }
34
35    pub fn degree(&self) -> Bidegree {
36        self.degree
37    }
38
39    pub fn n(&self) -> i32 {
40        self.degree.n()
41    }
42
43    pub fn vec(&self) -> FpSlice<'_> {
44        self.vec.as_slice()
45    }
46
47    pub fn into_vec(self) -> FpVector {
48        self.vec
49    }
50
51    /// Get the string representation of the element as a linear combination of generators. For
52    /// example, an element in bidegree `(n,s)` with vector `[0,2,1]` will be printed as `2 x_(n, s,
53    /// 1) + x_(n, s, 2)`.
54    pub fn to_basis_string(&self) -> String {
55        self.vec
56            .iter_nonzero()
57            .map(|(i, v)| {
58                let gen = BidegreeGenerator::new(self.degree(), i);
59                let coeff_str = if v != 1 {
60                    format!("{v} ")
61                } else {
62                    String::new()
63                };
64                format!("{coeff_str}x_{gen}")
65            })
66            .collect::<Vec<_>>()
67            .join(" + ")
68    }
69
70    /// An algebra-aware string representation. This assumes that the element belongs to `module`,
71    /// and uses the string representation of its underlying algebra's operations.
72    pub fn to_string_module<const U: bool, A: MuAlgebra<U>>(
73        &self,
74        module: &MuFreeModule<U, A>,
75        compact: bool,
76    ) -> String {
77        self.vec
78            .iter_nonzero()
79            .map(|(i, c)| {
80                let coeff_str = if c != 1 {
81                    format!("{c} ")
82                } else {
83                    String::new()
84                };
85                let opgen = module.index_to_op_gen(self.t(), i);
86                let mut op_str = module
87                    .algebra()
88                    .basis_element_to_string(opgen.operation_degree, opgen.operation_index);
89                op_str = if op_str != "1" {
90                    format!("{op_str} ")
91                } else {
92                    String::new()
93                };
94                let gen = BidegreeGenerator::s_t(
95                    self.s() - 1,
96                    opgen.generator_degree,
97                    opgen.generator_index,
98                );
99                let gen_str = if compact {
100                    format!("x_{gen:#}")
101                } else {
102                    format!("x_{gen}")
103                };
104                format!("{coeff_str}{op_str}{gen_str}")
105            })
106            .collect::<Vec<_>>()
107            .join(" + ")
108    }
109}
110
111impl Display for BidegreeElement {
112    fn fmt(&self, f: &mut Formatter) -> fmt::Result {
113        if f.alternate() {
114            write!(f, "({},{}){}", self.n(), self.s(), self.vec())
115        } else {
116            write!(f, "({}, {}, {})", self.n(), self.s(), self.vec())
117        }
118    }
