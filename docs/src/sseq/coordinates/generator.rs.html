8/// A *basis* element of a bigraded vector space. Most commonly used to index elements of spectral
9/// sequences.
10#[derive(Default, Debug, Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize)]
11pub struct BidegreeGenerator {
12    /// Bidegree of the element
13    degree: Bidegree,
14    /// Position in the canonical basis for this bidegree
15    idx: usize,
16}
17
18impl BidegreeGenerator {
19    pub fn new<T: Into<Bidegree>>(degree: T, idx: usize) -> Self {
20        Self {
21            degree: degree.into(),
22            idx,
23        }
24    }
25
26    pub fn s_t(s: i32, t: i32, idx: usize) -> Self {
27        Self::new(Bidegree::s_t(s, t), idx)
28    }
29
30    pub fn n_s(n: i32, s: i32, idx: usize) -> Self {
31        Self::new(Bidegree::n_s(n, s), idx)
32    }
33
34    pub fn s(&self) -> i32 {
35        self.degree.s()
36    }
37
38    pub fn t(&self) -> i32 {
39        self.degree.t()
40    }
41
42    pub fn degree(&self) -> Bidegree {
43        self.degree
44    }
45
46    pub fn n(&self) -> i32 {
47        self.degree.n()
48    }
49
50    pub fn x(&self) -> i32 {
51        self.degree.x()
52    }
53
54    pub fn y(&self) -> i32 {
55        self.degree.y()
56    }
57
58    pub fn idx(&self) -> usize {
59        self.idx
60    }
61
62    pub fn into_element(self, p: ValidPrime, ambient: usize) -> BidegreeElement {
63        let mut vec = FpVector::new(p, ambient);
64        vec.set_entry(self.idx, 1);
65        BidegreeElement::new(self.degree, vec)
66    }
67}
68
69impl Display for BidegreeGenerator {
70    fn fmt(&self, f: &mut Formatter) -> fmt::Result {
71        if f.alternate() {
72            // Compact representation
73            write!(f, "({},{},{})", self.n(), self.s(), self.idx())
74        } else {
75            write!(f, "({}, {}, {})", self.n(), self.s(), self.idx())
76        }
77    }
78}

80impl From<(Bidegree, usize)> for BidegreeGenerator {
81    fn from(tuple: (Bidegree, usize)) -> Self {
82        Self::new(tuple.0, tuple.1)
83    }
84}

86impl TryFrom<BidegreeElement> for BidegreeGenerator {
87    type Error = ();
88
89    fn try_from(value: BidegreeElement) -> Result<Self, Self::Error> {
90        if value.vec().iter().sum::<u32>() == 1 {
91            let (idx, _) = value.vec().iter_nonzero().next().unwrap();
92            Ok((value.degree(), idx).into())
93        } else {
94            Err(())
95        }
96    }
