28pub trait Backend {
29    type Error;
30
31    /// If the backend writes to a file, this is the extension commonly taken by the file type
32    const EXT: &'static str = "";
33
34    fn header(&mut self, max: Bidegree) -> Result<(), Self::Error>;
35    fn line(&mut self, start: Bidegree, end: Bidegree, style: &str) -> Result<(), Self::Error>;
36
37    fn text(
38        &mut self,
39        b: Bidegree,
40        content: impl Display,
41        orientation: Orientation,
42    ) -> Result<(), Self::Error>;
43
44    // We don't use BidegreeGenerator here because `n` represents the order of a bidegree instead of
45    // an index of an element within a bidegree
46    fn node(&mut self, b: Bidegree, n: usize) -> Result<(), Self::Error>;
47
48    fn structline(
49        &mut self,
50        source: BidegreeGenerator,
51        target: BidegreeGenerator,
52        style: Option<&str>,
53    ) -> Result<(), Self::Error>;
54
55    fn init(&mut self, max: Bidegree) -> Result<(), Self::Error> {
56        self.header(max)?;
57
58        for x in 0..=max.x() {
59            let on_x_axis = Bidegree::x_y(x, 0);
60            self.line(
61                on_x_axis,
62                Bidegree::x_y(x, max.y()),
63                if x % 4 == 0 { "major-grid" } else { "grid" },
64            )?;
65            if x % 4 == 0 {
66                self.text(on_x_axis, x, Orientation::Below)?;
67            }
68        }
69        for y in 0..=max.y() {
70            let on_y_axis = Bidegree::x_y(0, y);
71            self.line(
72                on_y_axis,
73                Bidegree::x_y(max.x(), y),
74                if y % 4 == 0 { "major-grid" } else { "grid" },
75            )?;
76            if y % 4 == 0 {
77                self.text(on_y_axis, y, Orientation::Left)?;
78            }
79        }
80        Ok(())
81    }
82
83    fn structline_matrix(
84        &mut self,
85        source: Bidegree,
86        target: Bidegree,
87        matrix: Vec<Vec<u32>>,
88        class: Option<&str>,
89    ) -> Result<(), Self::Error> {
90        for (k, row) in matrix.into_iter().enumerate() {
91            for (l, v) in row.into_iter().enumerate() {
92                if v != 0 {
93                    self.structline(
94                        BidegreeGenerator::new(source, k),
95                        BidegreeGenerator::new(target, l),
96                        class,
97                    )?;
98                }
99            }
100        }
101        Ok(())
102    }
103}

105pub struct SvgBackend<T: io::Write> {
106    out: T,
107    max: Bidegree,
108    num_nodes: HashMap<Bidegree, usize>,

111impl<T: io::Write> SvgBackend<T> {
112    const GRID_WIDTH: i32 = 20;
113    const MARGIN: i32 = 30;
114    const STYLES: &'static str = r#"
115    circle {
116        fill: black;
117    }
118    .structline {
119        stroke: black;
120        fill: none;
121    }
122    .d2 {
123        stroke: blue;
124    }
125    .major-grid {
126        stroke: black;
127        opacity: 20%;
128        shape-rendering: crispEdges;
129        fill: none;
130    }
131    .grid {
132        stroke: black;
133        opacity: 10%;
134        shape-rendering: crispEdges;
135        fill: none;
136    }
137    .x-label {
138     text-anchor: middle;
139     dominant-baseline: text-before-edge;
140    }
141    .y-label {
142     text-anchor: end;
143     dominant-baseline: middle;
144    }
145    "#;
146
147    /// Print the legend for node patterns
148    pub fn legend(mut out: T) -> std::io::Result<()> {
149        writeln!(
150            out,
151            r#"<svg version = "1.1" width="{}" height="100" xmlns="http://www.w3.org/2000/svg">"#,
152            PATTERNS.len() * 100,
153        )?;
154
155        for (n, (_, row)) in PATTERNS.iter().enumerate() {
156            writeln!(
157                out,
158                r#"<rect fill="none" stroke="black" width="80" height="80" y="10" x="{x}"/>"#,
159                x = 10 + n * 100
160            )?;
161            for (k, offset) in row[0..=n].iter().enumerate() {
162                writeln!(
163                    out,
164                    r#"<text x="{x}" y="{y}" text-anchor="middle" dominant-baseline="middle">{k}</text>"#,
165                    x = 50.0 + n as f32 * 100.0 + offset.0 * 6.0,
166                    y = 50.0 + offset.1 * 6.0,
167                )?;
168            }
169        }
170        writeln!(out, r#"</svg>"#,)?;
171
172        Ok(())
173    }
174
175    /// Returns r, x, y
176    fn get_coords(&self, g: BidegreeGenerator) -> (f32, f32, f32) {
177        let n = *self.num_nodes.get(&g.degree()).unwrap();
178
179        let (radius, patterns) = PATTERNS[n - 1];
180        let offset = patterns[g.idx()];
181
182        (
183            radius,
184            (g.x() * Self::GRID_WIDTH + Self::MARGIN) as f32 + offset.0,
185            ((self.max - g.degree()).y() * Self::GRID_WIDTH + Self::MARGIN) as f32 + offset.1,
186        )
187    }
188
189    pub fn new(out: T) -> Self {
190        Self {
191            out,
192            max: Bidegree::zero(),
193            num_nodes: HashMap::new(),
194        }
195    }
196}

198impl<T: io::Write> Backend for SvgBackend<T> {
199    type Error = io::Error;
200
201    const EXT: &'static str = "svg";
202
203    fn header(&mut self, max: Bidegree) -> Result<(), Self::Error> {
204        self.max = max;
205
206        let width = self.max.x() * Self::GRID_WIDTH + 2 * Self::MARGIN;
207        let height = self.max.y() * Self::GRID_WIDTH + 2 * Self::MARGIN;
208
209        writeln!(
210            self.out,
211            r#"<svg version = "1.1" width="{width}" height="{height}" xmlns="http://www.w3.org/2000/svg">"#,
212        )?;
213        writeln!(self.out, "<style>{}</style>", Self::STYLES)
214    }
215
216    fn line(&mut self, start: Bidegree, end: Bidegree, style: &str) -> Result<(), Self::Error> {
217        let height = self.max.y() * Self::GRID_WIDTH + 2 * Self::MARGIN;
218
219        writeln!(
220            self.out,
221            r#"<line class="{style}" x1="{start_x}" x2="{end_x}" y1="{start_y}" y2="{end_y}" />"#,
222            start_x = Self::MARGIN + start.x() * Self::GRID_WIDTH,
223            end_x = Self::MARGIN + end.x() * Self::GRID_WIDTH,
224            start_y = height - Self::MARGIN - start.y() * Self::GRID_WIDTH,
225            end_y = height - Self::MARGIN - end.y() * Self::GRID_WIDTH,
226        )
227    }
228
229    fn text(
230        &mut self,
231        b: Bidegree,
232        content: impl Display,
233        orientation: Orientation,
234    ) -> Result<(), Self::Error> {
235        let (offset, class) = match orientation {
236            Orientation::Left => ((-5, 0), "y-label"),
237            Orientation::Right => unimplemented!(),
238            Orientation::Below => ((0, 3), "x-label"),
239            Orientation::Above => unimplemented!(),
240        };
241
242        writeln!(
243            self.out,
244            r#"<text class="{class}" x="{x}" y="{y}">{content}</text>"#,
245            x = Self::MARGIN + b.x() * Self::GRID_WIDTH + offset.0,
246            y = Self::MARGIN + (self.max - b).y() * Self::GRID_WIDTH + offset.1,
247        )
248    }
249
250    fn node(&mut self, b: Bidegree, n: usize) -> Result<(), Self::Error> {
251        if n == 0 || b.x() > self.max.x() || b.y() > self.max.y() {
252            return Ok(());
253        }
254        self.num_nodes.insert(b, n);
255
256        for k in 0..n {
257            let (r, x, y) = self.get_coords(BidegreeGenerator::new(b, k));
258            writeln!(self.out, r#"<circle cx="{x}" cy="{y}" r="{r}"/>"#,)?;
259        }
260        Ok(())
261    }
262
263    fn structline(
264        &mut self,
265        source: BidegreeGenerator,
266        target: BidegreeGenerator,
267        style: Option<&str>,
268    ) -> Result<(), Self::Error> {
269        if source.x() > self.max.x()
270            || source.y() > self.max.y()
271            || target.x() > self.max.x()
272            || target.y() > self.max.y()
273        {
274            return Ok(());
275        }
276
277        let (_, source_x, source_y) = self.get_coords(source);
278        let (_, target_x, target_y) = self.get_coords(target);
279
280        writeln!(
281            self.out,
282            r#"<line class="{style}" x1="{source_x}" x2="{target_x}" y1="{source_y}" y2="{target_y}" />"#,
283            style = match &style {
284                Some(x) => format!("structline {x}"),
285                None => String::from("structline"),
286            },
287        )?;
288
289        Ok(())
290    }
291}

294    fn drop(&mut self) {
295        writeln!(self.out, "</svg>").unwrap();
296    }

299pub struct TikzBackend<T: io::Write> {
300    out: T,
301    max: Bidegree,
302    num_nodes: HashMap<Bidegree, usize>,
303}
304
305impl<T: io::Write> TikzBackend<T> {
306    const HEADER: &'static str = r"\begin{tikzpicture}[
307  major-grid/.style={ opacity = 0.2 },
308  grid/.style={ opacity = 0.1 },
309  d2/.style={ blue },
310]";
311
312    pub fn new(out: T) -> Self {
313        Self {
314            out,
315            max: Bidegree::zero(),
316            num_nodes: HashMap::new(),
317        }
318    }
319
320    /// Returns r, x, y
321    fn get_coords(&self, g: BidegreeGenerator) -> (f32, f32, f32) {
322        let n = *self.num_nodes.get(&g.degree()).unwrap();
323
324        let (radius, patterns) = PATTERNS[n - 1];
325        let offset = patterns[g.idx()];
326
327        (
328            radius / 20.0,
329            g.x() as f32 + offset.0 / 20.0,
330            // We subtract because in Tikz the origin is the bottom-left while in SVG it is the
331            // top-left
332            g.y() as f32 - offset.1 / 20.0,
333        )
334    }
335}

337impl<T: io::Write> Backend for TikzBackend<T> {
338    type Error = std::io::Error;
339
340    const EXT: &'static str = "tex";
341
342    fn header(&mut self, max: Bidegree) -> Result<(), Self::Error> {
343        self.max = max;
344        writeln!(self.out, "{}", Self::HEADER)
345    }
346
347    fn line(&mut self, start: Bidegree, end: Bidegree, style: &str) -> Result<(), Self::Error> {
348        writeln!(
349            self.out,
350            r#"\draw [{style}] ({start_x}, {start_y}) -- ({end_x}, {end_y});"#,
351            start_x = start.x(),
352            start_y = start.y(),
353            end_x = end.x(),
354            end_y = end.y(),
355        )
356    }
357
358    fn text(
359        &mut self,
360        b: Bidegree,
361        content: impl Display,
362        orientation: Orientation,
363    ) -> Result<(), Self::Error> {
364        let offset = match orientation {
365            Orientation::Left => "left",
366            Orientation::Right => "right",
367            Orientation::Below => "below",
368            Orientation::Above => "above",
369        };
370
371        writeln!(
372            self.out,
373            r#"\node [{offset}] at ({x}, {y}) {{{content}}};"#,
374            x = b.x(),
375            y = b.y()
376        )
377    }
378
379    fn node(&mut self, b: Bidegree, n: usize) -> Result<(), Self::Error> {
380        if n == 0 || b.x() > self.max.x() || b.y() > self.max.y() {
381            return Ok(());
382        }
383        self.num_nodes.insert(b, n);
384
385        for k in 0..n {
386            let (r, x, y) = self.get_coords(BidegreeGenerator::new(b, k));
387            writeln!(self.out, r#"\draw [fill] ({x}, {y}) circle ({r});"#,)?;
388        }
389        Ok(())
390    }
391
392    fn structline(
393        &mut self,
394        source: BidegreeGenerator,
395        target: BidegreeGenerator,
396        style: Option<&str>,
397    ) -> Result<(), Self::Error> {
398        if source.x() > self.max.x()
399            || source.y() > self.max.y()
400            || target.x() > self.max.x()
401            || target.y() > self.max.y()
402        {
403            return Ok(());
404        }
405
406        let (_, source_x, source_y) = self.get_coords(source);
407        let (_, target_x, target_y) = self.get_coords(target);
408
409        writeln!(
410            self.out,
411            r#"\draw [{style}] ({source_x}, {source_y}) -- ({target_x}, {target_y});"#,
412            style = style.unwrap_or(""),
413        )?;
414
415        Ok(())
416    }
417}

422    }
