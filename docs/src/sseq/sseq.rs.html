16/// The direction of the differentials
17pub trait SseqProfile {
18    const MIN_R: i32;
19    fn profile(r: i32, b: Bidegree) -> Bidegree;
20    fn profile_inverse(r: i32, b: Bidegree) -> Bidegree;
21    fn differential_length(offset: Bidegree) -> i32;
22}
23
24pub struct Adams;
25
26impl SseqProfile for Adams {
27    const MIN_R: i32 = 2;
28
29    fn profile(r: i32, b: Bidegree) -> Bidegree {
30        b + Bidegree::x_y(-1, r)
31    }
32
33    fn profile_inverse(r: i32, b: Bidegree) -> Bidegree {
34        b + Bidegree::x_y(1, -r)
35    }
36
37    fn differential_length(offset: Bidegree) -> i32 {
38        offset.y()
39    }
40}

49pub struct Sseq<P: SseqProfile = Adams> {
50    p: ValidPrime,
51
52    /// The first page of the spectral sequence
53    classes: Arc<DenseBigradedModule>,
54
55    /// x -> y -> r -> differential
56    ///
57    /// If the bidegree is valid (see [`Sseq::invalid`]), then the differential is reduced.
58    differentials: BiVec<BiVec<BiVec<Differential>>>,
59
60    /// x -> y -> permanent_classes
61    permanent_classes: BiVec<BiVec<Subspace>>,
62
63    /// x -> y -> r -> E_r^{x, y} as a subquotient of the original bidegree.
64    ///
65    /// The "quotient" part of the subquotient is always accurate, but the "sub" part may not be.
66    /// The `invalid` field tracks which bidegrees are valid.
67    ///
68    /// This is always the full ambient space when `r = P::MIN_R`, but we set `min_degree` to
69    /// `P::MIN_R` to make code a bit more streamlined.
70    ///
71    /// # Invariants:
72    ///  - if `differential[x][y][r]` is defined, then `page_data[x][y][r + 1]` and
73    ///    `page_data[tx][ty][r + 1]` are always defined,
74    page_data: BiVec<BiVec<BiVec<Subquotient>>>,
75
76    /// x -> y -> validity. A bidegree is invalid if the page_data is no longer accurate.
77    invalid: BiVec<BiVec<bool>>,
78
79    // `P` is itself a marker, so it's safe to claim that we own one. As opposed to
80    // `PhantomData<*const P>`, this lets us implement `Send` and `Sync`.
81    profile: PhantomData<P>,
82}

117    pub fn set_dimension(&mut self, b: Bidegree, dim: usize) {
118        // This already ensures it is valid to set b
119        self.classes.set_dimension(b, dim);
120        if self.differentials.len() == b.x() {
121            let min_y = self.classes.min().y();
122            self.differentials.push(BiVec::new(min_y));
123            self.permanent_classes.push(BiVec::new(min_y));
124            self.page_data.push(BiVec::new(min_y));
125            self.invalid.push(BiVec::new(min_y));
126        }
127
128        self.differentials[b.x()].push(BiVec::new(P::MIN_R));
129        self.page_data[b.x()].push(BiVec::new(P::MIN_R));
130        self.page_data[b.x()][b.y()].push(Subquotient::new_full(self.p, dim));
131        self.permanent_classes[b.x()].push(Subspace::new(self.p, dim));
132        self.invalid[b.x()].push(false);
133    }

157    /// # Returns
158    ///
159    /// Whether a new permanent class was added
160    pub fn add_permanent_class(&mut self, elem: &BidegreeElement) -> bool {
161        let old_dim = self.permanent_classes[elem.x()][elem.y()].dimension();
162        let new_dim = self.permanent_classes[elem.x()][elem.y()].add_vector(elem.vec());
163        if old_dim != new_dim {
164            // This was a new permanent class
165            for d in self.differentials[elem.x()][elem.y()].iter_mut() {
166                d.add(elem.vec(), None);
167            }
168            self.invalid[elem.x()][elem.y()] = true;
169        }
170        old_dim != new_dim
171    }

173    /// Ensure `self.differentials[b.x()][b.y()][r]` is defined. Must call `extend_page_data` on the source
174    /// and target after this.
175    fn extend_differential(&mut self, r: i32, b: Bidegree) {
176        let source_dim = self.classes.dimension(b);
177        while self.differentials[b.x()][b.y()].len() <= r {
178            let r = self.differentials[b.x()][b.y()].len();
179            let target = P::profile(r, b);
180            let mut differential =
181                Differential::new(self.p, source_dim, self.classes.dimension(target));
182
183            for class in self.permanent_classes[b.x()][b.y()].basis() {
184                differential.add(class.as_slice(), None);
185            }
186            self.differentials[b.x()][b.y()].push(differential);
187        }
188    }

190    /// Ensure `self.page_data[b.x()][b.y()][r]` is defined
191    fn extend_page_data(&mut self, r: i32, b: Bidegree) {
192        let page_data = &mut self.page_data[b.x()][b.y()];
193        while page_data.len() <= r {
194            page_data.push(page_data.last().unwrap().clone())
195        }
196    }

198    /// Add a $d_r$ differential from bidegree $(x, y)$, with the given `source` and `target`
199    /// classes.
200    ///
201    /// # Return
202    ///
203    /// Whether the differential is new
204    pub fn add_differential(&mut self, r: i32, source: &BidegreeElement, target: FpSlice) -> bool {
205        let target_b = P::profile(r, source.degree());
206
207        self.extend_differential(r, source.degree());
208        self.extend_page_data(r + 1, source.degree());
209        self.extend_page_data(r + 1, target_b);
210
211        for r in P::MIN_R..r {
212            self.differentials[source.x()][source.y()][r].add(source.vec(), None);
213            self.extend_page_data(r + 1, P::profile(r, source.degree()));
214        }
215        let is_new = self.differentials[source.x()][source.y()][r].add(source.vec(), Some(target));
216        if is_new {
217            self.invalid[source.x()][source.y()] = true;
218            if !target.is_zero() {
219                self.invalid[target_b.x()][target_b.y()] = true;
220                self.add_permanent_class(&BidegreeElement::new(target_b, target.to_owned()));
221                for r in r + 1..self.page_data[target_b.x()][target_b.y()].len() {
222                    self.page_data[target_b.x()][target_b.y()][r].quotient(target);
223
224                    let p = P::profile_inverse(r, target_b);
225                    if self.defined(p) {
226                        self.invalid[p.x()][p.y()] = true;
227                    }
228                }
229            }
230        }
231        is_new
232    }

234    pub fn invalid(&self, b: Bidegree) -> bool {
235        self.invalid[b.x()][b.y()]
236    }

238    pub fn update(&mut self) {
239        for x in self.invalid.range() {
240            for y in self.invalid[x].range() {
241                if self.invalid[x][y] {
242                    self.update_bidegree(Bidegree::x_y(x, y));
243                }
244            }
245        }
246    }

248    /// This returns the vec of differentials to draw on each page.
249    pub fn update_bidegree(&mut self, b: Bidegree) -> BiVec<Vec<Vec<u32>>> {
250        self.invalid[b.x()][b.y()] = false;
251        for (r, d) in self.differentials[b.x()][b.y()].iter_mut_enum() {
252            let target_b = P::profile(r, b);
253            d.reduce_target(self.page_data[target_b.x()][target_b.y()][r].zeros());
254        }
255
256        // For each page, the array of differentials to draw
257        let mut differentials: BiVec<Vec<Vec<u32>>> =
258            BiVec::with_capacity(P::MIN_R, self.differentials[b.x()][b.y()].len());
259
260        for r in self.page_data[b.x()][b.y()].range().skip(1) {
261            let target_b = P::profile(r - 1, b);
262
263            self.page_data[b.x()][b.y()][r].clear_gens();
264
265            if r > self.differentials[b.x()][b.y()].len()
266                || self.page_data[target_b.x()][target_b.y()][r - 1].is_empty()
267            {
268                let (prev, cur) = self.page_data[b.x()][b.y()].split_borrow_mut(r - 1, r);
269                for g in prev.gens() {
270                    cur.add_gen(g);
271                }
272                if r - 1 < self.differentials[b.x()][b.y()].len() {
273                    differentials.push(vec![
274                        Vec::new();
275                        self.page_data[b.x()][b.y()][r].dimension()
276                    ]);
277                }
278            } else {
279                let d = &self.differentials[b.x()][b.y()][r - 1];
280
281                let source_dim = self.dimension(b);
282                let target_dim = self.dimension(target_b);
283
284                let mut drawn_differentials: Vec<Vec<u32>> =
285                    Vec::with_capacity(self.page_data[b.x()][b.y()][r - 1].dimension());
286
287                let mut dvec = FpVector::new(self.p, target_dim);
288                let mut matrix = Matrix::new(
289                    self.p,
290                    self.page_data[b.x()][b.y()][r - 1].dimension(),
291                    source_dim + target_dim,
292                );
293
294                for (row, g) in std::iter::zip(
295                    matrix.iter_mut(),
296                    self.page_data[b.x()][b.y()][r - 1].gens(),
297                ) {
298                    row.slice_mut(target_dim, target_dim + source_dim).assign(g);
299
300                    d.evaluate(g, dvec.as_slice_mut());
301                    row.slice_mut(0, target_dim).assign(dvec.as_slice());
302
303                    drawn_differentials.push(
304                        self.page_data[target_b.x()][target_b.y()][r - 1]
305                            .reduce(dvec.as_slice_mut()),
306                    );
307                    dvec.set_to_zero();
308                }
309                differentials.push(drawn_differentials);
310
311                matrix.row_reduce();
312
313                let first_kernel_row = matrix.find_first_row_in_block(target_dim);
314
315                for row in &matrix[first_kernel_row..] {
316                    if row.is_zero() {
317                        break;
318                    }
319                    self.page_data[b.x()][b.y()][r]
320                        .add_gen(row.slice(target_dim, target_dim + source_dim));
321                }
322            }
323        }
324        differentials
325    }

327    /// Whether the calcuations at bidegree (x, y) are complete. This means all classes on the
328    /// final page are known to be permanent.
329    pub fn complete(&self, b: Bidegree) -> bool {
330        self.page_data[b.x()][b.y()]
331            .last()
332            .unwrap()
333            .gens()
334            .all(|v| self.permanent_classes[b.x()][b.y()].contains(v))
335    }

338    pub fn inconsistent(&self, b: Bidegree) -> bool {
339        self.differentials(b).iter().any(Differential::inconsistent)
340            || self.differentials_hitting(b).any(|(_, d)| d.inconsistent())

347    pub fn differentials_hitting(
348        &self,
349        b: Bidegree,
350    ) -> impl Iterator<Item = (i32, &'_ Differential)> + '_ {
351        let max_r = self.page_data[b.x()][b.y()].len() - 1;
352        (P::MIN_R..max_r).filter_map(move |r| {
353            let source_b = P::profile_inverse(r, b);
354            Some((
355                r,
356                self.differentials
357                    .get(source_b.x())?
358                    .get(source_b.y())?
359                    .get(r)?,
360            ))
361        })
362    }

372    /// Compute the product between `product` and the class `class` at `(x, y)`. Returns `None` if
373    /// the product is not yet computed.
374    pub fn multiply(&self, elem: &BidegreeElement, prod: &Product) -> Option<BidegreeElement> {
375        let mut result = FpVector::new(self.p, self.classes.get_dimension(elem.degree() + prod.b)?);
376        if let Some(matrix) = &prod.matrices.get(elem.x())?.get(elem.y())? {
377            matrix.apply(result.as_slice_mut(), 1, elem.vec());
378        }
379        Some(BidegreeElement::new(elem.degree() + prod.b, result))
380    }

382    /// Apply the Leibniz rule to obtain new differentials. The differential we start with is a d_r
383    /// differential from (x, y) with source `s` and target `t`. If the source is permanent, then r
384    /// should be set to [`i32::MAX`].
385    ///
386    /// # Arguments
387    ///  - `source_product` the product to multiply the class with
388    ///  - `target_product` the differential on `source_product`. If `source_product` is permanent,
389    ///    then this is None.
390    ///
391    /// # Return
392    ///
393    /// We return a tuple `(r, x, y, class)` recording the (source of) the new differential.
394    /// If the function returns None, this means no differential was added. This can either be
395    /// because the differential was trivial, or the data needed to compute the product is not yet
396    /// available.
397    pub fn leibniz(
398        &mut self,
399        r: i32,
400        elem: &BidegreeElement,
401        source_product: &Product,
402        target_product: Option<&Product>,
403    ) -> Option<(i32, BidegreeElement)> {
404        let source = self.multiply(elem, source_product)?;
405
406        // The class and the product are both permanent.
407        if r == i32::MAX && target_product.is_none() {
408            if self.add_permanent_class(&source) {
409                return Some((i32::MAX, source));
410            } else {
411                return None;
412            }
413        }
414
415        let neg_1 = self.p - 1;
416
417        let target_r = target_product
418            .map(|prod| P::differential_length(elem.degree() + prod.b - source.degree()))
419            .unwrap_or(i32::MAX);
420
421        let result_r = std::cmp::min(r, target_r);
422
423        let result_b = P::profile(result_r, source.degree());
424        let mut result = FpVector::new(self.p, self.classes.get_dimension(result_b)?);
425
426        if r == result_r {
427            let diffs = &self.differentials[elem.x()][elem.y()][r];
428            let d_b = P::profile(r, elem.degree());
429            let mut dx = FpVector::new(self.p, self.classes.dimension(d_b));
430            diffs.evaluate(elem.vec(), dx.as_slice_mut());
431            let d = BidegreeElement::new(d_b, dx);
432            let target = self.multiply(&d, source_product)?;
433
434            if source_product.left && source_product.b.x() % 2 != 0 {
435                result.add(&target.into_vec(), neg_1);
436            } else {
437                result.add(&target.into_vec(), 1);
438            }
439        }
440
441        if target_r == result_r {
442            let target = self.multiply(elem, target_product.unwrap())?;
443            // why is this x - 1 but not x? This is what the original code does and came from trial
444            // and error(?)
445            if !source_product.left && (elem.x() - 1) % 2 != 0 {
446                result.add(&target.into_vec(), neg_1);
447            } else {
448                result.add(&target.into_vec(), 1);
449            }
450        }
451
452        if self.add_differential(result_r, &source, result.as_slice()) {
453            Some((result_r, source))
454        } else {
455            None
456        }
457    }

459    /// This shifts the sseq horizontally so that the minimum x is 0.
460    pub fn write_to_graph<'a, T: chart::Backend>(
461        &self,
462        mut g: T,
463        r: i32,
464        differentials: bool,
465        products: impl Iterator<Item = &'a (String, Product)> + Clone,
466        header: impl FnOnce(&mut T) -> Result<(), T::Error>,
467    ) -> Result<(), T::Error> {
468        let min = self.min();
469        assert_eq!(min.y(), 0);
470
471        let max = self.max();
472
473        g.init((max - min).x(), max.y())?;
474        header(&mut g)?;
475
476        for x in min.x()..=max.x() {
477            for y in self.range(x) {
478                let b = Bidegree::x_y(x, y);
479                let shifted_b = b - min;
480
481                let data = self.page_data(b).get_max(r);
482                if data.is_empty() {
483                    continue;
484                }
485
486                g.node(shifted_b.x(), shifted_b.y(), data.dimension())?;
487
488                // Now add the products hitting this bidegree
489                for (name, prod) in products.clone() {
490                    let source_b = b - prod.b;
491                    let shifted_source = source_b - min;
492
493                    if !self.defined(source_b) {
494                        continue;
495                    }
496
497                    let source_data = self.page_data(source_b).get_max(r);
498                    if source_data.is_empty() {
499                        continue;
500                    }
501
502                    // For unstable charts this is None in low degrees.
503                    if let Some(matrix) = &prod.matrices[source_b.x()][source_b.y()] {
504                        let matrix = Subquotient::reduce_matrix(matrix, source_data, data);
505                        g.structline_matrix(
506                            (shifted_source.x(), shifted_source.y()),
507                            (shifted_b.x(), shifted_b.y()),
508                            matrix,
509                            Some(name),
510                        )?;
511                    }
512                }
513
514                // Finally add the differentials
515                if differentials {
516                    let target_b = P::profile(r, b);
517                    let shifted_target = target_b - min;
518
519                    if target_b.x() < 0 {
520                        continue;
521                    }
522                    let d = self.differentials(b);
523                    if d.len() <= r {
524                        continue;
525                    }
526                    let d = &d[r];
527                    let target_data = self.page_data(target_b).get_max(r);
528
529                    let pairs = d
530                        .get_source_target_pairs()
531                        .into_iter()
532                        .map(|(mut s, mut t)| {
533                            (
534                                data.reduce(s.as_slice_mut()),
535                                target_data.reduce(t.as_slice_mut()),
536                            )
537                        });
538
539                    for (source, target) in pairs {
540                        for (i, v) in source.into_iter().enumerate() {
541                            if v == 0 {
542                                continue;
543                            }
544                            for (j, &v) in target.iter().enumerate() {
545                                if v == 0 {
546                                    continue;
547                                }
548                                g.structline(
549                                    (shifted_b.x(), shifted_b.y(), i),
550                                    (shifted_target.x(), shifted_target.y(), j),
551                                    Some(&format!("d{r}")),
552                                )?;
553                            }
554                        }
555                    }
556                }
557            }
558        }
559
560        Ok(())
561    }
