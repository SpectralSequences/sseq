8pub struct SvgBackend<T: io::Write> {
9    out: T,
10    max: Bidegree,
11    num_nodes: HashMap<Bidegree, usize>,
12}

14impl<T: io::Write> SvgBackend<T> {
15    const GRID_WIDTH: i32 = 20;
16    const MARGIN: i32 = 30;
17    const STYLES: &'static str = r#"
18    circle {
19        fill: black;
20    }
21    .structline {
22        stroke: black;
23        fill: none;
24    }
25    .d2 {
26        stroke: blue;
27    }
28    .major-grid {
29        stroke: black;
30        opacity: 20%;
31        shape-rendering: crispEdges;
32        fill: none;
33    }
34    .grid {
35        stroke: black;
36        opacity: 10%;
37        shape-rendering: crispEdges;
38        fill: none;
39    }
40    .x-label {
41     text-anchor: middle;
42     dominant-baseline: text-before-edge;
43    }
44    .y-label {
45     text-anchor: end;
46     dominant-baseline: middle;
47    }
48    "#;
49
50    /// Print the legend for node patterns
51    pub fn legend(mut out: T) -> std::io::Result<()> {
52        writeln!(
53            out,
54            r#"<svg version = "1.1" width="{}" height="100" xmlns="http://www.w3.org/2000/svg">"#,
55            PATTERNS.len() * 100,
56        )?;
57
58        for (n, (_, row)) in PATTERNS.iter().enumerate() {
59            writeln!(
60                out,
61                r#"<rect fill="none" stroke="black" width="80" height="80" y="10" x="{x}"/>"#,
62                x = 10 + n * 100
63            )?;
64            for (k, offset) in row[0..=n].iter().enumerate() {
65                writeln!(
66                    out,
67                    r#"<text x="{x}" y="{y}" text-anchor="middle" dominant-baseline="middle">{k}</text>"#,
68                    x = 50.0 + n as f32 * 100.0 + offset.0 * 6.0,
69                    y = 50.0 + offset.1 * 6.0,
70                )?;
71            }
72        }
73        writeln!(out, r#"</svg>"#,)?;
74
75        Ok(())
76    }
77
78    /// Returns (radius, x, y) for the given generator.
79    ///
80    /// # Panics
81    ///
82    /// Panics if `node()` was not previously called for `g.degree()`.
83    fn get_coords(&self, g: BidegreeGenerator) -> (f32, f32, f32) {
84        let n = *self.num_nodes.get(&g.degree()).unwrap();
85
86        let (radius, patterns) = PATTERNS[n - 1];
87        let offset = patterns[g.idx()];
88
89        (
90            radius,
91            (g.x() * Self::GRID_WIDTH + Self::MARGIN) as f32 + offset.0,
92            ((self.max - g.degree()).y() * Self::GRID_WIDTH + Self::MARGIN) as f32 + offset.1,
93        )
94    }
95
96    pub fn new(out: T) -> Self {
97        Self {
98            out,
99            max: Bidegree::zero(),
100            num_nodes: HashMap::new(),
101        }
102    }
103}

105impl<T: io::Write> Backend for SvgBackend<T> {
106    type Error = io::Error;
107
108    const EXT: &'static str = "svg";
109
110    fn header(&mut self, max: Bidegree) -> Result<(), Self::Error> {
111        self.max = max;
112
113        let width = self.max.x() * Self::GRID_WIDTH + 2 * Self::MARGIN;
114        let height = self.max.y() * Self::GRID_WIDTH + 2 * Self::MARGIN;
115
116        writeln!(
117            self.out,
118            r#"<svg version = "1.1" width="{width}" height="{height}" xmlns="http://www.w3.org/2000/svg">"#,
119        )?;
120        writeln!(self.out, "<style>{}</style>", Self::STYLES)
121    }
122
123    fn line(&mut self, start: Bidegree, end: Bidegree, style: &str) -> Result<(), Self::Error> {
124        let height = self.max.y() * Self::GRID_WIDTH + 2 * Self::MARGIN;
125
126        writeln!(
127            self.out,
128            r#"<line class="{style}" x1="{start_x}" x2="{end_x}" y1="{start_y}" y2="{end_y}" />"#,
129            start_x = Self::MARGIN + start.x() * Self::GRID_WIDTH,
130            end_x = Self::MARGIN + end.x() * Self::GRID_WIDTH,
131            start_y = height - Self::MARGIN - start.y() * Self::GRID_WIDTH,
132            end_y = height - Self::MARGIN - end.y() * Self::GRID_WIDTH,
133        )
134    }
135
136    // SvgBackend only supports printing text to the left of or below a certain bidegree. This is
137    // used for printing axis labels.
138    fn text(
139        &mut self,
140        b: Bidegree,
141        content: impl Display,
142        orientation: Orientation,
143    ) -> Result<(), Self::Error> {
144        let (offset, class) = match orientation {
145            Orientation::Left => ((-5, 0), "y-label"),
146            Orientation::Right => unimplemented!(),
147            Orientation::Below => ((0, 3), "x-label"),
148            Orientation::Above => unimplemented!(),
149        };
150
151        writeln!(
152            self.out,
153            r#"<text class="{class}" x="{x}" y="{y}">{content}</text>"#,
154            x = Self::MARGIN + b.x() * Self::GRID_WIDTH + offset.0,
155            y = Self::MARGIN + (self.max - b).y() * Self::GRID_WIDTH + offset.1,
156        )
157    }
158
159    fn node(&mut self, b: Bidegree, n: usize) -> Result<(), Self::Error> {
160        if n == 0 || b.x() > self.max.x() || b.y() > self.max.y() {
161            return Ok(());
162        }
163        self.num_nodes.insert(b, n);
164
165        for k in 0..n {
166            let (r, x, y) = self.get_coords(BidegreeGenerator::new(b, k));
167            writeln!(self.out, r#"<circle cx="{x}" cy="{y}" r="{r}"/>"#,)?;
168        }
169        Ok(())
170    }
171
172    fn structline(
173        &mut self,
174        source: BidegreeGenerator,
175        target: BidegreeGenerator,
176        style: Option<&str>,
177    ) -> Result<(), Self::Error> {
178        if source.x() > self.max.x()
179            || source.y() > self.max.y()
180            || target.x() > self.max.x()
181            || target.y() > self.max.y()
182        {
183            return Ok(());
184        }
185
186        let (_, source_x, source_y) = self.get_coords(source);
187        let (_, target_x, target_y) = self.get_coords(target);
188
189        writeln!(
190            self.out,
191            r#"<line class="{style}" x1="{source_x}" x2="{target_x}" y1="{source_y}" y2="{target_y}" />"#,
192            style = match &style {
193                Some(x) => format!("structline {x}"),
194                None => String::from("structline"),
195            },
196        )?;
197
198        Ok(())
199    }
200}
