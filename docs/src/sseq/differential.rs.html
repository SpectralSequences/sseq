1use fp::{
2    matrix::{Matrix, Subspace},
3    prime::ValidPrime,
4    vector::{FpSlice, FpSliceMut, FpVector},
5};
6
7pub struct Differential {
8    pub matrix: Matrix,
9    first_empty_row: usize,
10    source_dim: usize,
11    target_dim: usize,
12    error: bool,
13}
14
15impl Differential {
16    pub fn new(p: ValidPrime, source_dim: usize, target_dim: usize) -> Self {
17        // Leave more rows to make room for inconsistent differentials
18        Self {
19            matrix: Matrix::new(p, source_dim + target_dim + 1, source_dim + target_dim),
20            source_dim,
21            target_dim,
22            error: false,
23            first_empty_row: 0,
24        }
25    }
26
27    pub fn set_to_zero(&mut self) {
28        self.matrix.set_to_zero();
29        self.first_empty_row = 0;
30        self.error = false;
31    }
32
33    /// Add a differential
34    ///
35    /// # Return
36    /// Whether a new differential was indeed added. If false, the differential is already
37    /// existent.
38    ///
39    /// # Arguments
40    ///  - `source`: The source of the differential
41    ///  - `target`: The target of the differential. If `None`, the differential is zero. This
42    ///    should be reduced by the known images of earlier differentials.
43    pub fn add(&mut self, source: FpSlice, target: Option<FpSlice>) -> bool {
44        let source_dim = self.source_dim;
45        let target_dim = self.target_dim;
46        let next_row = &mut self.matrix[self.first_empty_row];
47        next_row.slice_mut(0, source_dim).add(source, 1);
48
49        // The last row is always empty
50        if let Some(t) = target {
51            next_row
52                .slice_mut(source_dim, source_dim + target_dim)
53                .assign(t);
54        };
55        if next_row.is_zero() {
56            return false;
57        }
58        self.matrix.row_reduce();
59
60        if self.matrix[self.first_empty_row].is_zero() {
61            false
62        } else {
63            self.first_empty_row += 1;
64            true
65        }
66    }
67
68    /// An iterator of differentials in the form `(source, target)`
69    pub fn get_source_target_pairs(&self) -> Vec<(FpVector, FpVector)> {
70        let source_dim = self.source_dim;
71        let target_dim = self.target_dim;
72        self.matrix
73            .iter()
74            .filter(|d| !d.is_zero())
75            .map(move |d| {
76                (
77                    d.slice(0, source_dim).to_owned(),
78                    d.slice(source_dim, source_dim + target_dim).to_owned(),
79                )
80            })
81            .collect()
82    }
83
84    /// Given a subspace of the target space, project the target vectors to the complement.
85    pub fn reduce_target(&mut self, zeros: &Subspace) {
86        assert_eq!(zeros.ambient_dimension(), self.target_dim);
87
88        for row in self.matrix.iter_mut() {
89            zeros.reduce(row.slice_mut(self.source_dim, self.source_dim + self.target_dim));
90        }
91
92        self.matrix.row_reduce();
93
94        self.error = false;
95        for i in 0..self.target_dim {
96            if self.matrix.pivots()[self.source_dim + i] >= 0 {
97                self.error = true;
98                break;
99            }
100        }
101    }
102
103    /// This evaluates the differential on `source`, adding the result to `target`. This assumes
104    /// all unspecified differentials are zero. More precisely, it assumes every non-pivot column
105    /// of the differential matrix has zero differential. This may or may not be actually true
106    /// (e.g. if we only know d(a + b) = c, it might be that d(a) = c and d(b) = 0, or vice versa,
107    /// or neither. Here we assume d(a) = c and d(b) = 0.
108    pub fn evaluate(&self, source: FpSlice, mut target: FpSliceMut) {
109        for (i, c) in source.iter_nonzero() {
110            let row = self.matrix.pivots()[i];
111            if row < 0 {
112                continue;
113            }
114            let row = row as usize;
115
116            target.add(
117                self.matrix[row].slice(self.source_dim, self.source_dim + self.target_dim),
118                c,
119            );
120        }
121    }
122
123    pub fn prime(&self) -> ValidPrime {
124        self.matrix.prime()
125    }
126
127    /// Whether the current set of differentials is inconsistent. This should be called only after
128    /// `reduce_target` is called.
129    pub fn inconsistent(&self) -> bool {
130        self.error
131    }
132
133    /// Find the differential that hits `value`, and write the result to `result`.
134    ///
135    /// This computes the quasi-inverse from scratch and allocates two matrices, and should not be
136    /// used in a hot path.
137    pub fn quasi_inverse(&self, result: FpSliceMut, value: FpSlice) {
138        let mut matrix = Matrix::new(
139            self.matrix.prime(),
140            self.source_dim,
141            self.source_dim + self.target_dim,
142        );
143        // Transpose the source and target columns
144        for (target, source) in matrix.iter_mut().zip(self.matrix.iter()) {
145            target
146                .slice_mut(0, self.target_dim)
147                .assign(source.slice(self.source_dim, self.source_dim + self.target_dim));
148            target
149                .slice_mut(self.target_dim, self.target_dim + self.source_dim)
150                .assign(source.slice(0, self.source_dim));
151        }
152        matrix.row_reduce();
153        let qi = matrix.compute_quasi_inverse(self.target_dim, self.target_dim);
154        qi.apply(result, 1, value);
155    }
156}
