28pub trait Backend {
29    type Error;
30
31    /// If the backend writes to a file, this is the extension commonly taken by the file type
32    const EXT: &'static str = "";
33
34    fn header(&mut self, max_x: i32, max_y: i32) -> Result<(), Self::Error>;
35    fn line(
36        &mut self,
37        start_x: i32,
38        end_x: i32,
39        start_y: i32,
40        end_y: i32,
41        style: &str,
42    ) -> Result<(), Self::Error>;
43
44    fn text(
45        &mut self,
46        x: i32,
47        y: i32,
48        content: impl Display,
49        orientation: Orientation,
50    ) -> Result<(), Self::Error>;
51    fn node(&mut self, x: i32, y: i32, n: usize) -> Result<(), Self::Error>;
52
53    fn structline(
54        &mut self,
55        source: (i32, i32, usize),
56        target: (i32, i32, usize),
57        style: Option<&str>,
58    ) -> Result<(), Self::Error>;
59
60    fn init(&mut self, max_x: i32, max_y: i32) -> Result<(), Self::Error> {
61        self.header(max_x, max_y)?;
62
63        for x in 0..=max_x {
64            self.line(
65                x,
66                x,
67                0,
68                max_y,
69                if x % 4 == 0 { "major-grid" } else { "grid" },
70            )?;
71            if x % 4 == 0 {
72                self.text(x, 0, x, Orientation::Below)?;
73            }
74        }
75        for y in 0..=max_y {
76            self.line(
77                0,
78                max_x,
79                y,
80                y,
81                if y % 4 == 0 { "major-grid" } else { "grid" },
82            )?;
83            if y % 4 == 0 {
84                self.text(0, y, y, Orientation::Left)?;
85            }
86        }
87        Ok(())
88    }
89
90    fn structline_matrix(
91        &mut self,
92        source: (i32, i32),
93        target: (i32, i32),
94        matrix: Vec<Vec<u32>>,
95        class: Option<&str>,
96    ) -> Result<(), Self::Error> {
97        for (k, row) in matrix.into_iter().enumerate() {
98            for (l, v) in row.into_iter().enumerate() {
99                if v != 0 {
100                    self.structline((source.0, source.1, k), (target.0, target.1, l), class)?;
101                }
102            }
103        }
104        Ok(())
105    }
106}

109    out: T,

111    max_y: i32,
112    num_nodes: HashMap<(i32, i32), usize>,

115impl<T: io::Write> SvgBackend<T> {
116    const GRID_WIDTH: i32 = 20;
117    const MARGIN: i32 = 30;
118    const STYLES: &'static str = r#"
119    circle {
120        fill: black;
121    }
122    .structline {
123        stroke: black;
124        fill: none;
125    }
126    .d2 {
127        stroke: blue;
128    }
129    .major-grid {
130        stroke: black;
131        opacity: 20%;
132        shape-rendering: crispEdges;
133        fill: none;
134    }
135    .grid {
136        stroke: black;
137        opacity: 10%;
138        shape-rendering: crispEdges;
139        fill: none;
140    }
141    .x-label {
142     text-anchor: middle;
143     dominant-baseline: text-before-edge;
144    }
145    .y-label {
146     text-anchor: end;
147     dominant-baseline: middle;
148    }
149    "#;
150
151    /// Print the legend for node patterns
152    pub fn legend(mut out: T) -> std::io::Result<()> {
153        writeln!(
154            out,
155            r#"<svg version = "1.1" width="{}" height="100" xmlns="http://www.w3.org/2000/svg">"#,
156            PATTERNS.len() * 100,
157        )?;
158
159        for (n, (_, row)) in PATTERNS.iter().enumerate() {
160            writeln!(
161                out,
162                r#"<rect fill="none" stroke="black" width="80" height="80" y="10" x="{x}"/>"#,
163                x = 10 + n * 100
164            )?;
165            for (k, offset) in row[0..=n].iter().enumerate() {
166                writeln!(
167                    out,
168                    r#"<text x="{x}" y="{y}" text-anchor="middle" dominant-baseline="middle">{k}</text>"#,
169                    x = 50.0 + n as f32 * 100.0 + offset.0 * 6.0,
170                    y = 50.0 + offset.1 * 6.0,
171                )?;
172            }
173        }
174        writeln!(out, r#"</svg>"#,)?;
175
176        Ok(())
177    }
178
179    /// Returns r, x, y
180    fn get_coords(&self, x: i32, y: i32, i: usize) -> (f32, f32, f32) {
181        let n = *self.num_nodes.get(&(x, y)).unwrap();
182
183        let (radius, patterns) = PATTERNS[n - 1];
184        let offset = patterns[i];
185
186        (
187            radius,
188            (x * Self::GRID_WIDTH + Self::MARGIN) as f32 + offset.0,
189            ((self.max_y - y) * Self::GRID_WIDTH + Self::MARGIN) as f32 + offset.1,
190        )
191    }
192
193    pub fn new(out: T) -> Self {
194        Self {
195            out,
196            max_x: 0,
197            max_y: 0,
198            num_nodes: HashMap::new(),
199        }
200    }
201}

203impl<T: io::Write> Backend for SvgBackend<T> {
204    type Error = io::Error;
205
206    const EXT: &'static str = "svg";
207
208    fn header(&mut self, max_x: i32, max_y: i32) -> Result<(), Self::Error> {
209        self.max_x = max_x;
210        self.max_y = max_y;
211
212        let width = self.max_x * Self::GRID_WIDTH + 2 * Self::MARGIN;
213        let height = self.max_y * Self::GRID_WIDTH + 2 * Self::MARGIN;
214
215        writeln!(
216            self.out,
217            r#"<svg version = "1.1" width="{width}" height="{height}" xmlns="http://www.w3.org/2000/svg">"#,
218        )?;
219        writeln!(self.out, "<style>{}</style>", Self::STYLES)
220    }
221
222    fn line(
223        &mut self,
224        start_x: i32,
225        end_x: i32,
226        start_y: i32,
227        end_y: i32,
228        style: &str,
229    ) -> Result<(), Self::Error> {
230        let height = self.max_y * Self::GRID_WIDTH + 2 * Self::MARGIN;
231
232        writeln!(
233            self.out,
234            r#"<line class="{style}" x1="{start_x}" x2="{end_x}" y1="{start_y}" y2="{end_y}" />"#,
235            start_x = Self::MARGIN + start_x * Self::GRID_WIDTH,
236            end_x = Self::MARGIN + end_x * Self::GRID_WIDTH,
237            start_y = height - Self::MARGIN - start_y * Self::GRID_WIDTH,
238            end_y = height - Self::MARGIN - end_y * Self::GRID_WIDTH,
239        )
240    }
241
242    fn text(
243        &mut self,
244        x: i32,
245        y: i32,
246        content: impl Display,
247        orientation: Orientation,
248    ) -> Result<(), Self::Error> {
249        let (offset, class) = match orientation {
250            Orientation::Left => ((-5, 0), "y-label"),
251            Orientation::Right => unimplemented!(),
252            Orientation::Below => ((0, 3), "x-label"),
253            Orientation::Above => unimplemented!(),
254        };
255
256        writeln!(
257            self.out,
258            r#"<text class="{class}" x="{x}" y="{y}">{content}</text>"#,
259            x = Self::MARGIN + x * Self::GRID_WIDTH + offset.0,
260            y = Self::MARGIN + (self.max_y - y) * Self::GRID_WIDTH + offset.1,
261        )
262    }
263
264    fn node(&mut self, x: i32, y: i32, n: usize) -> Result<(), Self::Error> {
265        if n == 0 || x > self.max_x || y > self.max_y {
266            return Ok(());
267        }
268        self.num_nodes.insert((x, y), n);
269
270        for k in 0..n {
271            let (r, x, y) = self.get_coords(x, y, k);
272            writeln!(self.out, r#"<circle cx="{x}" cy="{y}" r="{r}"/>"#,)?;
273        }
274        Ok(())
275    }
276
277    fn structline(
278        &mut self,
279        source: (i32, i32, usize),
280        target: (i32, i32, usize),
281        style: Option<&str>,
282    ) -> Result<(), Self::Error> {
283        if source.0 > self.max_x
284            || source.1 > self.max_y
285            || target.0 > self.max_x
286            || target.1 > self.max_y
287        {
288            return Ok(());
289        }
290
291        let (_, source_x, source_y) = self.get_coords(source.0, source.1, source.2);
292        let (_, target_x, target_y) = self.get_coords(target.0, target.1, target.2);
293
294        writeln!(
295            self.out,
296            r#"<line class="{style}" x1="{source_x}" x2="{target_x}" y1="{source_y}" y2="{target_y}" />"#,
297            style = match &style {
298                Some(x) => format!("structline {x}"),
299                None => String::from("structline"),
300            },
301        )?;
302
303        Ok(())
304    }
305}

309        writeln!(self.out, "</svg>").unwrap();
310    }

313pub struct TikzBackend<T: io::Write> {
314    out: T,
315    max_x: i32,
316    max_y: i32,
317    num_nodes: HashMap<(i32, i32), usize>,
318}
319
320impl<T: io::Write> TikzBackend<T> {
321    const HEADER: &'static str = r"\begin{tikzpicture}[
322  major-grid/.style={ opacity = 0.2 },
323  grid/.style={ opacity = 0.1 },
324  d2/.style={ blue },
325]";
326
327    pub fn new(out: T) -> Self {
328        Self {
329            out,
330            max_x: 0,
331            max_y: 0,
332            num_nodes: HashMap::new(),
333        }
334    }
335
336    /// Returns r, x, y
337    fn get_coords(&self, x: i32, y: i32, i: usize) -> (f32, f32, f32) {
338        let n = *self.num_nodes.get(&(x, y)).unwrap();
339
340        let (radius, patterns) = PATTERNS[n - 1];
341        let offset = patterns[i];
342
343        (
344            radius / 20.0,
345            x as f32 + offset.0 / 20.0,
346            // We subtract because in Tikz the origin is the bottom-left while in SVG it is the
347            // top-left
348            y as f32 - offset.1 / 20.0,
349        )
350    }
351}

353impl<T: io::Write> Backend for TikzBackend<T> {
354    type Error = std::io::Error;
355
356    const EXT: &'static str = "tex";
357
358    fn header(&mut self, max_x: i32, max_y: i32) -> Result<(), Self::Error> {
359        self.max_x = max_x;
360        self.max_y = max_y;
361        writeln!(self.out, "{}", Self::HEADER)
362    }
363
364    fn line(
365        &mut self,
366        start_x: i32,
367        end_x: i32,
368        start_y: i32,
369        end_y: i32,
370        style: &str,
371    ) -> Result<(), Self::Error> {
372        writeln!(
373            self.out,
374            r#"\draw [{style}] ({start_x}, {start_y}) -- ({end_x}, {end_y});"#,
375        )
376    }
377
378    fn text(
379        &mut self,
380        x: i32,
381        y: i32,
382        content: impl Display,
383        orientation: Orientation,
384    ) -> Result<(), Self::Error> {
385        let offset = match orientation {
386            Orientation::Left => "left",
387            Orientation::Right => "right",
388            Orientation::Below => "below",
389            Orientation::Above => "above",
390        };
391
392        writeln!(self.out, r#"\node [{offset}] at ({x}, {y}) {{{content}}};"#,)
393    }
394
395    fn node(&mut self, x: i32, y: i32, n: usize) -> Result<(), Self::Error> {
396        if n == 0 || x > self.max_x || y > self.max_y {
397            return Ok(());
398        }
399        self.num_nodes.insert((x, y), n);
400
401        for k in 0..n {
402            let (r, x, y) = self.get_coords(x, y, k);
403            writeln!(self.out, r#"\draw [fill] ({x}, {y}) circle ({r});"#,)?;
404        }
405        Ok(())
406    }
407
408    fn structline(
409        &mut self,
410        source: (i32, i32, usize),
411        target: (i32, i32, usize),
412        style: Option<&str>,
413    ) -> Result<(), Self::Error> {
414        if source.0 > self.max_x
415            || source.1 > self.max_y
416            || target.0 > self.max_x
417            || target.1 > self.max_y
418        {
419            return Ok(());
420        }
421
422        let (_, source_x, source_y) = self.get_coords(source.0, source.1, source.2);
423        let (_, target_x, target_y) = self.get_coords(target.0, target.1, target.2);
424
425        writeln!(
426            self.out,
427            r#"\draw [{style}] ({source_x}, {source_y}) -- ({target_x}, {target_y});"#,
428            style = style.unwrap_or(""),
429        )?;
430
431        Ok(())
432    }
433}
