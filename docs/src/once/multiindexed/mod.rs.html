5/// A multi-dimensional array that allows efficient storage and retrieval of values using
6/// K-dimensional integer coordinates.
7///
8/// `MultiIndexed<K, V>` provides a thread-safe and wait-free way to store values indexed by
9/// multi-dimensional coordinates. It is implemented using a K-dimensional trie structure that
10/// efficiently handles sparse data, where each level of the trie corresponds to one dimension of
11/// the coordinate space.
12///
13/// The `MultiIndexed` is created with a fixed number of dimensions `K` and can store values of any
14/// type `V`. Each dimension can have both positive and negative indices.
15///
16/// # Thread Safety
17///
18/// `MultiIndexed` is designed to be thread-safe and wait-free, allowing concurrent insertions and
19/// retrievals from multiple threads. This makes it suitable for parallel algorithms that need to
20/// build up a shared data structure.
21///
22/// # Memory Efficiency
23///
24/// The underlying trie structure allocates memory only for coordinates that are actually used,
25/// making it memory-efficient for sparse data. The implementation uses a series of
26/// [`TwoEndedGrove`](crate::TwoEndedGrove) instances to store the trie nodes, which themselves use
27/// a block-based allocation strategy.
28///
29/// # Performance Characteristics
30///
31/// - **Insertion**: O(K) time complexity,
32/// - **Retrieval**: O(K) time complexity,
33/// - **Memory Usage**: amortized O(N) space complexity, where N is the number of inserted elements
34///
35/// # Warning
36///
37/// This data structure is designed for write-once semantics, meaning that once a value is inserted
38/// at a specific coordinate, it cannot be changed directly. If you need to update values, you can
39/// wrap them in a struct that allows for interior mutability (e.g., `RefCell`, `Mutex`, etc.).
40///
41/// Note that, for performance reasons, we do not allow `K = 0`.
42///
43/// # Examples
44///
45/// Correct usage:
46///
47/// ```
48/// use once::MultiIndexed;
49///
50/// // Create a 3-dimensional array
51/// let array = MultiIndexed::<3, i32>::new();
52///
53/// // Insert values at specific coordinates
54/// array.insert([1, 2, 3], 42);
55/// array.insert([5, 0, 2], 100);
56/// array.insert([-1, -2, 3], 200); // Negative coordinates are supported
57///
58/// // Retrieve values
59/// assert_eq!(array.get([1, 2, 3]), Some(&42));
60/// assert_eq!(array.get([5, 0, 2]), Some(&100));
61/// assert_eq!(array.get([-1, -2, 3]), Some(&200));
62/// assert_eq!(array.get([0, 0, 0]), None); // No value at these coordinates
63/// ```
64///
65/// Incorrect usage:
66///
67/// ```should_panic
68/// use once::MultiIndexed;
69///
70/// let array = MultiIndexed::<2, i32>::new();
71///
72/// array.insert([1, 2], 42);
73/// array.insert([1, 2], 43); // Panics because the value at [1, 2] is already set
74/// ```
75pub struct MultiIndexed<const K: usize, V>(KdTrie<V>);
76

102    /// Retrieves a reference to the value at the specified coordinates, if it exists.
103    ///
104    /// # Parameters
105    ///
106    /// * `coords`: An array of K integer coordinates
107    ///
108    /// # Returns
109    ///
110    /// * `Some(&V)` if a value exists at the specified coordinates
111    /// * `None` if no value exists at the specified coordinates
112    ///
113    /// # Examples
114    ///
115    /// ```
116    /// use once::MultiIndexed;
117    ///
118    /// // Basic retrieval in a 3D array
119    /// let array = MultiIndexed::<3, i32>::new();
120    /// array.insert([1, 2, 3], 42);
121    ///
122    /// assert_eq!(array.get([1, 2, 3]), Some(&42));
123    /// assert_eq!(array.get([0, 0, 0]), None);
124    ///
125    /// // Retrieval with negative coordinates
126    /// array.insert([-5, -10, 15], 100);
127    /// assert_eq!(array.get([-5, -10, 15]), Some(&100));
128    ///
129    /// // Retrieval in a 2D array
130    /// let array2d = MultiIndexed::<2, String>::new();
131    /// array2d.insert([0, 0], "Origin".to_string());
132    /// array2d.insert([10, -5], "Far point".to_string());
133    ///
134    /// assert_eq!(array2d.get([0, 0]), Some(&"Origin".to_string()));
135    /// assert_eq!(array2d.get([10, -5]), Some(&"Far point".to_string()));
136    /// assert_eq!(array2d.get([1, 1]), None);
137    ///
138    /// // Retrieval in a 1D array
139    /// let array1d = MultiIndexed::<1, f64>::new();
140    /// array1d.insert([0], 3.14);
141    /// array1d.insert([-10], -2.71);
142    ///
143    /// assert_eq!(array1d.get([0]), Some(&3.14));
144    /// assert_eq!(array1d.get([-10]), Some(&-2.71));
145    /// assert_eq!(array1d.get([5]), None);
146    /// ```
147    pub fn get(&self, coords: [i32; K]) -> Option<&V> {
148        self.0.get(&coords)
149    }
150
151    /// Inserts a value at the specified coordinates.
152    ///
153    /// This operation is thread-safe and can be called from multiple threads. However, this method
154    /// panics if a value already exists at the specified coordinates. Therefore, it should only be
155    /// called at most once for any given set of coordinates.
156    ///
157    /// # Parameters
158    ///
159    /// * `coords`: An array of K integer coordinates
160    /// * `value`: The value to insert at the specified coordinates
161    ///
162    /// # Examples
163    ///
164    /// ```
165    /// use once::MultiIndexed;
166    ///
167    /// // Basic insertion in a 3D array
168    /// let array = MultiIndexed::<3, i32>::new();
169    /// array.insert([1, 2, 3], 42);
170    ///
171    /// // Insertion with negative coordinates
172    /// array.insert([-5, 0, 10], 100);
173    /// array.insert([0, -3, -7], 200);
174    ///
175    /// // Insertion in a 2D array
176    /// let array2d = MultiIndexed::<2, String>::new();
177    /// array2d.insert([0, 0], "Origin".to_string());
178    /// array2d.insert([10, -5], "Far point".to_string());
179    ///
180    /// // Insertion in a 1D array
181    /// let array1d = MultiIndexed::<1, f64>::new();
182    /// array1d.insert([0], 3.14);
183    /// array1d.insert([-10], -2.71);
184    /// ```
185    ///
186    /// # Panics
187    ///
188    /// This method will panic if a value already exists at the specified coordinates:
189    ///
190    /// ```should_panic
191    /// use once::MultiIndexed;
192    ///
193    /// let array = MultiIndexed::<2, i32>::new();
194    /// array.insert([1, 2], 42);
195    /// array.insert([1, 2], 43); // Panics
196    /// ```
197    pub fn insert(&self, coords: [i32; K], value: V) {
198        self.0.insert(&coords, value);
199    }
200
201    pub fn try_insert(&self, coords: [i32; K], value: V) -> Result<(), V> {
202        self.0.try_insert(&coords, value)
203    }
204}

217    /// Generate all tuples of length K where the sum of coordinates equals n
218    /// ```
219    /// let result = once::get_nth_diagonal::<3>(4);
220    ///
221    /// assert_eq!(result.len(), 15);
222    /// assert!(result.contains(&[0, 0, 4]));
223    /// assert!(result.contains(&[0, 1, 3]));
224    /// assert!(result.contains(&[0, 2, 2]));
225    /// assert!(result.contains(&[0, 3, 1]));
226    /// assert!(result.contains(&[0, 4, 0]));
227    /// assert!(result.contains(&[1, 0, 3]));
228    /// assert!(result.contains(&[1, 1, 2]));
229    /// assert!(result.contains(&[1, 2, 1]));
230    /// assert!(result.contains(&[1, 3, 0]));
231    /// assert!(result.contains(&[2, 0, 2]));
232    /// assert!(result.contains(&[2, 1, 1]));
233    /// assert!(result.contains(&[2, 2, 0]));
234    /// assert!(result.contains(&[3, 0, 1]));
235    /// assert!(result.contains(&[3, 1, 0]));
236    /// assert!(result.contains(&[4, 0, 0]));
237    /// ```
238    pub fn get_nth_diagonal<const K: usize>(n: usize) -> Vec<[i32; K]> {
239        let mut result = Vec::new();
240        let mut tuple = vec![0; K];
241
242        // Generate all tuples where the sum of coordinates equals n
243        generate_tuples::<K>(&mut tuple, 0, n, &mut result);
244
245        result
246    }
247
248    /// Helper function to recursively generate the tuples
249    fn generate_tuples<const K: usize>(
250        tuple: &mut Vec<i32>,
251        index: usize,
252        sum: usize,
253        result: &mut Vec<[i32; K]>,
254    ) {
255        if index == K - 1 {
256            // The last element gets whatever is left to reach the sum
257            tuple[index] = sum as i32;
258            result.push(tuple.clone().try_into().unwrap()); // Convert to [i32; K]
259            return;
260        }
261
262        for i in 0..=sum {
263            tuple[index] = i as i32;
264            generate_tuples::<K>(tuple, index + 1, sum - i, result);
265        }
266    }
267
268    fn get_n_coords<const K: usize>(n: usize) -> Vec<[i32; K]> {
269        (0..).flat_map(get_nth_diagonal).take(n).collect()
270    }

296    #[test]
297    fn test_requires_drop() {
298        use std::{
299            sync::{
300                atomic::{AtomicUsize, Ordering},
301                Arc,
302            },
303            thread,
304        };
305
306        static ACTIVE_ALLOCS: AtomicUsize = AtomicUsize::new(0);
307
308        struct DropCounter;
309
310        impl DropCounter {
311            fn new() -> Self {
312                ACTIVE_ALLOCS.fetch_add(1, Ordering::Relaxed);
313                Self
314            }
315        }
316
317        impl Drop for DropCounter {
318            fn drop(&mut self) {
319                ACTIVE_ALLOCS.fetch_sub(1, Ordering::Relaxed);
320            }
321        }
322
323        let v = Arc::new(MultiIndexed::<3, DropCounter>::new());
324        assert_eq!(ACTIVE_ALLOCS.load(Ordering::Relaxed), 0);
325
326        let num_threads = crate::test_utils::num_threads() as i32;
327        let inserts_per_thread = crate::test_utils::values_per_thread() as i32;
328
329        thread::scope(|s| {
330            for thread_id in 0..num_threads {
331                let v = Arc::clone(&v);
332                s.spawn(move || {
333                    for i in (-inserts_per_thread / 2)..(inserts_per_thread / 2) {
334                        v.insert([thread_id, i, 4], DropCounter::new());
335                    }
336                });
337            }
338        });
339
340        assert_eq!(
341            ACTIVE_ALLOCS.load(Ordering::Relaxed),
342            (num_threads * inserts_per_thread) as usize
343        );
344
345        drop(v);
346
347        assert_eq!(ACTIVE_ALLOCS.load(Ordering::Relaxed), 0);
348    }

358        /// Generate a strategy for a single i32 coordinate.
359        fn coord_strategy(max: u32) -> impl Strategy<Value = i32> {
360            -(max as i32)..=max as i32
361        }
362
363        // Generate a strategy for arrays of i32 with a specific dimension
364        fn coords_strategy<const K: usize>(max: u32) -> impl Strategy<Value = [i32; K]> {
365            proptest::collection::vec(coord_strategy(max), K)
366                .prop_map(|v| std::array::from_fn(|i| v[i]))
367        }
368
369        #[derive(Debug, Clone, Copy)]
370        enum Operation<const K: usize> {
371            Insert([i32; K], i32),
372            Get([i32; K]),
373        }
374
375        // Generate a strategy for a single operation (insert or get)
376        fn operation_strategy<const K: usize>(max: u32) -> impl Strategy<Value = Operation<K>> {
377            proptest::bool::ANY.prop_flat_map(move |is_insert| {
378                coords_strategy::<K>(max).prop_flat_map(move |coords| {
379                    if is_insert {
380                        proptest::num::i32::ANY
381                            .prop_map(move |value| Operation::Insert(coords, value))
382                            .boxed()
383                    } else {
384                        proptest::strategy::Just(Operation::Get(coords)).boxed()
385                    }
386                })
387            })
388        }
389
390        // Generate a strategy for a list of operations (insert or get)
391        fn operations_strategy<const K: usize>(
392            max_ops: usize,
393        ) -> impl Strategy<Value = Vec<Operation<K>>> {
394            // This is chosen so that the max number of operations is twice the number of elements in
395            // the hypercube (-max..=max)^K
396            let max = ((max_ops as f32 / 2.0).powf(1.0 / K as f32) / 2.0).ceil() as u32;
397            proptest::collection::vec(operation_strategy::<K>(max), 1..max_ops)
398        }
399
400        fn proptest_multiindexed_kd<const K: usize>(ops: Vec<Operation<K>>) {
401            let arr = MultiIndexed::<K, i32>::new();
402            let mut reference = HashMap::new();
403
404            for op in ops {
405                match op {
406                    Operation::Insert(coords, value) => {
407                        // Only insert if the key doesn't exist yet (to avoid panics)
408                        if let std::collections::hash_map::Entry::Vacant(e) =
409                            reference.entry(coords)
410                        {
411                            arr.insert(coords, value);
412                            e.insert(value);
413                        } else {
414                            // If the key already exists, test that try_insert returns an error
415                            assert!(arr.try_insert(coords, value).is_err());
416                        }
417                    }
418                    Operation::Get(coords) => {
419                        // Check that get returns the same as our reference HashMap
420                        let actual = arr.get(coords);
421                        let expected = reference.get(&coords);
422                        assert_eq!(actual, expected);
423                    }
424                }
425            }
426        }
427
428        proptest! {
429            #[test]
430            fn proptest_multiindexed_2d(ops in operations_strategy::<2>(10000)) {
431                proptest_multiindexed_kd::<2>(ops);
432            }
433
434            #[test]
435            fn proptest_multiindexed_3d(ops in operations_strategy::<3>(10000)) {
436                proptest_multiindexed_kd::<3>(ops);

446        #[test]
447        fn loom_concurrent_insert_get() {
448            loom::model(|| {
449                let arr = Arc::new(MultiIndexed::<2, i32>::new());
450
451                // Thread 1: Insert values
452                let arr1 = Arc::clone(&arr);
453                let t1 = thread::spawn(move || {
454                    arr1.insert([0, 0], 10);
455                    arr1.insert([0, 1], 20);
456                    arr1.insert([1, 0], 30);
457                });
458
459                // Thread 2: Insert different values
460                let arr2 = Arc::clone(&arr);
461                let t2 = thread::spawn(move || {
462                    arr2.insert([1, 1], 40);
463                    arr2.insert([2, 0], 50);
464                    arr2.insert([0, 2], 60);
465                });
466
467                // Thread 3: Read values
468                let arr3 = Arc::clone(&arr);
469                let t3 = thread::spawn(move || {
470                    // These may or may not be set yet
471                    let _ = arr3.get([0, 0]);
472                    let _ = arr3.get([1, 1]);
473                    let _ = arr3.get([2, 2]); // This one is never set
474                });
475
476                t1.join().unwrap();
477                t2.join().unwrap();
478                t3.join().unwrap();
479
480                // Verify final state
481                assert_eq!(arr.get([0, 0]), Some(&10));
482                assert_eq!(arr.get([0, 1]), Some(&20));
483                assert_eq!(arr.get([1, 0]), Some(&30));
484                assert_eq!(arr.get([1, 1]), Some(&40));
485                assert_eq!(arr.get([2, 0]), Some(&50));
486                assert_eq!(arr.get([0, 2]), Some(&60));
487                assert_eq!(arr.get([2, 2]), None);
488            });
489        }
