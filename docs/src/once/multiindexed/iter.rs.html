8/// A stack frame in the depth-first traversal of a [`KdTrie`].
9///
10/// Each frame records the current node, its depth in the trie (i.e. which coordinate dimension it
11/// indexes), and the remaining range of indices to visit at this node.
12struct IterFrame<R> {
13    depth: usize,
14    current_node: R,
15    range: Range<i32>,
16}
17
18/// A depth-first iterator over a [`KdTrie`], generic over:
19///
20/// - `R: NodeRef` — the node handle type, either shared (`&Node<V>`) or exclusive
21///   (`NodePtrMut<'_, V>`), determining whether values are yielded as `&V` or `&mut V`.
22/// - `C: Coordinates` — the coordinate accumulator, either `[i32; K]` (fixed-size, for
23///   [`MultiIndexed`]) or `Vec<i32>` (dynamic, for [`KdTrie`]).
24///
25/// The iterator walks the trie in lexicographic order of coordinates, yielding `(C, R::Value)` for
26/// each stored entry.
27struct KdIterator<R, C> {
28    dimensions: usize,
29    stack: Vec<IterFrame<R>>,
30    coordinates: C,
31}
32
33impl<R: NodeRef, C> KdIterator<R, C> {
34    fn new(dimensions: usize, root: R, coordinates: C) -> Self {
35        let root_range = unsafe { root.range(dimensions == 1) };
36        Self {
37            dimensions,
38            stack: vec![IterFrame {
39                depth: 0,
40                current_node: root,
41                range: root_range,
42            }],
43            coordinates,
44        }
45    }
46}
47
48impl<R: NodeRef, C: Coordinates> Iterator for KdIterator<R, C> {
49    type Item = (C, R::Value);
50
51    fn next(&mut self) -> Option<Self::Item> {
52        while let Some(IterFrame {
53            depth,
54            current_node,
55            mut range,
56        }) = self.stack.pop()
57        {
58            self.coordinates.truncate_to(depth);
59
60            // Find the next index in the current range that has a value
61            while let Some(idx) = range.next() {
62                if depth == self.dimensions - 1 {
63                    // This is a leaf node, check if there's a value at this index
64                    if let Some(value) = unsafe { current_node.value(idx) } {
65                        // Push back the remaining range for this node
66                        if !range.is_empty() {
67                            self.stack.push(IterFrame {
68                                depth,
69                                current_node,
70                                range,
71                            });
72                        }
73
74                        self.coordinates.set_coord(depth, idx);
75                        return Some((self.coordinates.get(), value));
76                    }
77                } else if let Some(child_node) = unsafe { current_node.child(idx) } {
78                    // This is an inner node, check if there's a child at this index
79
80                    // Push back the remaining range for this node
81                    if !range.is_empty() {
82                        self.stack.push(IterFrame {
83                            depth,
84                            current_node,
85                            range,
86                        });
87                    }
88
89                    // Add the current index to coordinates and push the child
90                    self.coordinates.set_coord(depth, idx);
91                    let child_range = unsafe { child_node.range(depth + 1 == self.dimensions - 1) };
92                    self.stack.push(IterFrame {
93                        depth: depth + 1,
94                        current_node: child_node,
95                        range: child_range,
96                    });
97
98                    // Go to the next iteration of the outer loop, which will process the child
99                    break;
100                }
101            }
102        }
103
104        None
105    }
106}

108// --- NodeRef ---
109
110/// Abstraction over shared (`&Node<V>`) and exclusive (`*mut Node<V>`) node access.
111///
112/// This trait allows [`KdIterator`] to be generic over the borrowing mode, so a single iterator
113/// implementation drives both `iter` (shared) and `iter_mut` (exclusive).
114///
115/// # Safety
116///
117/// Implementations must ensure that:
118/// - `range`, `child`, and `value` uphold the safety preconditions of the underlying [`Node`]
119///   methods (i.e. leaf methods are only called on leaf nodes, and inner methods on inner nodes).
120/// - For mutable implementations, the returned value references do not alias.
121unsafe trait NodeRef: Copy {
122    type Value;
123
124    /// Returns the range of indices for this node.
125    ///
126    /// # Safety
127    ///
128    /// `is_leaf` must correctly indicate whether this is a leaf node.
129    unsafe fn range(self, is_leaf: bool) -> Range<i32>;
130
131    /// Returns a handle to the child node at `idx`, or `None` if no child exists.
132    ///
133    /// # Safety
134    ///
135    /// Must only be called on inner nodes.
136    unsafe fn child(self, idx: i32) -> Option<Self>;
137
138    /// Returns a reference to the value at `idx`, or `None` if the slot is empty.
139    ///
140    /// # Safety
141    ///
142    /// Must only be called on leaf nodes.
143    unsafe fn value(self, idx: i32) -> Option<Self::Value>;
144}
145
146/// Shared node reference. Yields `&V` values.
147unsafe impl<'a, V> NodeRef for &'a Node<V> {
148    type Value = &'a V;
149
150    unsafe fn range(self, is_leaf: bool) -> Range<i32> {
151        if is_leaf {
152            unsafe { self.leaf() }.range()
153        } else {
154            unsafe { self.inner() }.range()
155        }
156    }
157
158    unsafe fn child(self, idx: i32) -> Option<Self> {
159        unsafe { self.inner().get(idx) }
160    }
161
162    unsafe fn value(self, idx: i32) -> Option<Self::Value> {
163        unsafe { self.leaf().get(idx) }
164    }
165}
166
167/// A `Copy` wrapper around `*mut Node<V>` that serves as the exclusive counterpart to
168/// `&Node<V>` in the [`NodeRef`] trait.
169///
170/// The phantom lifetime `'a` ties the yielded `&'a mut V` references back to the original
171/// `&'a mut MultiIndexed` (or `&'a mut KdTrie`), ensuring soundness.
172///
173/// This is safe to use because:
174/// - It is only constructed from `&mut MultiIndexed` / `&mut KdTrie`, guaranteeing exclusive access
175///   to the entire tree.
176/// - The tree structure ensures that nodes at different positions are disjoint in memory.
177/// - The iterator yields each value at most once.
178struct NodePtrMut<'a, V>(*mut Node<V>, PhantomData<&'a mut V>);
179
180impl<V> Copy for NodePtrMut<'_, V> {}
181
182impl<V> Clone for NodePtrMut<'_, V> {
183    fn clone(&self) -> Self {
184        *self
185    }
186}
187
188/// Exclusive node reference. Yields `&mut V` values.
189unsafe impl<'a, V> NodeRef for NodePtrMut<'a, V> {
190    type Value = &'a mut V;
191
192    unsafe fn range(self, is_leaf: bool) -> Range<i32> {
193        if is_leaf {
194            unsafe { (*self.0).leaf() }.range()
195        } else {
196            unsafe { (*self.0).inner() }.range()
197        }
198    }
199
200    unsafe fn child(self, idx: i32) -> Option<Self> {
201        let child = unsafe { (*self.0).get_child_mut(idx) }?;
202        Some(Self(child as *mut Node<V>, PhantomData))
203    }
204
205    unsafe fn value(self, idx: i32) -> Option<Self::Value> {
206        unsafe { (*self.0).get_value_mut(idx) }
207    }
208}

210// --- Coordinates ---
211
212/// Trait for managing coordinates during iteration.
213///
214/// The iterator accumulates coordinates dimension-by-dimension as it descends. When it backtracks,
215/// it calls [`truncate_to`](Coordinates::truncate_to) to discard coordinates from deeper
216/// dimensions. When it yields an entry, it calls [`get`](Coordinates::get) to snapshot the current
217/// coordinates.
218trait Coordinates {
219    /// Sets the coordinate at the given `depth` (dimension index) to `value`.
220    fn set_coord(&mut self, depth: usize, value: i32);
221
222    /// Discards any coordinate data beyond `depth`, preparing for backtracking.
223    fn truncate_to(&mut self, depth: usize);
224
225    /// Returns a snapshot of the current coordinates.
226    fn get(&self) -> Self;
227}
228
229/// Fixed-size coordinate accumulator for [`MultiIndexed`].
230///
231/// `truncate_to` is a no-op since all dimensions are always present in the array; stale values at
232/// deeper indices are simply overwritten by `set_coord` before they are ever read.
233impl<const K: usize> Coordinates for [i32; K] {
234    fn set_coord(&mut self, depth: usize, value: i32) {
235        self[depth] = value;
236    }
237
238    fn truncate_to(&mut self, _depth: usize) {}
239
240    fn get(&self) -> Self {
241        *self
242    }
243}
244
245/// Dynamic coordinate accumulator for [`KdTrie`].
246///
247/// `set_coord` pushes a new coordinate (asserting that `depth == len`, i.e. coordinates are always
248/// built in order), and `truncate_to` pops coordinates back to the given depth.
249impl Coordinates for Vec<i32> {
250    fn set_coord(&mut self, depth: usize, value: i32) {
251        assert_eq!(self.len(), depth);
252        self.push(value);
253    }
254
255    fn truncate_to(&mut self, depth: usize) {
256        self.truncate(depth);
257    }
258
259    fn get(&self) -> Self {
260        self.clone()
261    }
262}

266impl<V> KdTrie<V> {
267    pub fn iter(&self) -> impl Iterator<Item = (Vec<i32>, &V)> + '_ {
268        let dimensions = self.dimensions();
269        KdIterator::new(dimensions, self.root(), Vec::with_capacity(dimensions))
270    }
271
272    pub fn iter_mut(&mut self) -> impl Iterator<Item = (Vec<i32>, &mut V)> + '_ {
273        let dimensions = self.dimensions();
274        let root = NodePtrMut(self.root_mut() as *mut Node<V>, PhantomData);
275        KdIterator::new(dimensions, root, Vec::with_capacity(dimensions))
276    }
277}

279impl<const K: usize, V> MultiIndexed<K, V> {
280    /// Returns an iterator over all coordinate-value pairs in the array.
281    ///
282    /// The iterator yields `([i32; K], &V)` tuples in lexicographic order of coordinates.
283    ///
284    /// # Examples
285    ///
286    /// ```
287    /// use once::MultiIndexed;
288    ///
289    /// let array = MultiIndexed::<2, i32>::new();
290    /// array.insert([3, 4], 10);
291    /// array.insert([1, 2], 20);
292    ///
293    /// let mut items: Vec<_> = array.iter().collect();
294    ///
295    /// assert_eq!(items, vec![([1, 2], &20), ([3, 4], &10)]);
296    /// ```
297    pub fn iter(&self) -> impl Iterator<Item = ([i32; K], &V)> {
298        KdIterator::new(K, self.0.root(), [0; K])
299    }
300
301    /// Returns a mutable iterator over all coordinate-value pairs in the array.
302    ///
303    /// The iterator yields `([i32; K], &mut V)` tuples in lexicographic order of coordinates.
304    ///
305    /// # Examples
306    ///
307    /// ```
308    /// use once::MultiIndexed;
309    ///
310    /// let mut array = MultiIndexed::<2, i32>::new();
311    /// array.insert([1, 2], 10);
312    /// array.insert([3, 4], 20);
313    ///
314    /// for (_, v) in array.iter_mut() {
315    ///     *v *= 2;
316    /// }
317    ///
318    /// assert_eq!(array.get([1, 2]), Some(&20));
319    /// assert_eq!(array.get([3, 4]), Some(&40));
320    /// ```
321    pub fn iter_mut(&mut self) -> impl Iterator<Item = ([i32; K], &mut V)> {
322        let root = NodePtrMut(self.0.root_mut() as *mut Node<V>, PhantomData);
323        KdIterator::new(K, root, [0; K])
324    }
325}

328mod tests {
329    use super::*;
330
331    #[test]
332    fn test_iter_mut_no_aliasing() {
333        let mut arr = MultiIndexed::<3, i32>::new();
334        arr.insert([0, 0, 0], 10);
335        arr.insert([0, 0, 1], 20);
336        arr.insert([0, 1, 0], 30);
337        arr.insert([1, 0, 0], 40);
338
339        let mut it = arr.iter_mut();
340        let (_, a) = it.next().unwrap();
341        let (_, b) = it.next().unwrap();
342        let (_, c) = it.next().unwrap();
343        let (_, d) = it.next().unwrap();
344
345        // Miri detects borrow-model violations if any of the references alias, even before the
346        // writes below.
347        *a += 1;
348        *b += 2;
349        *c += 3;
350        *d += 4;
351        drop(it);
352
353        assert_eq!(arr.get([0, 0, 0]), Some(&11));
354        assert_eq!(arr.get([0, 0, 1]), Some(&22));
355        assert_eq!(arr.get([0, 1, 0]), Some(&33));
356        assert_eq!(arr.get([1, 0, 0]), Some(&44));
357    }
