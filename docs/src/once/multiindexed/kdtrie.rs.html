3/// A K-dimensional trie data structure that efficiently stores values indexed by multi-dimensional
4/// coordinates.
5///
6/// The difference between `KdTrie` and `MultiIndexed` is that the latter has the dimension as part
7/// of the type, which allows for more type safety. The `KdTrie` itself may be useful in a situation
8/// where the dimension is not known at compile time.
9///
10/// The trie is structured as a tree where each level corresponds to one dimension of the coordinate
11/// space. For example, in a 3D trie, the first level corresponds to the x-coordinate, the second
12/// level to the y-coordinate, and the third level to the z-coordinate. This structure allows for
13/// efficient storage and retrieval of values in a sparse coordinate space.
14///
15/// # Thread Safety
16///
17/// `KdTrie` is designed to be thread-safe and wait-free, allowing concurrent insertions and
18/// retrievals from multiple threads. This is achieved through the use of atomic operations and the
19/// thread-safe properties of the underlying [`TwoEndedGrove`](crate::TwoEndedGrove) data structure.
20///
21/// # Memory Efficiency
22///
23/// The trie only allocates memory for coordinates that are actually used, making it
24/// memory-efficient for sparse data. Each node in the trie is either an inner node (which contains
25/// child nodes) or a leaf node (which contains values).
26///
27/// # Type Parameters
28///
29/// * `V`: The type of values stored in the trie
30pub struct KdTrie<V> {
31    root: Node<V>,
32    dimensions: usize,
33}
34
35impl<V> KdTrie<V> {
36    /// Creates a new `KdTrie` with the specified number of dimensions.
37    ///
38    /// # Parameters
39    ///
40    /// * `dimensions`: The number of dimensions for the trie (must be greater than 0)
41    ///
42    /// # Panics
43    ///
44    /// Panics if `dimensions` is 0.
45    pub fn new(dimensions: usize) -> Self {
46        assert!(dimensions > 0);
47
48        let root = if dimensions == 1 {
49            Node::new_leaf()
50        } else {
51            Node::new_inner()
52        };
53
54        Self { root, dimensions }
55    }
56
57    /// Retrieves a reference to the value at the specified coordinates, if it exists.
58    ///
59    /// # Parameters
60    ///
61    /// * `coords`: A slice of coordinates with length equal to `self.dimensions`
62    ///
63    /// # Returns
64    ///
65    /// * `Some(&V)` if a value exists at the specified coordinates
66    /// * `None` if no value exists at the specified coordinates
67    ///
68    /// # Panics
69    ///
70    /// Panics if the length of `coords` does not match the number of dimensions.
71    pub fn get(&self, coords: &[i32]) -> Option<&V> {
72        assert!(coords.len() == self.dimensions);
73
74        // When's the last time you saw a mutable shared reference?
75        let mut node = &self.root;
76
77        for &coord in coords.iter().take(self.dimensions - 1) {
78            node = unsafe { node.get_child(coord)? };
79        }
80
81        unsafe { node.get_value(coords[self.dimensions - 1]) }
82    }
83
84    /// Inserts a value at the specified coordinates.
85    ///
86    /// This method traverses the trie structure to find the appropriate location
87    /// for the value, creating nodes as needed along the way.
88    ///
89    /// # Parameters
90    ///
91    /// * `coords`: A slice of coordinates with length equal to `self.dimensions`
92    /// * `value`: The value to insert at the specified coordinates
93    ///
94    /// # Panics
95    ///
96    /// Panics if the length of `coords` does not match the number of dimensions.
97    pub fn insert(&self, coords: &[i32], value: V) {
98        assert!(coords.len() == self.dimensions);
99
100        let mut node = &self.root;
101
102        for &coord in coords.iter().take(self.dimensions.saturating_sub(2)) {
103            node = unsafe { node.ensure_child(coord, Node::new_inner()) };
104        }
105        if self.dimensions > 1 {
106            node = unsafe { node.ensure_child(coords[self.dimensions - 2], Node::new_leaf()) };
107        }
108
109        unsafe { node.set_value(coords[self.dimensions - 1], value) };
110    }
111
112    /// Attempts to insert a value at the specified coordinates.
113    ///
114    /// This method traverses the trie structure to find the appropriate location for the value,
115    /// creating nodes as needed along the way.
116    ///
117    /// This method will only insert the value if the coordinate is not already occupied. If the
118    /// coordinate is already occupied, the value is returned in the `Err` variant.
119    ///
120    /// # Parameters
121    ///
122    /// * `coords`: A slice of coordinates with length equal to `self.dimensions`
123    /// * `value`: The value to insert at the specified coordinates
124    ///
125    /// # Returns
126    ///
127    /// * `Ok(())` if the value was successfully inserted
128    /// * `Err(value)` if the coordinate was already occupied, returning the value that we tried to
129    ///   insert
130    ///
131    /// # Panics
132    ///
133    /// Panics if the length of `coords` does not match the number of dimensions.
134    ///
135    /// # Examples
136    ///
137    /// ```
138    /// use once::multiindexed::KdTrie;
139    ///
140    /// let trie = KdTrie::<i32>::new(2);
141    ///
142    /// assert_eq!(trie.try_insert(&[-3, 1], 10), Ok(()));
143    /// assert_eq!(trie.try_insert(&[-3, 1], 10), Err(10)); // Coordinate already occupied
144    /// ```
145    pub fn try_insert(&self, coords: &[i32], value: V) -> Result<(), V> {
146        assert!(coords.len() == self.dimensions);
147
148        let mut node = &self.root;
149
150        for &coord in coords.iter().take(self.dimensions.saturating_sub(2)) {
151            node = unsafe { node.ensure_child(coord, Node::new_inner()) };
152        }
153        if self.dimensions > 1 {
154            node = unsafe { node.ensure_child(coords[self.dimensions - 2], Node::new_leaf()) };
155        }
156
157        unsafe { node.try_set_value(coords[self.dimensions - 1], value) }
158    }
159}
160
161impl<V> Drop for KdTrie<V> {
162    fn drop(&mut self) {
163        self.root.drop_level(self.dimensions, 0);
164    }
