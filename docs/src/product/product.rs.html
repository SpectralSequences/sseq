1//! Computes products in Ext by left-multiplication by a fixed class.
2//!
3//! The program asks for a module `M` and a class `x ∈ Ext(M, k)`. It then prints the products of
4//! `x` with every basis class of `Ext(k, k)` that lands in a computed bidegree.
5//!
6//! This is the primary (i.e. non-secondary) analogue of [`secondary_product`](../secondary_product),
7//! written against the [`ExtAlgebra`] abstraction so the plumbing stays out of the way.
8
9use std::sync::Arc;
10
11use ext::{chain_complex::FreeChainComplex, ext_algebra::ExtAlgebra, utils::query_module};
12use sseq::coordinates::Bidegree;
13
14fn main() -> anyhow::Result<()> {
15    ext::utils::init_logging()?;
16
17    let resolution = Arc::new(query_module(None, true)?);
18    let e2 = ExtAlgebra::from_resolution(resolution)?;
19
20    let shift = Bidegree::n_s(
21        query::raw("n of Ext class", str::parse),
22        query::raw("s of Ext class", str::parse),
23    );
24
25    let dim = e2.dimension(shift);
26    if dim == 0 {
27        panic!("No classes in bidegree {shift}");
28    }
29    let v: Vec<u32> = query::vector("Input Ext class", dim);
30    let x = e2.element(shift, &v);
31
32    for b in e2.unit().iter_nonzero_stem() {
33        // `None` means `b + shift` is out of the computed range, so skip it.
34        let Some(rows) = e2.multiply_into(&x, b) else {
35            continue;
36        };
37        for (g, row) in e2.unit_basis(b).into_iter().zip(rows.iter()) {
38            let coords: Vec<u32> = row.iter().collect();
39            if coords.iter().any(|&c| c != 0) {
40                println!("x · x_{g} = {coords:?}");
41            }
42        }
43    }
44    Ok(())
45}
