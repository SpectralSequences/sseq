1//! Computes algebraic Mahowald invariants (aka algebraic root invariants).
2//!
3//! Sample output (with `Max k = 7`):
4//! ```
5//! M({basis element}) = {mahowald_invariant}[ mod {indeterminacy}]
6//! M(x_(0, 0, 0)) = x_(0, 0, 0)
7//! M(x_(1, 1, 0)) = x_(1, 2, 0)
8//! M(x_(2, 2, 0)) = x_(2, 4, 0)
9//! M(x_(1, 2, 0)) = x_(1, 4, 0)
10//! M(x_(3, 3, 0)) = x_(3, 6, 0)
11//! M(x_(2, 4, 0)) = x_(2, 8, 0)
12//! M(x_(1, 4, 0)) = x_(1, 8, 0)
13//! M(x_(2, 5, 0)) = x_(2, 10, 0)
14//! M(x_(3, 6, 0)) = x_(3, 12, 0)
15//! ```
16//!
17//! ---
18//!
19//! Here is a brief overview of what this example computes.
20//! For details and beyond, see for instance
21//! "[The root invariant in homotopy theory][mahowald--ravenel]" or
22//! "[The Bredon-LÃ¶ffler conjecture][bruner--greenlees]" (where the latter also contains machine
23//! computations similar to what this example does).
24//! In the following, we abbreviate `Ext^{s,t}_A(-, F_2)` as `Ext^{s,t}(-)`.
25//!
26//! Let `M_k` be the cohomology of `RP_-k_inf`.
27//! There is an isomorphism  `Ext^{s, t}(F_2) ~ lim_k Ext^{s, t-1}(M_k)`
28//! induced by the (-1)-cell `S^{-1} -> RP_-k_inf` at each level.
29//! Let `x` be a class in `Ext^{s, t}(F_2)`.
30//! Then there is a minimal `k` such that its image in `Ext^{s, t-1}(M_k)` is non-trivial.
31//! Using the long exact sequence induced by the (co)fiber sequence
32//! `S^{-k} -> RP_-k_inf -> RP_{-k+1}_inf` on the level of `Ext`, that image can be lifted to a
33//! class `M(x)` in `Ext^{s, t + k - 1}`, which is (a representative for) the *(algebraic) Mahowald
34//! invariant of `x`*.
35//!
36//! This script computes these lifts (and their indeterminacy) by resolving
37//! `F_2` resp. `M_k`s and constructing [`ResolutionHomomorphism`]s
38//! corresponding to the bottom and (-1)-cells.
39//! Given `Max k`, it will print Mahowald invariants of the `F_2`-basis elements of
40//! `Ext^{*,*}(F_2)` that are detected in `Ext^{*,*}(M_k)` for the first time for some
41//! `k <= Max k`.
42//!
43//! [mahowald--ravenel]: https://www.sciencedirect.com/science/article/pii/004093839390055Z
44//! [bruner--greenlees]: https://projecteuclid.org/journals/experimental-mathematics/volume-4/issue-4/The-Bredon-L%C3%B6ffler-conjecture/em/1047674389.full
45

63fn main() -> Result<()> {
64    ext::utils::init_logging()?;
65
66    let s_2_path: Option<PathBuf> = query::optional("Save directory for S_2", str::parse);
67    let p_k_prefix: Option<PathBuf> = query::optional(
68        "Directory containing save directories for RP_-k_inf's",
69        str::parse,
70    );
71    // Going up to k=25 is nice because then we see an invariant that is not a basis element
72    // and one that has non-trivial indeterminacy.
73    let k_max = query::with_default("Max k (positive)", "25", str::parse::<NonZeroI32>).get();
74
75    let s_2_resolution = resolve_s_2(s_2_path, k_max)?;
76
77    println!("M({{basis element}}) = {{mahowald_invariant}}[ mod {{indeterminacy}}]");
78    for k in 1..=k_max {
79        let p_k = PKData::try_new(k, &p_k_prefix, &s_2_resolution)?;
80        for mi in p_k.mahowald_invariants() {
81            println!("{mi}")
82        }
83    }
84
85    Ok(())
86}

88type Resolution =
89    MuResolution<false, FiniteChainComplex<Box<dyn Module<Algebra = SteenrodAlgebra>>>>;
90
91type Homomorphism = MuResolutionHomomorphism<false, Resolution, Resolution>;
92
93struct PKData {
94    k: i32,
95    resolution: Arc<Resolution>,
96    bottom_cell: Homomorphism,
97    minus_one_cell: Homomorphism,
98    s_2_resolution: Arc<Resolution>,
99}
100
101struct MahowaldInvariant {
102    gen: BidegreeGenerator,
103    output_t: i32,
104    invariant: FpVector,
105    indeterminacy_basis: Vec<FpVector>,
106}
107
108fn resolve_s_2(s_2_path: Option<PathBuf>, k_max: i32) -> Result<Arc<Resolution>> {
109    let s_2_resolution = Arc::new(utils::construct_standard("S_2", s_2_path)?);
110    // Here are some bounds on the bidegrees in which we have should have resolutions available.
111    //
112    // A class in stem n won't be detected before RP_-{n+1}_inf, so we can only detect Mahowald
113    // invariants of classes in stems <=k_max-1.
114    // If an element in stem k_max-1 is detected in RP_-{k_max}_inf, then its Mahowald invariant
115    // will be in stem 2*k_max-2, so we should resolve S_2 up to that stem.
116    //
117    // As for the filtration s, resolving up to (k/2)+1 will cover all classes in positive stems up
118    // to k-1 because of the Adams vanishing line.
119    // In the zero stem, the Mahowald invariant of x_(i, i, 0) (i.e. (h_0)^i) is the first element
120    // of filtration i that is in a positive stem.
121    // As that element appears by stem 2*i, resolving RP_-k_inf up to filtration (k/2)+1 is also
122    // sufficient to detect Mahowald invariants of elements in the zero stem.
123    s_2_resolution.compute_through_stem(Bidegree::n_s(2 * k_max - 2, k_max / 2 + 1));
124    Ok(s_2_resolution)
125}

127impl PKData {
128    fn try_new(
129        k: i32,
130        p_k_prefix: &Option<PathBuf>,
131        s_2_resolution: &Arc<Resolution>,
132    ) -> Result<Self> {
133        let p_k_config = json! ({
134            "p": 2,
135            "type": "real projective space",
136            "min": -k,
137        });
138        let mut p_k_path = p_k_prefix.clone();
139        if let Some(p) = p_k_path.as_mut() {
140            p.push(PathBuf::from(&format!("RP_{minus_k}_inf", minus_k = -k)));
141        };
142        let resolution = Arc::new(utils::construct_standard(
143            (p_k_config, AlgebraType::Milnor),
144            p_k_path,
145        )?);
146        // As mentioned before, RP_-k_inf won't detect Mahowald invariants of any classes in the
147        // k-stem and beyond or of any classes of filtration higher than k/2+1.
148        resolution.compute_through_stem(Bidegree::n_s(k - 2, k / 2 + 1));
149
150        let bottom_cell = ResolutionHomomorphism::from_class(
151            String::from("bottom_cell"),
152            resolution.clone(),
153            s_2_resolution.clone(),
154            Bidegree::s_t(0, -k),
155            &[1],
156        );
157        bottom_cell.extend_all();
158
159        let minus_one_cell = ResolutionHomomorphism::from_class(
160            String::from("minus_one_cell"),
161            resolution.clone(),
162            s_2_resolution.clone(),
163            Bidegree::s_t(0, -1),
164            &[1],
165        );
166        minus_one_cell.extend_all();
167
168        Ok(PKData {
169            k,
170            resolution,
171            bottom_cell,
172            minus_one_cell,
173            s_2_resolution: s_2_resolution.clone(),
174        })
175    }
176
177    fn mahowald_invariants(&self) -> impl Iterator<Item = MahowaldInvariant> + '_ {
178        self.s_2_resolution
179            .iter_stem()
180            .flat_map(|b| self.mahowald_invariants_for_bidegree(b))
181    }
182
183    fn mahowald_invariants_for_bidegree(
184        &self,
185        b: Bidegree,
186    ) -> Box<dyn Iterator<Item = MahowaldInvariant> + '_> {
187        let b_p_k = b - Bidegree::s_t(0, 1);
188        if self.resolution.has_computed_bidegree(b_p_k) {
189            let b_bottom = b_p_k + Bidegree::s_t(0, self.k);
190            let bottom_s_2_gens = self.s_2_resolution.number_of_gens_in_bidegree(b_bottom);
191            let minus_one_s_2_gens = self.s_2_resolution.number_of_gens_in_bidegree(b);
192            let p_k_gens = self.resolution.number_of_gens_in_bidegree(b_p_k);
193            if bottom_s_2_gens > 0 && minus_one_s_2_gens > 0 && p_k_gens > 0 {
194                let bottom_cell_map = self.bottom_cell.get_map(b_bottom.s());
195                let mut matrix = vec![vec![0; p_k_gens]; bottom_s_2_gens];
196                for p_k_gen in 0..p_k_gens {
197                    let output = bottom_cell_map.output(b_p_k.t(), p_k_gen);
198                    for (s_2_gen, row) in matrix.iter_mut().enumerate() {
199                        let index = bottom_cell_map.target().operation_generator_to_index(
200                            0,
201                            0,
202                            b_bottom.t(),
203                            s_2_gen,
204                        );
205                        row[p_k_gen] = output.entry(index);
206                    }
207                }
208                let (padded_columns, mut matrix) = Matrix::augmented_from_vec(TWO, &matrix);
209                let rank = matrix.row_reduce();
210
211                if rank > 0 {
212                    let kernel_subspace = matrix.compute_kernel(padded_columns);
213                    let indeterminacy_basis = kernel_subspace.basis().to_vec();
214                    let image_subspace = matrix.compute_image(p_k_gens, padded_columns);
215                    let quasi_inverse = matrix.compute_quasi_inverse(p_k_gens, padded_columns);
216
217                    let it = (0..minus_one_s_2_gens).filter_map(move |i| {
218                        let mut image = FpVector::new(TWO, p_k_gens);
219                        let gen = BidegreeGenerator::new(b, i);
220                        self.minus_one_cell.act(image.as_slice_mut(), 1, gen);
221                        if !image.is_zero() && image_subspace.contains(image.as_slice()) {
222                            let mut invariant = FpVector::new(TWO, bottom_s_2_gens);
223                            quasi_inverse.apply(invariant.as_slice_mut(), 1, image.as_slice());
224                            Some(MahowaldInvariant {
225                                gen,
226                                output_t: b_bottom.t(),
227                                invariant,
228                                indeterminacy_basis: indeterminacy_basis.clone(),
229                            })
230                        } else {
231                            None
232                        }
233                    });
234                    return Box::new(it);
235                }
236            }
237        }
238
239        Box::new(iter::empty())
240    }
241}

243impl fmt::Display for MahowaldInvariant {
244    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> fmt::Result {
245        let output_t = self.output_t;
246        let f2_vec_to_sum = |v: &FpVector| {
247            let elt = BidegreeElement::new(Bidegree::s_t(self.gen.s(), output_t), v.clone());
248            elt.to_basis_string()
249        };
250        let indeterminacy_info = if self.indeterminacy_basis.is_empty() {
251            String::new()
252        } else {
253            format!(
254                " mod <{inner}>",
255                inner = self
256                    .indeterminacy_basis
257                    .iter()
258                    .map(f2_vec_to_sum)
259                    .collect::<Vec<_>>()
260                    .join(", ")
261            )
262        };
263        let invariant = f2_vec_to_sum(&self.invariant);
264        write!(
265            f,
266            "M(x_{gen}) = {invariant}{indeterminacy_info}",
267            gen = self.gen
268        )
269    }
270}

272#[cfg(test)]
273mod tests {
274    use rstest::rstest;
275
276    use super::*;
277
278    #[rstest]
279    #[case(1, 0, 0, 0, 0, vec![1], 0)]
280    #[case(5, 1, 4, 0, 8, vec![1], 0)]
281    #[case(18, 3, 17, 0, 34, vec![0, 1], 0)]
282    #[case(25, 6, 20, 0, 44, vec![1, 0], 1)]
283    fn test_mahowald_invariants(
284        #[case] k: i32,
285        #[case] s: i32,
286        #[case] input_t: i32,
287        #[case] input_i: usize,
288        #[case] output_t: i32,
289        #[case] invariant: Vec<u32>,
290        #[case] indeterminacy_dim: usize,
291    ) {
292        let gen = BidegreeGenerator::new(Bidegree::s_t(s, input_t), input_i);
293        let s_2_resolution = resolve_s_2(None, k).unwrap();
294        let p_k = PKData::try_new(k, &None, &s_2_resolution).unwrap();
295        for mi in p_k.mahowald_invariants_for_bidegree(gen.degree()) {
296            if mi.gen.idx() == gen.idx() {
297                assert_eq!(mi.output_t, output_t);
298                assert_eq!(Vec::from(&mi.invariant), invariant);
299                assert_eq!(mi.indeterminacy_basis.len(), indeterminacy_dim);
300                return;
301            }
302        }
303        panic!("could not find Mahowald invariant")
304    }
