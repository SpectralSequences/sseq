1use std::{fs::File, sync::Arc};
2

9
10fn main() -> anyhow::Result<()> {
11    ext::utils::init_logging()?;
12
13    let resolution = Arc::new(ext::utils::query_module(
14        Some(algebra::AlgebraType::Milnor),
15        true,
16    )?);
17
18    let lift = SecondaryResolution::new(Arc::clone(&resolution));
19    lift.extend_all();
20
21    let sseq = lift.e3_page();
22    let products: Vec<_> = resolution
23        .algebra()
24        .default_filtration_one_products()
25        .into_iter()
26        .map(|(name, op_deg, op_idx)| (name, resolution.filtration_one_products(op_deg, op_idx)))
27        .collect();
28
29    let write = |path, page, diff, prod| {
30        const EXT: &str = Backend::<File>::EXT;
31        let backend = Backend::new(File::create(format!(
32            "{}_{}.{}",
33            path,
34            resolution.name(),
35            EXT
36        ))?);
37        sseq.write_to_graph(backend, page, diff, products.iter().take(prod), |_| Ok(()))?;
38        <Result<(), std::io::Error>>::Ok(())
39    };
40
41    write("e2", 2, false, 3)?;
42    write("e2_d2", 2, true, 3)?;
43    write("e3", 3, false, 3)?;
44
45    write("e2_clean", 2, false, 2)?;
46    write("e2_d2_clean", 2, true, 2)?;
47    write("e3_clean", 3, false, 2)?;
48
49    Ok(())
50}
