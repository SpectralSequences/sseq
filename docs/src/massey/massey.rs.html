1//! Computes the triple Massey product up to a sign
2//!
3//! This is optimized to compute <a, b, -> for fixed a, b and all -, where a and b have small
4//! degree.
5
6use std::sync::Arc;
7
8use ext::{chain_complex::ChainComplex, ext_algebra::ExtAlgebra};
9use sseq::coordinates::Bidegree;
10
11fn main() -> anyhow::Result<()> {
12    ext::utils::init_logging()?;
13
14    let resolution = Arc::new(ext::utils::query_module(None, true)?);
15    let e2 = ExtAlgebra::from_resolution(Arc::clone(&resolution))?;
16
17    eprintln!("\nComputing Massey products <a, b, ->");
18    eprintln!("\nEnter a:");
19
20    let a_deg = Bidegree::n_s(
21        query::raw("n of Ext class a", str::parse),
22        query::raw("s of Ext class a", str::parse::<std::num::NonZeroI32>).get(),
23    );
24    e2.unit().compute_through_stem(a_deg);
25    let a_class = query::vector("Input Ext class a", e2.unit_dimension(a_deg));
26    let a = e2.unit_element(a_deg, &a_class);
27
28    eprintln!("\nEnter b:");
29
30    let b_deg = Bidegree::n_s(
31        query::raw("n of Ext class b", str::parse),
32        query::raw("s of Ext class b", str::parse::<std::num::NonZeroI32>).get(),
33    );
34    e2.unit().compute_through_stem(b_deg);
35    let b_class = query::vector("Input Ext class b", e2.unit_dimension(b_deg));
36    let b = e2.unit_element(b_deg, &b_class);
37
38    // The Massey product shifts the bidegree by this amount.
39    let shift = a_deg + b_deg - Bidegree::s_t(1, 0);
40
41    if !e2.is_unit() {
42        e2.unit().compute_through_stem(shift);
43    }
44
45    if !resolution.has_computed_bidegree(shift + Bidegree::s_t(0, resolution.min_degree())) {
46        eprintln!("No computable bidegrees");
47        return Ok(());
48    }
49
50    for (c, result) in e2.massey_iter_c(&a, &b) {
51        println!("<a, b, x_{c}> = {output}", output = result.coset);
52    }
53
54    Ok(())
55}
