1//! Computes the suspension map between different unstable Ext groups.
2//!
3//! Given an unstable Steenrod module $M$, compute the unstable Ext groups of $\Sigma^k M$ for all
4//! $k$ up till the stable range. Each result is printed in the form
5//! ```
6//! n s k: num_gens - matrix
7//! ```
8//! The entries are to be interpreted as follows:
9//!  - `n` is the stem, which is defined to be `t - s - min_degree`
10//!  - `s` is the Adams filtration
11//!  - `k` is the shift
12//!  - `num_gens` is the number of generators in this Ext group
13//!  - `matrix` is the matrix representing the suspension map from $\Sigma^k M$. This is omitted if
14//!    the source or target of the suspension map is trivial, or if they have the same dimension
15//!    and the matrix is the identity matrix.
16//!
17//! The output is best read after sorting with `sort -n -k 1 -k 2 -k 3`.
18
19use std::{path::PathBuf, sync::Arc};
20
21use algebra::module::{Module, SuspensionModule};
22use ext::{
23    chain_complex::{FiniteChainComplex, FreeChainComplex},
24    resolution::UnstableResolution,
25    resolution_homomorphism::UnstableResolutionHomomorphism,
26};
27use fp::vector::FpVector;
28use sseq::coordinates::Bidegree;
29
30fn main() -> anyhow::Result<()> {
31    ext::utils::init_logging()?;
32
33    let module = Arc::new(ext::utils::query_unstable_module_only()?);
34    let save_dir = {
35        let base = query::optional("Module save directory", |x| {
36            core::result::Result::<PathBuf, std::convert::Infallible>::Ok(PathBuf::from(x))
37        });
38        move |shift| {
39            base.as_ref().cloned().map(|mut x| {
40                x.push(format!("suspension{shift}"));
41                x
42            })
43        }
44    };
45
46    let max = Bidegree::n_s(
47        query::raw("Max n", str::parse),
48        query::raw("Max s", str::parse),
49    );
50    let min_degree = Bidegree::s_t(0, module.min_degree());
51
52    let mut res_a;
53    let mut res_b: Arc<UnstableResolution<FiniteChainComplex<_>>> =
54        Arc::new(UnstableResolution::new_with_save(
55            Arc::new(FiniteChainComplex::ccdz(Arc::new(SuspensionModule::new(
56                Arc::clone(&module),
57                0,
58            )))),
59            save_dir(0),
60        )?);
61    res_b.compute_through_stem(max);
62
63    for n in min_degree.n()..=max.n() {
64        for s in 0..=max.s() {
65            let b = Bidegree::n_s(n, s);
66            let source_num_gens = res_b.number_of_gens_in_bidegree(b);
67            println!("{n} {s} 0: {source_num_gens}");
68        }
69    }
70
71    for shift_t in 1..(max - min_degree).n() + 3 {
72        let shift = Bidegree::s_t(0, shift_t);
73        res_a = res_b;
74        res_b = Arc::new(UnstableResolution::new_with_save(
75            Arc::new(FiniteChainComplex::ccdz(Arc::new(SuspensionModule::new(
76                Arc::clone(&module),
77                shift_t,
78            )))),
79            save_dir(shift_t),
80        )?);
81
82        res_b.compute_through_stem(max + shift);
83
84        let suspension_shift = Bidegree::s_t(0, 1);
85        let hom = UnstableResolutionHomomorphism::new(
86            String::from("suspension"),
87            Arc::clone(&res_b),
88            Arc::clone(&res_a),
89            suspension_shift,
90        );
91
92        hom.extend_step_raw(
93            min_degree + shift,
94            Some(vec![FpVector::from_slice(module.prime(), &[1])]),
95        );
96        hom.extend_all();
97
98        for n in 2 * ((min_degree + shift).n() - 1)..=(max + shift).n() {
99            if n < (min_degree + shift).n() {
100                continue;
101            }
102            for s in 0..=max.s() {
103                let source = Bidegree::n_s(n, s);
104                let target = source - suspension_shift;
105                let source_num_gens = res_b.number_of_gens_in_bidegree(source);
106                let target_num_gens = res_a.number_of_gens_in_bidegree(target);
107                let m = if source_num_gens == 0 || target_num_gens == 0 {
108                    String::new()
109                } else {
110                    let m = hom.get_map(target.s()).hom_k(target.t());
111                    if source_num_gens == target_num_gens
112                        && m.iter().enumerate().all(|(n, x)| {
113                            x.iter()
114                                .enumerate()
115                                .all(|(m, &z)| if m == n { z == 1 } else { z == 0 })
116                        })
117                    {
118                        String::new()
119                    } else {
120                        format!(" - {m:?}")
121                    }
122                };
123                println!("{n} {s} {shift_t}: {source_num_gens}{m}", n = n - shift_t);
124            }
125        }
126    }
127
128    Ok(())
129}
