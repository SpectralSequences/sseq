7/// This uses a lookup table for n choose k when n and k are both less than p.
8/// Lucas's theorem reduces general binomial coefficients to this case.
9///
10/// Calling this function safely requires that `k, n < p`.  These invariants are often known
11/// apriori because k and n are obtained by reducing mod p, so it is better to expose an unsafe
12/// interface that avoids these checks.
13unsafe fn direct_binomial(p: ValidPrime, n: usize, k: usize) -> u32 {
14    *BINOMIAL_TABLE
15        .get_unchecked(PRIME_TO_INDEX_MAP[p.as_usize()])
16        .get_unchecked(n)
17        .get_unchecked(k)
18}

20/// A number satisfying the Binomial trait supports computing various binomial coefficients. This
21/// is implemented using a macro, since the implementation for all types is syntactically the same.
22pub trait Binomial: Sized {
23    /// mod 2 multinomial coefficient
24    fn multinomial2(k: &[Self]) -> Self;
25
26    /// mod 2 binomial coefficient n choose k
27    fn binomial2(n: Self, k: Self) -> Self;
28
29    /// Binomial coefficients mod 4. We pre-compute the coefficients for small values of n. For large
30    /// n, we recursively use the fact that if n = 2^k + l, l < 2^k, then
31    ///
32    ///    n choose r = l choose r + 2 (l choose (r - 2^{k - 1})) + (l choose (r - 2^k))
33    ///
34    /// This is easy to verify using the fact that
35    ///
36    ///    (x + y)^{2^k} = x^{2^k} + 2 x^{2^{k - 1}} y^{2^{k - 1}} + y^{2^k}
37    fn binomial4(n: Self, k: Self) -> Self;
38
39    /// Compute binomial coefficients mod 4 using the recursion relation in the documentation of
40    /// [Binomial::binomial4]. This calls into binomial4 instead of binomial4_rec. The main purpose
41    /// of this is to separate out the logic for testing.
42    fn binomial4_rec(n: Self, k: Self) -> Self;
43
44    /// Computes the multinomial coefficient mod p using Lucas' theorem. This modifies the
45    /// underlying list. For p = 2 it is more efficient to use multinomial2
46    fn multinomial_odd(p: ValidPrime, l: &mut [Self]) -> Self;
47
48    /// Compute odd binomial coefficients mod p, where p is odd. For p = 2 it is more efficient to
49    /// use binomial2
50    fn binomial_odd(p: ValidPrime, n: Self, k: Self) -> Self;
51
52    /// Checks whether n choose k is zero mod p. Since we don't have to compute the value, this is
53    /// faster than binomial_odd.
54    fn binomial_odd_is_zero(p: ValidPrime, n: Self, k: Self) -> bool;
55
56    /// Multinomial coefficient of the list l
57    fn multinomial(p: ValidPrime, l: &mut [Self]) -> Self {
58        if p == 2 {
59            Self::multinomial2(l)
60        } else {
61            Self::multinomial_odd(p, l)
62        }
63    }
64
65    /// Binomial coefficient n choose k.
66    fn binomial(p: ValidPrime, n: Self, k: Self) -> Self {
67        if p == 2 {
68            Self::binomial2(n, k)
69        } else {
70            Self::binomial_odd(p, n, k)
71        }
72    }
73}

75macro_rules! impl_binomial {
76    ($T:ty) => {
77        impl Binomial for $T {
78            #[inline]
79            fn multinomial2(l: &[Self]) -> Self {
80                let mut bit_or: Self = 0;
81                let mut sum: Self = 0;
82                for &e in l {
83                    sum += e;
84                    bit_or |= e;
85                }
86                if bit_or == sum {
87                    1
88                } else {
89                    0
90                }
91            }
92
93            #[inline]
94            fn binomial2(n: Self, k: Self) -> Self {
95                if n < k {
96                    0
97                } else if (n - k) & k == 0 {
98                    1
99                } else {
100                    0
101                }
102            }
103
104            #[inline]
105            fn multinomial_odd(p_: ValidPrime, l: &mut [Self]) -> Self {
106                let p = p_.as_u32() as Self;
107
108                let mut n: Self = l.iter().sum();
109                if n == 0 {
110                    return 1;
111                }
112                let mut answer = 1;
113
114                while n > 0 {
115                    let mut multi: Self = 1;
116
117                    let total_entry = n % p;
118                    n /= p;
119
120                    let mut partial_sum: Self = l[0] % p;
121                    l[0] /= p;
122
123                    for ll in l.iter_mut().skip(1) {
124                        let entry = *ll % p;
125                        *ll /= p;
126
127                        partial_sum += entry;
128                        if partial_sum > total_entry {
129                            // This early return is necessary because direct_binomial only works when
130                            // partial_sum < 19
131                            return 0;
132                        }
133                        // This is safe because p < 20, partial_sum <= total_entry < p and entry < p.
134                        multi *=
135                            unsafe { direct_binomial(p_, partial_sum as usize, entry as usize) }
136                                as Self;
137                        multi %= p;
138                    }
139                    answer *= multi;
140                    answer %= p;
141                }
142                answer
143            }
144
145            #[inline]
146            fn binomial_odd(p_: ValidPrime, mut n: Self, mut k: Self) -> Self {
147                let p = p_.as_u32() as Self;
148
149                // We have both signed and unsigned types
150                #[allow(unused_comparisons)]
151                if n < k || k < 0 {
152                    return 0;
153                }
154
155                let mut answer = 1;
156
157                while n > 0 {
158                    // This is safe because p < 20 and anything mod p is < p.
159                    answer *=
160                        unsafe { direct_binomial(p_, (n % p) as usize, (k % p) as usize) } as Self;
161                    answer %= p;
162                    n /= p;
163                    k /= p;
164                }
165                answer
166            }
167
168            #[inline]
169            fn binomial_odd_is_zero(p: ValidPrime, mut n: Self, mut k: Self) -> bool {
170                let p = p.as_u32() as Self;
171
172                while n > 0 {
173                    if n % p < k % p {
174                        return true;
175                    }
176                    n /= p;
177                    k /= p;
178                }
179                false
180            }
181
182            fn binomial4(n: Self, j: Self) -> Self {
183                if (n as usize) < BINOMIAL4_TABLE_SIZE {
184                    return BINOMIAL4_TABLE[n as usize][j as usize] as Self;
185                }
186                if (n - j) & j == 0 {
187                    // Answer is odd
188                    Self::binomial4_rec(n, j)
189                } else if (n - j).count_ones() + j.count_ones() - n.count_ones() == 1 {
190                    2
191                } else {
192                    0
193                }
194            }
195
196            #[inline]
197            fn binomial4_rec(n: Self, j: Self) -> Self {
198                let k = (std::mem::size_of::<Self>() * 8) as u32 - n.leading_zeros() - 1;
199                let l = n - (1 << k);
200                let mut ans = 0;
201                if j <= l {
202                    ans += Self::binomial4(l, j)
203                }
204                let pow = 1 << (k - 1);
205                if pow <= j && j <= l + pow {
206                    ans += 2 * Self::binomial2(l, j - pow);
207                }
208                if j >= (1 << k) {
209                    ans += Self::binomial4(l, j - (1 << k));
210                }
211                ans % 4
212            }
213        }
214    };
215}
