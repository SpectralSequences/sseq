6/// A vector over a finite field.
7///
8/// Interally, it packs entries of the vectors into limbs. However, this is an abstraction that must
9/// not leave the `fp` library.
10#[derive(Debug, Hash, Eq, PartialEq, Clone)]
11pub struct FqVector<F: Field> {
12    pub(super) fq: F,
13    pub(super) len: usize,
14    pub(super) limbs: Vec<Limb>,
15}
16
17/// A slice of an `FqVector`.
18///
19/// This immutably borrows the vector and implements `Copy`.
20#[derive(Debug, Copy, Clone)]
21pub struct FqSlice<'a, F: Field> {
22    pub(super) fq: F,
23    pub(super) limbs: &'a [Limb],
24    pub(super) start: usize,
25    pub(super) end: usize,
26}
27
28/// A mutable slice of an `FqVector`.
29///
30/// This mutably borrows the vector. Since it is a mutable borrow, it cannot implement `Copy`.
31/// However, it has a [`FqSliceMut::copy`] function that imitates the reborrowing, that mutably
32/// borrows `FqSliceMut` and returns a `FqSliceMut` with a shorter lifetime.
33#[derive(Debug)]
34pub struct FqSliceMut<'a, F: Field> {
35    pub(super) fq: F,
36    pub(super) limbs: &'a mut [Limb],
37    pub(super) start: usize,
38    pub(super) end: usize,
39}
