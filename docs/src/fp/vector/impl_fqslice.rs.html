16impl<'a, F: Field> FqSlice<'a, F> {
17    pub fn prime(&self) -> ValidPrime {
18        self.fq.characteristic().to_dyn()
19    }
20
21    pub fn len(&self) -> usize {
22        self.end - self.start
23    }
24
25    pub const fn is_empty(&self) -> bool {
26        self.start == self.end
27    }
28
29    pub fn entry(&self, index: usize) -> FieldElement<F> {
30        debug_assert!(
31            index < self.len(),
32            "Index {} too large, length of vector is only {}.",
33            index,
34            self.len()
35        );
36        let bit_mask = self.fq.bitmask();
37        let limb_index = self.fq.limb_bit_index_pair(index + self.start);
38        let mut result = self.limbs[limb_index.limb];
39        result >>= limb_index.bit_index;
40        result &= bit_mask;
41        self.fq.decode(result)
42    }
43
44    /// TODO: implement prime 2 version
45    pub fn iter(self) -> FqVectorIterator<'a, F> {
46        FqVectorIterator::new(self)
47    }
48
49    pub fn iter_nonzero(self) -> FqVectorNonZeroIterator<'a, F> {
50        FqVectorNonZeroIterator::new(self)
51    }
52
53    pub fn is_zero(&self) -> bool {
54        let limb_range = self.limb_range();
55        if limb_range.is_empty() {
56            return true;
57        }
58        let (min_mask, max_mask) = self.limb_masks();
59        if self.limbs[limb_range.start] & min_mask != 0 {
60            return false;
61        }
62
63        let inner_range = self.limb_range_inner();
64        if !inner_range.is_empty() && self.limbs[inner_range].iter().any(|&x| x != 0) {
65            return false;
66        }
67        if self.limbs[limb_range.end - 1] & max_mask != 0 {
68            return false;
69        }
70        true
71    }
72
73    #[must_use]
74    pub fn slice(self, start: usize, end: usize) -> Self {
75        assert!(start <= end && end <= self.len());
76
77        FqSlice {
78            fq: self.fq,
79            limbs: self.limbs,
80            start: self.start + start,
81            end: self.start + end,
82        }
83    }
84
85    /// Converts a slice to an owned FqVector. This is vastly more efficient if the start of the vector is aligned.
86    #[must_use]
87    pub fn to_owned(self) -> FqVector<F> {
88        let mut new = FqVector::new(self.fq, self.len());
89        if self.start.is_multiple_of(self.fq.entries_per_limb()) {
90            let limb_range = self.limb_range();
91            new.limbs[0..limb_range.len()].copy_from_slice(&self.limbs[limb_range]);
92            if !new.limbs.is_empty() {
93                let len = new.limbs.len();
94                new.limbs[len - 1] &= self.limb_masks().1;
95            }
96        } else {
97            new.as_slice_mut().assign(self);
98        }
99        new
100    }
101}

103// Limb methods
104impl<F: Field> FqSlice<'_, F> {
105    #[inline]
106    pub(super) fn offset(&self) -> usize {
107        let bit_length = self.fq.bit_length();
108        let entries_per_limb = self.fq.entries_per_limb();
109        (self.start % entries_per_limb) * bit_length
110    }
111
112    #[inline]
113    pub(super) fn limb_range(&self) -> std::ops::Range<usize> {
114        self.fq.range(self.start, self.end)
115    }
116
117    /// This function underflows if `self.end == 0`, which happens if and only if we are taking a
118    /// slice of width 0 at the start of an `FpVector`. This should be a very rare edge case.
119    /// Dealing with the underflow properly would probably require using `saturating_sub` or
120    /// something of that nature, and that has a nontrivial (10%) performance hit.
121    #[inline]
122    pub(super) fn limb_range_inner(&self) -> std::ops::Range<usize> {
123        let range = self.limb_range();
124        (range.start + 1)..(usize::max(range.start + 1, range.end - 1))
125    }
126
127    #[inline(always)]
128    pub(super) fn min_limb_mask(&self) -> Limb {
129        !0 << self.offset()
130    }
131
132    #[inline(always)]
133    pub(super) fn max_limb_mask(&self) -> Limb {
134        let num_entries = 1 + (self.end - 1) % self.fq.entries_per_limb();
135        let bit_max = num_entries * self.fq.bit_length();
136
137        (!0) >> (constants::BITS_PER_LIMB - bit_max)
138    }
139
140    #[inline(always)]
141    pub(super) fn limb_masks(&self) -> (Limb, Limb) {
142        if self.limb_range().len() == 1 {
143            (
144                self.min_limb_mask() & self.max_limb_mask(),
145                self.min_limb_mask() & self.max_limb_mask(),
146            )
147        } else {
148            (self.min_limb_mask(), self.max_limb_mask())
149        }
150    }
151}

159impl<F: Field> std::fmt::Display for FqSlice<'_, F> {
160    /// # Example
161    /// ```
162    /// # use fp::field::{Field, SmallFq};
163    /// # use fp::prime::{P2, ValidPrime};
164    /// # use fp::vector::FqVector;
165    /// let fq = SmallFq::new(P2, 3);
166    /// let v = FqVector::from_slice(fq, &[fq.zero(), fq.one(), fq.a(), fq.a() * fq.a()]);
167    /// assert_eq!(&format!("{v}"), "[0, 1, a, a^2]");
168    ///
169    /// // This only looks reasonable over prime fields of order less than 10
170    /// assert_eq!(&format!("{v:#}"), "01aa^2");
171    /// ```
172    fn fmt(&self, f: &mut std::fmt::Formatter) -> std::fmt::Result {
173        if f.alternate() {
174            for v in self.iter() {
175                // If self.p >= 11, this will look funky
176                write!(f, "{v}")?;
177            }
178            Ok(())
179        } else {
180            write!(f, "[{}]", self.iter().format(", "))
181        }
182    }
