6cfg_if::cfg_if! {

18pub(crate) unsafe fn load(limb: *const Limb) -> SimdLimb {
19    cfg_if::cfg_if! {
20        if #[cfg(target_feature="avx2")] {
21            x86_64::_mm256_loadu_si256(limb as *const SimdLimb)
22        } else if #[cfg(target_feature="avx")] {
23            x86_64::_mm256_loadu_ps(limb as *const f32)
24        } else if #[cfg(target_feature="sse2")] {
25            x86_64::_mm_loadu_si128(limb as *const SimdLimb)
26        } else {
27            *limb
28        }
29    }
30}

32pub(crate) unsafe fn store(limb: *mut Limb, val: SimdLimb) {
33    cfg_if::cfg_if! {
34        if #[cfg(target_feature="avx2")] {
35            x86_64::_mm256_storeu_si256(limb as *mut SimdLimb, val);
36        } else if #[cfg(target_feature="avx")] {
37            x86_64::_mm256_storeu_ps(limb as *mut f32, val);
38        } else if #[cfg(target_feature="sse2")] {
39            x86_64::_mm_storeu_si128(limb as *mut SimdLimb, val)
40        } else {
41            *limb = val;
42        }
43    }
44}

46pub(crate) unsafe fn xor(left: SimdLimb, right: SimdLimb) -> SimdLimb {
47    cfg_if::cfg_if! {
48        if #[cfg(target_feature="avx2")] {
49            x86_64::_mm256_xor_si256(left, right)
50        } else if #[cfg(target_feature="avx")] {
51            x86_64::_mm256_xor_ps(left, right)
52        } else if #[cfg(target_feature="sse2")] {
53            x86_64::_mm_xor_si128(left, right)
54        } else {
55            left ^ right
56        }
57    }
