12/// Given a matrix M, a quasi-inverse Q is a map from the co-domain to the domain such that xQM = x
13/// for all x in the image (recall our matrices act on the right).
14///
15/// # Fields
16///  * `image` - The image of the original matrix. If the image is omitted, it is assumed to be
17///    everything (with the standard basis).
18///  * `preimage` - The actual quasi-inverse, where the basis of the image is that given by
19///    `image`.
20#[derive(Debug, Clone, PartialEq, Eq)]
21pub struct QuasiInverse {
22    image: Option<Vec<isize>>,
23    preimage: Matrix,
24}
25
26impl QuasiInverse {
27    pub fn new(image: Option<Vec<isize>>, preimage: Matrix) -> Self {
28        Self { image, preimage }
29    }
30
31    pub fn image_dimension(&self) -> usize {
32        self.preimage.rows()
33    }
34
35    pub fn source_dimension(&self) -> usize {
36        self.preimage.columns()
37    }
38
39    pub fn target_dimension(&self) -> usize {
40        match self.image.as_ref() {
41            Some(v) => v.len(),
42            None => self.image_dimension(),
43        }
44    }
45
46    pub fn to_bytes(&self, buffer: &mut impl io::Write) -> io::Result<()> {
47        buffer.write_u64::<LittleEndian>(self.source_dimension() as u64)?;
48        buffer.write_u64::<LittleEndian>(self.target_dimension() as u64)?;
49        buffer.write_u64::<LittleEndian>(self.image_dimension() as u64)?;
50
51        match self.image.as_ref() {
52            None => {
53                for i in 0..self.preimage.rows() {
54                    buffer.write_i64::<LittleEndian>(i as i64)?;
55                }
56            }
57            Some(v) => {
58                Matrix::write_pivot(v, buffer)?;
59            }
60        }
61        self.preimage.to_bytes(buffer)
62    }
63
64    pub fn from_bytes(p: ValidPrime, data: &mut impl io::Read) -> io::Result<Self> {
65        let source_dim = data.read_u64::<LittleEndian>()? as usize;
66        let target_dim = data.read_u64::<LittleEndian>()? as usize;
67        let image_dim = data.read_u64::<LittleEndian>()? as usize;
68
69        let image = Matrix::read_pivot(target_dim, data)?;
70        let preimage = Matrix::from_bytes(p, image_dim, source_dim, data)?;
71        Ok(Self {
72            image: Some(image),
73            preimage,
74        })
75    }
76
77    /// Given a data file containing a quasi-inverse, apply it to all the vectors in `input`
78    /// and write the results to the corresponding vectors in `results`. This reads in the
79    /// quasi-inverse row by row to minimize memory usage.
80    pub fn stream_quasi_inverse<T, S>(
81        p: ValidPrime,
82        data: &mut impl io::Read,
83        results: &mut [T],
84        inputs: &[S],
85    ) -> io::Result<()>
86    where
87        for<'a> &'a mut T: Into<FpSliceMut<'a>>,
88        for<'a> &'a S: Into<FpSlice<'a>>,
89    {
90        let source_dim = data.read_u64::<LittleEndian>()? as usize;
91        let target_dim = data.read_u64::<LittleEndian>()? as usize;
92        let _image_dim = data.read_u64::<LittleEndian>()? as usize;
93
94        let image = Matrix::read_pivot(target_dim, data)?;
95        let mut v = FpVector::new(p, source_dim);
96
97        assert_eq!(results.len(), inputs.len());
98        for result in &mut *results {
99            assert_eq!(result.into().as_slice().len(), source_dim);
100        }
101
102        for (i, r) in image.into_iter().enumerate() {
103            if r < 0 {
104                continue;
105            }
106
107            v.update_from_bytes(data)?;
108            for (input, result) in inputs.iter().zip_eq(&mut *results) {
109                result.into().add(v.as_slice(), input.into().entry(i));
110            }
111        }
112        Ok(())
113    }
114
115    pub fn preimage(&self) -> &Matrix {
116        &self.preimage
117    }
118
119    pub fn pivots(&self) -> Option<&[isize]> {
120        self.image.as_deref()
121    }
122
123    pub fn prime(&self) -> ValidPrime {
124        self.preimage.prime()
125    }
126
127    /// Apply the quasi-inverse to an input vector and add a constant multiple of the result
128    /// to an output vector
129    ///
130    /// # Arguments
131    ///  * `target` - The output vector
132    ///  * `coeff` - The constant multiple above
133    ///  * `input` - The input vector, expressed in the basis of the ambient space
134    pub fn apply(&self, mut target: FpSliceMut, coeff: u32, input: FpSlice) {
135        let p = self.prime();
136        let mut row = 0;
137        for (i, c) in input.iter().enumerate() {
138            if let Some(pivots) = self.pivots() {
139                if i >= pivots.len() || pivots[i] < 0 {
140                    continue;
141                }
142            }
143            if c != 0 {
144                target.add(self.preimage[row].as_slice(), (coeff * c) % p);
145            }
146            row += 1;
147        }
148    }

152mod tests {
153    use super::*;
154
155    #[test]
156    fn test_stream_qi() {
157        let p = ValidPrime::new(2);
158        let qi = QuasiInverse {
159            image: Some(vec![0, -1, 1, -1, 2, 3]),
160            preimage: Matrix::from_vec(
161                p,
162                &[
163                    vec![1, 0, 1, 1],
164                    vec![1, 1, 0, 0],
165                    vec![0, 1, 0, 1],
166                    vec![1, 1, 1, 0],
167                ],
168            ),
169        };
170        let v0 = FpVector::from_slice(p, &[1, 1, 0, 0, 1, 0]);
171        let v1 = FpVector::from_slice(p, &[0, 0, 1, 0, 1, 1]);
172
173        let mut out0 = FpVector::new(p, 4);
174        let mut out1 = FpVector::new(p, 4);
175
176        let mut cursor = io::Cursor::new(Vec::<u8>::new());
177        qi.to_bytes(&mut cursor).unwrap();
178        cursor.set_position(0);
179
180        QuasiInverse::stream_quasi_inverse(
181            p,
182            &mut cursor,
183            &mut [out0.as_slice_mut(), out1.as_slice_mut()],
184            &[v0.as_slice(), v1.as_slice()],
185        )
186        .unwrap();
187
188        let mut bench0 = FpVector::new(p, 4);
189        let mut bench1 = FpVector::new(p, 4);
190
191        qi.apply(bench0.as_slice_mut(), 1, v0.as_slice());
192        qi.apply(bench1.as_slice_mut(), 1, v1.as_slice());
193
194        assert_eq!(out0, bench0, "{out0} != {bench0}");
195        assert_eq!(out1, bench1, "{out1} != {bench1}");
196
197        assert_eq!(cursor.position() as usize, cursor.get_ref().len());
198    }
