53impl<P: Prime> FieldInternal for Fp<P> {
54    type ElementContainer = u32;
55
56    fn el(self, value: Self::ElementContainer) -> FieldElement<Self> {
57        FieldElement::new(self, value % self.p.as_u32())
58    }
59
60    fn add_assign(self, a: &mut FieldElement<Self>, b: FieldElement<Self>) {
61        a.value = self.p.sum(**a, *b);
62    }
63
64    fn mul_assign(self, a: &mut FieldElement<Self>, b: FieldElement<Self>) {
65        a.value = self.p.product(**a, *b);
66    }
67
68    fn inv(self, a: FieldElement<Self>) -> Option<FieldElement<Self>> {
69        if *a == 0 {
70            None
71        } else {
72            Some(self.el(crate::prime::inverse(self.p, *a)))
73        }
74    }
75
76    fn neg(self, a: FieldElement<Self>) -> FieldElement<Self> {
77        self.el(if *a == 0 { 0 } else { self.p.as_u32() - *a })
78    }
79
80    fn frobenius(self, a: FieldElement<Self>) -> FieldElement<Self> {
81        a
82    }
83
84    fn encode(self, element: FieldElement<Self>) -> Limb {
85        element.value as Limb
86    }
87
88    fn decode(self, element: Limb) -> FieldElement<Self> {
89        // We have to pass in the already reduced value to `Self::el` because we have no guarantee
90        // that this Limb fits in a u32. For example, `element` could be the result of `fma_limb(0,
91        // 1_000_000, 1_000_000)`, if the prime is large enough.
92        let prime_limb = self.p.as_u32() as Limb;
93        self.el((element % prime_limb) as u32)
94    }
95
96    fn bit_length(self) -> usize {
97        let p = self.characteristic().as_u32() as u64;
98        match p {
99            // 2 is a special case b/c bitwise xor does the add and reduce together so we only need enough bits to fit p-1.
100            2 => 1,
101            _ => (BITS_PER_LIMB as u32 - (p * (p - 1)).leading_zeros()) as usize,
102        }
103    }
104
105    fn fma_limb(self, limb_a: Limb, limb_b: Limb, coeff: FieldElement<Self>) -> Limb {
106        if self.characteristic() == 2 {
107            limb_a ^ (coeff.value as Limb * limb_b)
108        } else {
109            limb_a + (coeff.value as Limb) * limb_b
110        }
111    }
112
113    /// Contributed by Robert Burklund.
114    fn reduce(self, limb: Limb) -> Limb {
115        match self.characteristic().as_u32() {
116            2 => limb,
117            3 => {
118                // Set top bit to 1 in every limb
119                const TOP_BIT: Limb = (!0 / 7) << (2 - BITS_PER_LIMB % 3);
120                let mut limb_2 = ((limb & TOP_BIT) >> 2) + (limb & (!TOP_BIT));
121                let mut limb_3s = limb_2 & (limb_2 >> 1);
122                limb_3s |= limb_3s << 1;
123                limb_2 ^= limb_3s;
124                limb_2
125            }
126            5 => {
127                // Set bottom bit to 1 in every limb
128                const BOTTOM_BIT: Limb = (!0 / 31) >> (BITS_PER_LIMB % 5);
129                const BOTTOM_TWO_BITS: Limb = BOTTOM_BIT | (BOTTOM_BIT << 1);
130                const BOTTOM_THREE_BITS: Limb = BOTTOM_BIT | (BOTTOM_TWO_BITS << 1);
131                let a = (limb >> 2) & BOTTOM_THREE_BITS;
132                let b = limb & BOTTOM_TWO_BITS;
133                let m = (BOTTOM_BIT << 3) - a + b;
134                let mut c = (m >> 3) & BOTTOM_BIT;
135                c |= c << 1;
136                let d = m & BOTTOM_THREE_BITS;
137                d + c - BOTTOM_TWO_BITS
138            }
139            // Slow generic fallback. If anyone cares enough about some larger prime, they can add a faster implementation
140            _ => self.pack(self.unpack(limb)),
141        }
142    }
143}
