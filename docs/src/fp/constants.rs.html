6macro_rules! const_for {
7    ($i:ident in $a:literal.. $b:ident $contents:block) => {
8        let mut $i = $a;
9        while $i < $b {
10            $contents;
11            $i += 1;
12        }
13    };
14}

16pub(crate) static INVERSE_TABLE: [[u32; MAX_PRIME]; NUM_PRIMES] = const {
17    let mut result = [[0; MAX_PRIME]; NUM_PRIMES];
18    const_for! { i in 0 .. NUM_PRIMES {
19        let p = PRIMES[i];
20        const_for! { k in 1 .. p {
21            result[i][k as usize] = crate::prime::power_mod(p, k, p - 2);
22        }}
23    }};
24    result

27macro_rules! populate_binomial_table {
28    ($res:expr, $size:ident, $mod:expr) => {
29        const_for! { n in 0 .. $size {
30            $res[n][0] = 1;
31            const_for! { k in 0 .. n {
32                $res[n][k + 1] = ($res[n - 1][k] + $res[n - 1][k + 1]) % $mod;
33            }}
34        }}
35    };
36}
