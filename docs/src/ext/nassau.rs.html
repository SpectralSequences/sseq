1//! This module implements [Nassau's algorithm](https://arxiv.org/abs/1910.04063).
2//!
3//! The main export is the [`Resolution`] object, which is a resolution of the sphere at the prime 2
4//! using Nassau's algorithm. It aims to provide an API similar to
5//! [`resolution::Resolution`](crate::resolution::Resolution). From an API point of view, the main
6//! difference between the two is that our `Resolution` is a chain complex over [`MilnorAlgebra`]
7//! over [`SteenrodAlgebra`](algebra::SteenrodAlgebra).
8//!
9//! To make use of this resolution in the example scripts, enable the `nassau` feature. This will
10//! cause [`utils::query_module`](crate::utils::query_module) to return the `Resolution` from this
11//! module instead of [`resolution`](crate::resolution). There is no formal polymorphism involved;
12//! the feature changes the return type of the function. While this is an incorrect use of features,
13//! we find that this the easiest way to make all scripts support both types of resolutions.
14

66/// A Milnor subalgebra to be used in [Nassau's algorithm](https://arxiv.org/abs/1910.04063). This
67/// is equipped with an ordering of the signature as in Lemma 2.4 of the paper.
68///
69/// To simplify implementation, we pick the ordering so that the (reverse) lexicographic ordering
70/// in Lemma 2.4 is just the (reverse) lexicographic ordering of the P parts. This corresponds to
71/// the ordering of $\mathcal{P}$ where $P^s_t < P^{s'}_t$ if $s < s'$).
72#[derive(Clone)]
73struct MilnorSubalgebra {
74    profile: Vec<u8>,
75}
76
77impl MilnorSubalgebra {
78    /// This should be used when you want an entry of the profile to be infinity
79    #[allow(dead_code)]
80    const INFINITY: u8 = (std::mem::size_of::<PPartEntry>() * 4 - 1) as u8;
81
82    fn new(profile: Vec<u8>) -> Self {
83        Self { profile }
84    }
85
86    /// The algebra with trivial profile, corresponding to the trivial algebra.
87    fn zero_algebra() -> Self {
88        Self { profile: vec![] }
89    }
90
91    /// Computes the signature of an element
92    fn has_signature(&self, ppart: &[PPartEntry], signature: &[PPartEntry]) -> bool {
93        for (i, (&profile, &signature)) in self.profile.iter().zip(signature).enumerate() {
94            let ppart = ppart.get(i).copied().unwrap_or(0);
95            if ppart & ((1 << profile) - 1) != signature {
96                return false;
97            }
98        }
99        true
100    }
101
102    fn zero_signature(&self) -> Vec<PPartEntry> {
103        vec![0; self.profile.len()]
104    }
105
106    /// Give a list of basis elements in degree `degree` that has signature `signature`.
107    ///
108    /// This requires passing the algebra for borrow checker reasons.
109    fn signature_mask<'a>(
110        &'a self,
111        algebra: &'a MilnorAlgebra,
112        module: &'a FreeModule<MilnorAlgebra>,
113        degree: i32,
114        signature: &'a [PPartEntry],
115    ) -> impl Iterator<Item = usize> + 'a {
116        module.iter_gen_offsets([degree]).flat_map(
117            move |GeneratorData {
118                      gen_deg,
119                      start: [offset],
120                      end: _,
121                  }| {
122                algebra
123                    .ppart_table(degree - gen_deg)
124                    .iter()
125                    .enumerate()
126                    .filter_map(move |(n, op)| {
127                        if self.has_signature(op, signature) {
128                            Some(offset + n)
129                        } else {
130                            None
131                        }
132                    })
133            },
134        )
135    }
136
137    /// Get the matrix of a free module homomorphism when restricted to the subquotient given by
138    /// the signature.
139    fn signature_matrix(
140        &self,
141        hom: &FreeModuleHomomorphism<FreeModule<MilnorAlgebra>>,
142        degree: i32,
143        signature: &[PPartEntry],
144    ) -> Matrix {
145        let p = hom.prime();
146        let source = hom.source();
147        let target = hom.target();
148        let algebra = target.algebra();
149        let target_degree = degree - hom.degree_shift();
150
151        let target_mask: Vec<usize> = self
152            .signature_mask(&algebra, &target, degree - hom.degree_shift(), signature)
153            .collect();
154
155        let source_mask: Vec<usize> = self
156            .signature_mask(&algebra, &source, degree, signature)
157            .collect();
158
159        let mut scratch = FpVector::new(p, target.dimension(target_degree));
160        let mut result = Matrix::new(p, source_mask.len(), target_mask.len());
161
162        for (row, &masked_index) in std::iter::zip(result.iter_mut(), &source_mask) {
163            scratch.set_to_zero();
164            hom.apply_to_basis_element(scratch.as_slice_mut(), 1, degree, masked_index);
165
166            row.as_slice_mut()
167                .add_masked(scratch.as_slice(), 1, &target_mask);
168        }
169        result
170    }
171
172    /// Iterate through all signatures of this algebra that contain elements of degree at most
173    /// `degree` (inclusive). This skips the initial zero signature.
174    fn iter_signatures(&self, degree: i32) -> impl Iterator<Item = Vec<PPartEntry>> + '_ {
175        SignatureIterator::new(self, degree)
176    }
177
178    fn top_degree(&self) -> i32 {
179        self.profile
180            .iter()
181            .map(|&entry| (1 << entry) - 1)
182            .enumerate()
183            .map(|(idx, entry)| ((1 << (idx + 1)) - 1) * entry)
184            .sum()
185    }
186
187    fn optimal_for(b: Bidegree) -> Self {
188        let b_is_in_vanishing_region = |subalgebra: &Self| {
189            let coeff = (1 << subalgebra.profile.len()) - 1;
190            b.t() >= coeff * (b.s() + 1) + subalgebra.top_degree()
191        };
192        SubalgebraIterator::new()
193            .take_while(b_is_in_vanishing_region)
194            .last()
195            .unwrap_or(Self::zero_algebra())
196    }
197
198    fn to_bytes(&self, buffer: &mut impl io::Write) -> io::Result<()> {
199        buffer.write_u64::<LittleEndian>(self.profile.len() as u64)?;
200        buffer.write_all(&self.profile)?;
201
202        let len = self.profile.len();
203        let zeros = [0; 8];
204        let padding = len - ((len / 8) * 8);
205        buffer.write_all(&zeros[0..padding])
206    }
207
208    fn from_bytes(data: &mut impl io::Read) -> io::Result<Self> {
209        let len = data.read_u64::<LittleEndian>()? as usize;
210        let mut profile = vec![0; len];
211
212        data.read_exact(&mut profile)?;
213
214        let padding = len - ((len / 8) * 8);
215        if padding > 0 {
216            let mut buf: [u8; 8] = [0; 8];
217            data.read_exact(&mut buf[0..padding])?;
218            assert_eq!(buf, [0; 8]);
219        }
220        Ok(Self { profile })
221    }
222
223    fn signature_to_bytes(signature: &[PPartEntry], buffer: &mut impl io::Write) -> io::Result<()> {
224        if cfg!(target_endian = "little") && std::mem::size_of::<PPartEntry>() == 2 {
225            unsafe {
226                let buf: &[u8] = std::slice::from_raw_parts(
227                    signature.as_ptr() as *const u8,
228                    signature.len() * 2,
229                );
230                buffer.write_all(buf).unwrap();
231            }
232        } else {
233            for &entry in signature {
234                buffer.write_u16::<LittleEndian>(entry as u16)?;
235            }
236        }
237
238        let len = signature.len();
239        let zeros = [0; 8];
240        let padding = len - ((len / 4) * 4);
241
242        if padding > 0 {
243            buffer.write_all(&zeros[0..padding * 2])?;
244        }
245        Ok(())
246    }
247
248    fn signature_from_bytes(&self, data: &mut impl io::Read) -> io::Result<Vec<PPartEntry>> {
249        let len = self.profile.len();
250        let mut signature: Vec<PPartEntry> = vec![0; len];
251
252        if cfg!(target_endian = "little") && std::mem::size_of::<PPartEntry>() == 2 {
253            unsafe {
254                let buf: &mut [u8] =
255                    std::slice::from_raw_parts_mut(signature.as_mut_ptr() as *mut u8, len * 2);
256                data.read_exact(buf).unwrap();
257            }
258        } else {
259            for entry in &mut signature {
260                *entry = data.read_u16::<LittleEndian>()? as PPartEntry;
261            }
262        }
263
264        let padding = len - ((len / 4) * 4);
265        if padding > 0 {
266            let mut buffer: [u8; 8] = [0; 8];
267            data.read_exact(&mut buffer[0..padding * 2])?;
268            assert_eq!(buffer, [0; 8]);
269        }
270        Ok(signature)
271    }
272}

274impl Display for MilnorSubalgebra {
275    fn fmt(&self, out: &mut std::fmt::Formatter<'_>) -> std::result::Result<(), std::fmt::Error> {
276        if self.profile.is_empty() {
277            write!(out, "F_2")
278        } else if self.profile.len() as u8 == self.profile[0] {

282        }

286/// An iterator that iterates through a sequence of [`MilnorSubalgebra`] of increasing size. This
287/// is used by [`MilnorSubalgebra::optimal_for`] to find the largest subalgebra in this sequence
288/// that is applicable to a bidegree.
289struct SubalgebraIterator {
290    current: MilnorSubalgebra,
291}
292
293impl SubalgebraIterator {
294    fn new() -> Self {
295        Self {
296            current: MilnorSubalgebra::new(vec![]),
297        }
298    }
299}
300
301impl Iterator for SubalgebraIterator {
302    type Item = MilnorSubalgebra;
303
304    fn next(&mut self) -> Option<Self::Item> {
305        if self.current.profile.is_empty()
306            || self.current.profile[0] == self.current.profile.len() as u8
307        {
308            // We are at F_2 or at A(n) where n = self.current.profile.len() - 1.
309            self.current.profile.push(1);
310            Some(self.current.clone())
311        } else {
312            // We find the first entry that can be incremented and increment it
313            if let Some((_, entry)) = self
314                .current
315                .profile
316                .iter_mut()
317                .rev()
318                .enumerate()
319                .find(|(idx, entry)| **entry == *idx as u8)
320            {
321                *entry += 1;
322            }
323            Some(self.current.clone())
324        }
325    }
326}

328/// See [`MilnorSubalgebra::iter_signatures`].
329struct SignatureIterator<'a> {
330    subalgebra: &'a MilnorSubalgebra,
331    current: Vec<PPartEntry>,
332    signature_degree: i32,
333    degree: i32,
334}
335
336impl<'a> SignatureIterator<'a> {
337    fn new(subalgebra: &'a MilnorSubalgebra, degree: i32) -> Self {
338        Self {
339            current: vec![0; subalgebra.profile.len()],
340            degree,
341            subalgebra,
342            signature_degree: 0,
343        }
344    }
345}
346
347impl Iterator for SignatureIterator<'_> {
348    type Item = Vec<PPartEntry>;
349
350    fn next(&mut self) -> Option<Self::Item> {
351        let xi_degrees = combinatorics::xi_degrees(TWO);
352        let len = self.current.len();
353        for (i, current) in self.current.iter_mut().enumerate() {
354            *current += 1;
355            self.signature_degree += xi_degrees[i];
356
357            if self.signature_degree > self.degree || *current == 1 << self.subalgebra.profile[i] {
358                self.signature_degree -= xi_degrees[i] * *current as i32;
359                *current = 0;
360                if i + 1 == len {
361                    return None;
362                }
363            } else {
364                return Some(self.current.clone());
365            }
366        }
367        // This only happens when the profile is trivial
368        assert!(self.current.is_empty());
369        None
370    }
371}

380/// A resolution of `S_2` using Nassau's algorithm.
381///
382/// This aims to have an API similar to that of
383/// [`resolution::Resolution`](crate::resolution::Resolution). From an API point of view, the main
384/// difference between the two is that this is a chain complex over [`MilnorAlgebra`] over
385/// [`SteenrodAlgebra`](algebra::SteenrodAlgebra).
386pub struct Resolution<M: ZeroModule<Algebra = MilnorAlgebra>> {
387    lock: Mutex<()>,
388    name: String,
389    max_degree: i32,
390    modules: OnceBiVec<Arc<FreeModule<MilnorAlgebra>>>,
391    zero_module: Arc<FreeModule<MilnorAlgebra>>,
392    differentials: OnceBiVec<Arc<FreeModuleHomomorphism<FreeModule<MilnorAlgebra>>>>,
393    target: Arc<FiniteChainComplex<M>>,
394    chain_maps: OnceBiVec<Arc<FreeModuleHomomorphism<M>>>,
395    save_dir: SaveDirectory,
396}

398impl<M: ZeroModule<Algebra = MilnorAlgebra>> Resolution<M> {
399    pub fn name(&self) -> &str {
400        &self.name
401    }
402
403    pub fn set_name(&mut self, name: String) {
404        self.name = name;
405    }

407    pub fn new(module: Arc<M>) -> Self {
408        Self::new_with_save(module, None).unwrap()
409    }

411    pub fn new_with_save(
412        module: Arc<M>,
413        save_dir: impl Into<SaveDirectory>,
414    ) -> anyhow::Result<Self> {
415        let save_dir = save_dir.into();
416        let max_degree = module
417            .max_degree()
418            .ok_or_else(|| anyhow!("Nassau's algorithm requires bounded module"))?;
419        let target = Arc::new(FiniteChainComplex::ccdz(module));
420
421        if let Some(p) = save_dir.write() {
422            for subdir in SaveKind::nassau_data() {
423                subdir.create_dir(p)?;
424            }
425        }
426
427        Ok(Self {
428            lock: Mutex::new(()),
429            zero_module: Arc::new(FreeModule::new(target.algebra(), "F_{-1}".to_string(), 0)),
430            name: String::new(),
431            modules: OnceBiVec::new(0),
432            differentials: OnceBiVec::new(0),
433            chain_maps: OnceBiVec::new(0),
434            target,
435            max_degree,
436            save_dir,
437        })
438    }

440    fn add_generators(&self, b: Bidegree, num_new_gens: usize) {
441        let gen_names = (0..num_new_gens)
442            .map(|idx| format!("x_{:#}", BidegreeGenerator::new(b, idx)))
443            .collect();
444        self.module(b.s())
445            .add_generators(b.t(), num_new_gens, Some(gen_names));
446    }

448    /// This function prepares the Resolution object to perform computations up to the
449    /// specified s degree. It does *not* perform any computations by itself. It simply lengthens
450    /// the `OnceVec`s `modules`, `chain_maps`, etc. to the right length.
451    fn extend_through_degree(&self, max_s: i32) {
452        let min_degree = self.min_degree();
453
454        self.modules.extend(max_s, |i| {
455            Arc::new(FreeModule::new(
456                Arc::clone(&self.algebra()),
457                format!("F{i}"),
458                min_degree,
459            ))
460        });
461
462        self.differentials.extend(0, |_| {
463            Arc::new(FreeModuleHomomorphism::new(
464                Arc::clone(&self.modules[0]),
465                Arc::clone(&self.zero_module),
466                0,
467            ))
468        });
469
470        self.differentials.extend(max_s, |i| {
471            Arc::new(FreeModuleHomomorphism::new(
472                Arc::clone(&self.modules[i]),
473                Arc::clone(&self.modules[i - 1]),
474                0,
475            ))
476        });
477
478        self.chain_maps.extend(max_s, |i| {
479            Arc::new(FreeModuleHomomorphism::new(
480                Arc::clone(&self.modules[i]),
481                self.target.module(i),
482                0,
483            ))
484        });
485    }

487    #[tracing::instrument(skip_all, fields(signature = ?signature, throughput))]
488    fn write_qi(
489        f: &mut Option<impl io::Write>,
490        scratch: &mut FpVector,
491        signature: &[PPartEntry],
492        next_mask: &[usize],
493        full_matrix: &Matrix,
494        masked_matrix: &AugmentedMatrix<2>,
495    ) -> io::Result<()> {
496        let f = match f {
497            Some(f) => f,
498            None => return Ok(()),
499        };
500
501        let mut own_f = LogWriter::new(f);
502        let f = &mut own_f;
503
504        let pivots = &masked_matrix.pivots()[0..masked_matrix.end[0]];
505        if !pivots.iter().any(|&x| x >= 0) {
506            return Ok(());
507        }
508
509        // Write signature if non-zero.
510        if signature.iter().any(|&x| x > 0) {
511            f.write_u64::<LittleEndian>(Magic::Signature as u64)?;
512            MilnorSubalgebra::signature_to_bytes(signature, f)?;
513        }
514
515        // Write quasi-inverses
516        for (col, &row) in pivots.iter().enumerate() {
517            if row < 0 {
518                continue;
519            }
520            f.write_u64::<LittleEndian>(next_mask[col] as u64)?;
521            let preimage = masked_matrix.row_segment(row as usize, 1, 1);
522            scratch.set_scratch_vector_size(preimage.len());
523            scratch.as_slice_mut().assign(preimage);
524            scratch.to_bytes(f)?;
525
526            scratch.set_scratch_vector_size(full_matrix.columns());
527            for (i, _) in preimage.iter_nonzero() {
528                scratch.as_slice_mut().add(full_matrix.row(i), 1);
529            }
530            scratch.to_bytes(f)?;
531        }
532
533        tracing::Span::current().record(
534            "throughput",
535            tracing::field::display(own_f.into_throughput()),
536        );
537        Ok(())
538    }

560    #[tracing::instrument(skip(self), fields(%b, %subalgebra, num_new_gens, density))]
561    fn step_resolution_with_subalgebra(
562        &self,
563        b: Bidegree,
564        subalgebra: MilnorSubalgebra,
565    ) -> anyhow::Result<()> {
566        let end = || {
567            tracing::Span::current().record("num_new_gens", self.number_of_gens_in_bidegree(b));
568            tracing::Span::current().record(
569                "density",
570                self.differentials[b.s()].differential_density(b.t()) * 100.0,
571            );
572        };
573
574        let p = self.prime();
575        let mut scratch = FpVector::new(p, 0);
576
577        let target = &*self.modules[b.s() - 1];
578        let algebra = target.algebra();
579
580        let zero_sig = subalgebra.zero_signature();
581        let target_dim = target.dimension(b.t());
582        let target_mask: Vec<usize> = subalgebra
583            .signature_mask(&algebra, target, b.t(), &zero_sig)
584            .collect();
585        let target_masked_dim = target_mask.len();
586
587        let next = &self.modules[b.s() - 2];
588        next.compute_basis(b.t());
589
590        let mut f = if let Some(dir) = self.save_dir().write() {
591            let mut f = self
592                .save_file(SaveKind::NassauQi, b - Bidegree::s_t(1, 0))
593                .create_file(dir.to_owned(), true);
594            f.write_u64::<LittleEndian>(next.dimension(b.t()) as u64)?;
595            f.write_u64::<LittleEndian>(target_masked_dim as u64)?;
596            subalgebra.to_bytes(&mut f)?;
597            Some(f)
598        } else {
599            None
600        };
601
602        let next_mask: Vec<usize> = subalgebra
603            .signature_mask(&algebra, &self.modules[b.s() - 2], b.t(), &zero_sig)
604            .collect();
605        let next_masked_dim = next_mask.len();
606
607        let full_matrix = self.differentials[b.s() - 1].get_partial_matrix(b.t(), &target_mask);
608        let mut masked_matrix =
609            AugmentedMatrix::new(p, target_masked_dim, [next_masked_dim, target_masked_dim]);
610
611        masked_matrix
612            .segment(0, 0)
613            .add_masked(&full_matrix, &next_mask);
614        masked_matrix.segment(1, 1).add_identity();
615        masked_matrix.row_reduce();
616        let kernel = masked_matrix.compute_kernel();
617
618        Self::write_qi(
619            &mut f,
620            &mut scratch,
621            &zero_sig,
622            &next_mask,
623            &full_matrix,
624            &masked_matrix,
625        )?;
626
627        if let Some(f) = &mut f {
628            if target.max_computed_degree() < b.t() {
629                f.write_u64::<LittleEndian>(Magic::Fix as u64)?;
630            }
631        }
632
633        // Compute image
634        let mut n = subalgebra.signature_matrix(&self.differentials[b.s()], b.t(), &zero_sig);
635        n.row_reduce();
636        let next_row = n.rows();
637
638        let num_new_gens = n.extend_image(0, n.columns(), &kernel, 0).len();
639
640        if b.t() < b.s() {
641            assert_eq!(num_new_gens, 0, "Adding generators at {b}");
642        }
643
644        self.add_generators(b, num_new_gens);
645
646        let mut xs = vec![FpVector::new(p, target_dim); num_new_gens];
647        let mut dxs = vec![FpVector::new(p, next.dimension(b.t())); num_new_gens];
648
649        for ((x, x_masked), dx) in xs.iter_mut().zip_eq(&n[next_row..]).zip_eq(&mut dxs) {
650            x.as_slice_mut()
651                .add_unmasked(x_masked.as_slice(), 1, &target_mask);
652            for (i, _) in x_masked.iter_nonzero() {
653                dx.add(&full_matrix[i], 1);
654            }
655        }
656
657        // Now add correction terms
658        let mut target_mask: Vec<usize> = Vec::new();
659        let mut next_mask: Vec<usize> = Vec::new();
660
661        for signature in subalgebra.iter_signatures(b.t()) {
662            target_mask.clear();
663            next_mask.clear();
664            target_mask.extend(subalgebra.signature_mask(&algebra, target, b.t(), &signature));
665            next_mask.extend(subalgebra.signature_mask(&algebra, next, b.t(), &signature));
666
667            let full_matrix = self
668                .differential(b.s() - 1)
669                .get_partial_matrix(b.t(), &target_mask);
670
671            let mut masked_matrix =
672                AugmentedMatrix::new(p, target_mask.len(), [next_mask.len(), target_mask.len()]);
673            masked_matrix
674                .segment(0, 0)
675                .add_masked(&full_matrix, &next_mask);
676            masked_matrix.segment(1, 1).add_identity();
677            masked_matrix.row_reduce();
678
679            let qi = masked_matrix.compute_quasi_inverse();
680            let pivots = qi.pivots().unwrap();
681            let preimage = qi.preimage();
682
683            for (x, dx) in xs.iter_mut().zip(&mut dxs) {
684                scratch.set_scratch_vector_size(target_mask.len());
685                let mut row = 0;
686                for (i, &v) in next_mask.iter().enumerate() {
687                    if pivots[i] < 0 {
688                        continue;
689                    }
690                    if dx.entry(v) != 0 {
691                        scratch.add(&preimage[row], 1);
692                    }
693                    row += 1;
694                }
695                for (i, _) in scratch.iter_nonzero() {
696                    x.add_basis_element(target_mask[i], 1);
697                    dx.add(&full_matrix[i], 1);
698                }
699            }
700            Self::write_qi(
701                &mut f,
702                &mut scratch,
703                &signature,
704                &next_mask,
705                &full_matrix,
706                &masked_matrix,
707            )?;
708        }
709        for dx in &dxs {
710            assert!(dx.is_zero(), "dx non-zero at {b}");
711        }
712        self.differential(b.s()).add_generators_from_rows(b.t(), xs);
713
714        end();
715
716        if let Some(f) = &mut f {
717            f.write_u64::<LittleEndian>(Magic::End as u64)?;
718        }
719
720        self.write_differential(b, num_new_gens, target_dim)?;
721        Ok(())
722    }
723
724    /// Step resolution for s = 0
725    #[tracing::instrument(skip(self))]
726    fn step0(&self, t: i32) {
727        self.zero_module.extend_by_zero(t);
728
729        let source_module = &self.modules[0];
730        let target_module = self.target.module(0);
731
732        let chain_map = &self.chain_maps[0];
733        let d = &self.differentials[0];
734
735        let source_dim = source_module.dimension(t);
736        let target_dim = target_module.dimension(t);
737
738        source_module.compute_basis(t);
739        target_module.compute_basis(t);
740
741        if target_dim == 0 {
742            source_module.extend_by_zero(t);
743            chain_map.extend_by_zero(t);
744        } else {
745            let mut matrix = AugmentedMatrix::<2>::new_with_capacity(
746                self.prime(),
747                source_dim,
748                &[target_dim, source_dim],
749                source_dim + target_dim,
750                0,
751            );
752            chain_map.get_matrix(matrix.segment(0, 0), t);
753            matrix.segment(1, 1).add_identity();
754
755            matrix.row_reduce();
756
757            let num_new_gens = matrix.extend_to_surjection(0, target_dim, 0).len();
758
759            self.add_generators(Bidegree::s_t(0, t), num_new_gens);
760
761            chain_map.add_generators_from_matrix_rows(
762                t,
763                matrix
764                    .segment(0, 0)
765                    .row_slice(source_dim, source_dim + num_new_gens),
766            );
767        }
768        chain_map.compute_auxiliary_data_through_degree(t);
769
770        d.set_kernel(t, None);
771        d.set_image(t, None);
772        d.set_quasi_inverse(t, None);
773        d.extend_by_zero(t);
774    }
775
776    /// Step resolution for s = 1
777    #[tracing::instrument(skip(self))]
778    fn step1(&self, t: i32) -> anyhow::Result<()> {
779        let p = self.prime();
780
781        let source_module = &self.modules[1];
782        let target_module = &self.modules[0];
783        let cc_module = self.target.module(0);
784
785        let source_dim = source_module.dimension(t);
786        let target_dim = target_module.dimension(t);
787
788        let mut matrix =
789            AugmentedMatrix::<2>::new(p, target_dim, [cc_module.dimension(t), target_dim]);
790        self.chain_maps[0].get_matrix(matrix.segment(0, 0), t);
791        matrix.segment(1, 1).add_identity();
792        matrix.row_reduce();
793        let desired_image = matrix.compute_kernel();
794
795        let mut matrix = AugmentedMatrix::<2>::new_with_capacity(
796            p,
797            source_dim,
798            &[target_dim, source_dim],
799            source_dim + MAX_NEW_GENS,
800            0,
801        );
802        self.differentials[1].get_matrix(matrix.segment(0, 0), t);
803        matrix.segment(1, 1).add_identity();
804        matrix.row_reduce();
805
806        let num_new_gens = matrix.extend_image(0, target_dim, &desired_image, 0).len();
807
808        self.add_generators(Bidegree::s_t(1, t), num_new_gens);
809
810        self.differentials[1].add_generators_from_matrix_rows(
811            t,
812            matrix
813                .segment(0, 0)
814                .row_slice(source_dim, source_dim + num_new_gens),
815        );
816
817        self.write_differential(Bidegree::s_t(1, t), num_new_gens, target_dim)?;
818        Ok(())
819    }
820
821    fn step_resolution_with_result(&self, b: Bidegree) -> anyhow::Result<()> {
822        let p = self.prime();
823        let set_data = || {
824            let d = &self.differentials[b.s()];
825            let c = &self.chain_maps[b.s()];
826
827            d.set_kernel(b.t(), None);
828            d.set_image(b.t(), None);
829            d.set_quasi_inverse(b.t(), None);
830
831            c.set_kernel(b.t(), None);
832            c.set_image(b.t(), None);
833            c.set_quasi_inverse(b.t(), None);
834        };
835        self.modules[b.s()].compute_basis(b.t());
836        if b.s() > 0 {
837            self.modules[b.s() - 1].compute_basis(b.t());
838        }
839
840        if b.s() == 0 {
841            self.step0(b.t());
842            return Ok(());
843        }
844
845        if let Some(dir) = self.save_dir.read() {
846            if let Some(mut f) = self
847                .save_file(SaveKind::NassauDifferential, b)
848                .open_file(dir.clone())
849            {
850                tracing::info!(%b, "Loading differential");
851
852                let num_new_gens = f.read_u64::<LittleEndian>()? as usize;
853                // This need not be equal to `target_res_dimension`. If we saved a big resolution
854                // and now only want to load up to a small stem, then `target_res_dimension` will
855                // be smaller. If we have previously saved a small resolution up to a stem and now
856                // want to resolve further, it will be bigger.
857                let saved_target_res_dimension = f.read_u64::<LittleEndian>()? as usize;
858
859                self.add_generators(b, num_new_gens);
860
861                let mut d_targets = Vec::with_capacity(num_new_gens);
862
863                for _ in 0..num_new_gens {
864                    d_targets.push(FpVector::from_bytes(p, saved_target_res_dimension, &mut f)?);
865                }
866
867                self.differentials[b.s()].add_generators_from_rows(b.t(), d_targets);
868
869                set_data();
870
871                return Ok(());
872            }
873        }
874
875        if b.s() == 1 {
876            self.step1(b.t())?;
877            set_data();
878            return Ok(());
879        }
880
881        self.step_resolution_with_subalgebra(
882            b,
883            MilnorSubalgebra::optimal_for(b - Bidegree::s_t(0, self.max_degree)),
884        )?;
885        self.chain_maps[b.s()].extend_by_zero(b.t());
886
887        set_data();
888        Ok(())
889    }
890
891    fn step_resolution(&self, b: Bidegree) {
892        self.step_resolution_with_result(b)
893            .unwrap_or_else(|e| panic!("Error computing bidegree {b}: {e}"));
894    }

896    /// This function resolves up till a fixed stem instead of a fixed t.
897    #[tracing::instrument(skip(self), fields(self = self.name, %max))]
898    pub fn compute_through_stem(&self, max: Bidegree) {
899        let _lock = self.lock.lock();
900
901        self.extend_through_degree(max.s());
902        self.algebra().compute_basis(max.t());
903
904        let tracing_span = tracing::Span::current();
905        maybe_rayon::in_place_scope(|scope| {
906            let _tracing_guard = tracing_span.enter();
907
908            // This algorithm is not optimal, as we compute (s, t) only after computing (s - 1, t)
909            // and (s, t - 1). In theory, it suffices to wait for (s, t - 1) and (s - 1, t - 1),
910            // but having the dimensions of the modules change halfway through the computation is
911            // annoying to do correctly. It seems more prudent to improve parallelism elsewhere.
912
913            // Things that we have finished computing.
914            let mut progress: Vec<i32> = vec![-1; max.s() as usize + 1];
915            // We will kickstart the process by pretending we have computed (0, - 1). So
916            // we must pretend we have only computed up to (0, - 2);
917            progress[0] = -2;
918
919            let (sender, receiver) = mpsc::channel();
920            SenderData::send(Bidegree::s_t(0, -1), sender);
921
922            let f = |b, sender| {
923                if self.has_computed_bidegree(b) {
924                    SenderData::send(b, sender);
925                } else {
926                    let tracing_span = tracing_span.clone();
927                    scope.spawn(move |_| {
928                        let _tracing_guard = tracing_span.enter();
929                        self.step_resolution(b);
930                        SenderData::send(b, sender);
931                    });
932                }
933            };
934
935            while let Ok(SenderData { b, sender }) = receiver.recv() {
936                assert!(progress[b.s() as usize] == b.t() - 1);
937                progress[b.s() as usize] = b.t();
938
939                // How far we are from the last one for this s.
940                let distance = max.n() - b.n() + 1;
941
942                if b.s() < max.s() && progress[b.s() as usize + 1] == b.t() - 1 {
943                    f(b + Bidegree::s_t(1, 0), sender.clone());
944                }
945
946                if distance > 1 && (b.s() == 0 || progress[b.s() as usize - 1] > b.t()) {
947                    // We are computing a normal step
948                    f(b + Bidegree::s_t(0, 1), sender);
949                } else if distance == 1 && b.s() < max.s() {
950                    SenderData::send(b + Bidegree::s_t(0, 1), sender);
951                }
952            }
953        });
954    }

957impl<M: ZeroModule<Algebra = MilnorAlgebra>> ChainComplex for Resolution<M> {
958    type Algebra = MilnorAlgebra;
959    type Homomorphism = FreeModuleHomomorphism<FreeModule<Self::Algebra>>;
960    type Module = FreeModule<Self::Algebra>;
961
962    fn prime(&self) -> ValidPrime {
963        TWO
964    }
965
966    fn algebra(&self) -> Arc<Self::Algebra> {
967        self.zero_module.algebra()
968    }
969
970    fn module(&self, s: i32) -> Arc<Self::Module> {
971        Arc::clone(&self.modules[s])
972    }
973
974    fn zero_module(&self) -> Arc<Self::Module> {
975        Arc::clone(&self.zero_module)
976    }
977
978    fn min_degree(&self) -> i32 {
979        0
980    }
981
982    fn has_computed_bidegree(&self, b: Bidegree) -> bool {
983        self.differentials.len() > b.s() && self.differential(b.s()).next_degree() > b.t()
984    }
985
986    fn differential(&self, s: i32) -> Arc<Self::Homomorphism> {
987        Arc::clone(&self.differentials[s])
988    }
989
990    #[tracing::instrument(skip(self), fields(self = self.name, %max))]
991    fn compute_through_bidegree(&self, max: Bidegree) {
992        let _lock = self.lock.lock();
993
994        self.extend_through_degree(max.s());
995        self.algebra().compute_basis(max.t());
996
997        for t in 0..=max.t() {
998            for s in 0..=max.s() {
999                let b = Bidegree::s_t(s, t);
1000                if self.has_computed_bidegree(b) {
1001                    continue;
1002                }
1003                self.step_resolution(b);
1004            }
1005        }
1006    }
1007
1008    fn next_homological_degree(&self) -> i32 {
1009        self.modules.len()
1010    }
1011
1012    fn save_dir(&self) -> &SaveDirectory {
1013        &self.save_dir
1014    }
1015
1016    fn apply_quasi_inverse<T, S>(&self, results: &mut [T], b: Bidegree, inputs: &[S]) -> bool
1017    where
1018        for<'a> &'a mut T: Into<FpSliceMut<'a>>,
1019        for<'a> &'a S: Into<FpSlice<'a>>,
1020    {
1021        let mut f = if let Some(dir) = self.save_dir.read() {
1022            if let Some(f) = self.save_file(SaveKind::NassauQi, b).open_file(dir.clone()) {
1023                f
1024            } else {
1025                return false;
1026            }
1027        } else {
1028            return false;
1029        };
1030
1031        let p = self.prime();
1032
1033        let target_dim = f.read_u64::<LittleEndian>().unwrap() as usize;
1034        let zero_mask_dim = f.read_u64::<LittleEndian>().unwrap() as usize;
1035        let subalgebra = MilnorSubalgebra::from_bytes(&mut f).unwrap();
1036        let source = &self.modules[b.s()];
1037        let target = &self.modules[b.s() - 1];
1038        let algebra = target.algebra();
1039
1040        let mut inputs: Vec<FpVector> = inputs.iter().map(|x| x.into().to_owned()).collect();
1041        let mut mask: Vec<usize> = Vec::with_capacity(zero_mask_dim + 8);
1042        mask.extend(subalgebra.signature_mask(
1043            &algebra,
1044            source,
1045            b.t(),
1046            &subalgebra.zero_signature(),
1047        ));
1048
1049        let mut scratch0 = FpVector::new(p, zero_mask_dim);
1050        let mut scratch1 = FpVector::new(p, target_dim);
1051
1052        // If the quasi-inverse was computed using incomplete information, we need to figure out
1053        // what the differentials in this bidegree hit and use them to lift. these variables are
1054        // trivial if there is no such problem.
1055        //
1056        // target_zero_mask is the signature mask of the target under the zero signature.
1057        //
1058        // dx_matrix is an AugmentedMatrix::<3>.
1059        //
1060        // Each row of this matrix is of the form [r; dx; x], where x is an element of the source
1061        // of signature zero, expressed in the masked basis, and dx is the value of the
1062        // differential on x. Then r is the entries of dx that have zero signature, which we
1063        // include so that the rref of the matix is nice. In practice, we keep r empty until the
1064        // very end, and then populate it manually.
1065        //
1066        // At the beginning the x's will be the new generators in this bidegree. As we read in the
1067        // quasi-inverses for the zero signature, we keep on reducing this so that dx is zero in
1068        // the pivot columns of the quasi-inverse. We can then use (the rref of) this matrix to
1069        // lift remaining elements with zero signature.
1070        let (mut target_zero_mask, mut dx_matrix) = if zero_mask_dim != mask.len() {
1071            let num_new_gens = source.number_of_gens_in_degree(b.t());
1072            assert_eq!(mask.len(), zero_mask_dim + num_new_gens);
1073
1074            let target_zero_mask: Vec<usize> = subalgebra
1075                .signature_mask(&algebra, target, b.t(), &subalgebra.zero_signature())
1076                .collect();
1077            let mut matrix = AugmentedMatrix::<3>::new(
1078                p,
1079                num_new_gens,
1080                [target_zero_mask.len(), target.dimension(b.t()), mask.len()],
1081            );
1082
1083            for i in 0..num_new_gens {
1084                let dx = self.differentials[b.s()].output(b.t(), i);
1085                matrix
1086                    .row_segment_mut(i, 1, 1)
1087                    .slice_mut(0, dx.len())
1088                    .add(dx.as_slice(), 1);
1089                matrix
1090                    .row_segment_mut(i, 2, 2)
1091                    .add_basis_element(zero_mask_dim + i, 1);
1092            }
1093
1094            (target_zero_mask, matrix)
1095        } else {
1096            (Vec::new(), AugmentedMatrix::<3>::new(p, 0, [0, 0, 0]))
1097        };
1098
1099        loop {
1100            let col = f.read_u64::<LittleEndian>().unwrap() as usize;
1101            if col == Magic::End as usize {
1102                break;
1103            } else if col == Magic::Signature as usize {
1104                let signature = subalgebra.signature_from_bytes(&mut f).unwrap();
1105
1106                mask.clear();
1107                mask.extend(subalgebra.signature_mask(&algebra, source, b.t(), &signature));
1108                scratch0.set_scratch_vector_size(mask.len());
1109            } else if col == Magic::Fix as usize {
1110                // We need to fix the differential problem
1111                //
1112                // First manually add_masked the second segment to the first, which we use for
1113                // row reduction. We do this manually for borrow checker reasons.
1114                for (j, &k) in target_zero_mask.iter().enumerate() {
1115                    for i in 0..dx_matrix.rows() {
1116                        if dx_matrix.row_segment(i, 1, 1).entry(k) != 0 {
1117                            dx_matrix.row_segment_mut(i, 0, 0).add_basis_element(j, 1);
1118                        }
1119                    }
1120                }
1121                dx_matrix.row_reduce();
1122
1123                // Now reduce by these elements
1124                for i in 0..dx_matrix.rows() {
1125                    let masked_col = dx_matrix[i].first_nonzero().unwrap().0;
1126                    assert_eq!(dx_matrix.pivots()[masked_col], i as isize);
1127                    let col = target_zero_mask[masked_col];
1128
1129                    for (input, output) in inputs.iter_mut().zip(results.iter_mut()) {
1130                        let entry = input.entry(col);
1131                        if entry != 0 {
1132                            output
1133                                .into()
1134                                .add_unmasked(dx_matrix.row_segment(i, 2, 2), 1, &mask);
1135                            input.as_slice_mut().add(dx_matrix.row_segment(i, 1, 1), 1);
1136                        }
1137                    }
1138                }
1139
1140                // Drop these objects to save a bit of memory
1141                target_zero_mask = Vec::new();
1142                dx_matrix = AugmentedMatrix::<3>::new(p, 0, [0, 0, 0]);
1143            } else {
1144                scratch0.update_from_bytes(&mut f).unwrap();
1145                scratch1.update_from_bytes(&mut f).unwrap();
1146                for (input, output) in inputs.iter_mut().zip(results.iter_mut()) {
1147                    let entry = input.entry(col);
1148                    if entry != 0 {
1149                        output.into().add_unmasked(scratch0.as_slice(), 1, &mask);
1150                        // If we resume a resolve_through_stem, input may be longer than scratch1.
1151                        input
1152                            .slice_mut(0, scratch1.len())
1153                            .add(scratch1.as_slice(), 1);
1154                    }
1155                }
1156
1157                // Row reduce the differentials
1158                if !target_zero_mask.is_empty() {
1159                    for i in 0..dx_matrix.rows() {
1160                        if dx_matrix.row_segment(i, 1, 1).entry(col) != 0 {
1161                            dx_matrix
1162                                .row_segment_mut(i, 2, 2)
1163                                .slice_mut(0, zero_mask_dim)
1164                                .add(scratch0.as_slice(), 1);
1165                            dx_matrix
1166                                .row_segment_mut(i, 1, 1)
1167                                .slice_mut(0, target_dim)
1168                                .add(scratch1.as_slice(), 1);
1169                        }
1170                    }
1171                }
1172            }
1173        }
1174        // Make sure we have finished reading everything
1175        drop(f);
1176
1177        for dx in inputs {
1178            assert!(
1179                dx.is_zero(),
1180                "remainder non-zero at {b}\nAlgebra: {subalgebra}\ndx: {}",
1181                target.element_to_string(b.t(), dx.as_slice())
1182            );
1183        }
1184        true
1185    }
1186}
