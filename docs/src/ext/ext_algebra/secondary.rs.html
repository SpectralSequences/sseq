1//! The secondary ($d_2$) layer of [`ExtAlgebra`].
2//!
3//! [`SecondaryExtAlgebra`] composes an [`ExtAlgebra`] with the secondary resolutions of `M` and
4//! the unit `k`, and exposes:
5//! - the secondary differential [`d2`](SecondaryExtAlgebra::d2) (and the survival check
6//!   [`survives`](SecondaryExtAlgebra::survives)),
7//! - the $E_3$-page data [`page_data`](SecondaryExtAlgebra::page_data), and
8//! - the $\Mod_{C\lambda^2}$ secondary product
9//!   [`secondary_multiply_into`](SecondaryExtAlgebra::secondary_multiply_into).
10//!
11//! These wrap [`SecondaryResolution`] and [`SecondaryResolutionHomomorphism`]; no new linear
12//! algebra is implemented here. The layer is split out from [`ExtAlgebra`] because the secondary
13//! machinery requires `CC::Algebra: PairAlgebra`, a bound the primary layer does not impose.
14

31/// A single secondary product `x · y` in $\Mod_{C\lambda^2}$, where `y` is an $E_3$-surviving
32/// class. See [`SecondaryExtAlgebra::secondary_multiply_into`].
33pub struct SecondaryProduct {
34    /// The multiplicand: an $E_3$-surviving generator of the unit at the queried bidegree `b`.
35    pub source: BidegreeElement,
36    /// The $\Ext$ part of the product, in bidegree `b + x.degree()`.
37    pub ext_part: FpVector,
38    /// The $\lambda$ part of the product, in bidegree `b + x.degree() + LAMBDA_BIDEGREE`, already
39    /// reduced by the image of $d_2$.
40    pub lambda_part: FpVector,
41}
42
43/// The secondary layer over an [`ExtAlgebra`]: the $d_2$ differential and the $\Mod_{C\lambda^2}$
44/// product. See the [module documentation](self).
45pub struct SecondaryExtAlgebra<CC: FreeChainComplex>
46where
47    CC::Algebra: PairAlgebra,
48{
49    alg: Arc<ExtAlgebra<CC>>,
50    res_lift: Arc<SecondaryResolution<CC>>,
51    /// `Arc`-shared with `res_lift` when `M == k`.
52    unit_lift: Arc<SecondaryResolution<CC>>,
53    /// $E_3$ page of the resolution, filled by [`extend_all`](Self::extend_all).
54    res_sseq: Mutex<Option<Arc<sseq::Sseq<2, sseq::Adams>>>>,
55    /// $E_3$ page of the unit, filled by [`extend_all`](Self::extend_all).
56    unit_sseq: Mutex<Option<Arc<sseq::Sseq<2, sseq::Adams>>>>,
57    /// Secondary lift of the multiplication map, cached per multiplier class `(degree, coords)`.
58    secondary_products: DashMap<BidegreeElement, Arc<SecondaryResolutionHomomorphism<CC, CC>>>,
59}
60
61impl<CC: FreeChainComplex> SecondaryExtAlgebra<CC>
62where
63    CC::Algebra: PairAlgebra,
64{
65    /// Build the secondary layer over `alg`. Construction is cheap; call [`extend_all`](Self::extend_all)
66    /// to actually compute the secondary resolutions and $E_3$ pages.
67    pub fn new(alg: Arc<ExtAlgebra<CC>>) -> Self {
68        let res_lift = Arc::new(SecondaryResolution::new(Arc::clone(alg.resolution())));
69        let unit_lift = if alg.is_unit() {
70            Arc::clone(&res_lift)
71        } else {
72            Arc::new(SecondaryResolution::new(Arc::clone(alg.unit())))
73        };
74        Self {
75            alg,
76            res_lift,
77            unit_lift,
78            res_sseq: Mutex::new(None),
79            unit_sseq: Mutex::new(None),
80            secondary_products: DashMap::new(),
81        }
82    }
83
84    /// Extend the secondary resolutions as far as the underlying resolutions allow, then compute
85    /// the $E_3$ pages. Must be called before [`d2`](Self::d2), [`page_data`](Self::page_data) or
86    /// [`secondary_multiply_into`](Self::secondary_multiply_into).
87    pub fn extend_all(&self) {
88        self.res_lift.extend_all();
89        if !self.alg.is_unit() {
90            self.unit_lift.extend_all();
91        }
92
93        *self.res_sseq.lock().unwrap() = Some(Arc::new(self.res_lift.e3_page()));
94        let unit = if self.alg.is_unit() {
95            Arc::clone(self.res_sseq.lock().unwrap().as_ref().unwrap())
96        } else {
97            Arc::new(self.unit_lift.e3_page())
98        };
99        *self.unit_sseq.lock().unwrap() = Some(unit);
100    }
101
102    /// Sharding entry point: compute only the secondary resolution data for filtration `s`,
103    /// distributed across machines sharing a save directory (see the `secondary` example docs).
104    /// Mirrors [`SecondaryLift::compute_partial`]. Returns before any $E_3$ page is built.
105    pub fn compute_partial(&self, s: i32) {
106        self.res_lift.compute_partial(s);
107        if !self.alg.is_unit() {
108            self.unit_lift.compute_partial(s);
109        }
110    }
111
112    /// The primary [`ExtAlgebra`] this is built on.
113    pub fn ext_algebra(&self) -> &Arc<ExtAlgebra<CC>> {
114        &self.alg
115    }
116
117    fn prime(&self) -> fp::prime::ValidPrime {
118        self.alg.prime()
119    }
120
121    /// The secondary differential $d_2(x)$, a class in bidegree `(n - 1, s + 2)`.
122    ///
123    /// Returns `None` if the target bidegree has not been computed (so $d_2$ is unknown). A
124    /// computed-but-zero differential is `Some` of a zero class.
125    pub fn d2(&self, x: &BidegreeElement) -> Option<BidegreeElement> {
126        let b = x.degree();
127        let target = b + Bidegree::n_s(-1, 2);
128        let res = self.res_lift.underlying();
129        if !(b.t() > 0 && res.has_computed_bidegree(target)) {
130            return None;
131        }
132
133        let target_dim = res.number_of_gens_in_bidegree(target);
134        let mut out = FpVector::new(self.prime(), target_dim);
135
136        // `m[i]` is the d2 of the i-th generator of `b`, as a vector at `target`. This is exactly
137        // the matrix `SecondaryResolution::e3_page` reads to install d2 differentials.
138        let m = self.res_lift.homotopy(b.s() + 2).homotopies.hom_k(b.t());
139        if !m.is_empty() && !m[0].is_empty() {
140            let p = self.prime().as_u32();
141            for (i, c) in x.vec().iter_nonzero() {
142                for (k, &v) in m[i].iter().enumerate() {
143                    out.add_basis_element(k, (c * v) % p);
144                }
145            }
146        }
147        Some(BidegreeElement::new(target, out))
148    }
149
150    /// Whether `x` is a $d_2$-cycle (a permanent class through $E_3$).
151    pub fn survives(&self, x: &BidegreeElement) -> Option<bool> {
152        self.d2(x).map(|d| d.vec().is_zero())
153    }
154
155    /// The $E_3$-page subquotient of $\Ext(M, k)$ at bidegree `b`.
156    pub fn page_data(&self, b: Bidegree) -> Subquotient {
157        let g = self.res_sseq.lock().unwrap();
158        Self::e3_page_data(g.as_ref().expect("call extend_all() first"), b).clone()
159    }
160
161    /// The $E_3$-page subquotient of the unit $\Ext(k, k)$ at bidegree `b`.
162    pub fn unit_page_data(&self, b: Bidegree) -> Subquotient {
163        let g = self.unit_sseq.lock().unwrap();
164        Self::e3_page_data(g.as_ref().expect("call extend_all() first"), b).clone()
165    }
166
167    fn e3_page_data(sseq: &sseq::Sseq<2, sseq::Adams>, b: Bidegree) -> &Subquotient {
168        let d = sseq.page_data(b);
169        &d[std::cmp::min(3, d.len() - 1)]
170    }
171}

173impl<CC: FreeChainComplex + crate::chain_complex::AugmentedChainComplex> SecondaryExtAlgebra<CC>
174where
175    CC::Algebra: PairAlgebra,
176{
177    /// The secondary lift of multiplication by `x`, built and cached per multiplier class. The
178    /// returned lift is *not* extended; [`secondary_multiply_into`](Self::secondary_multiply_into)
179    /// extends it as needed. Exposed so callers can drive sharded computation
180    /// (`lift.underlying().extend_all()` then `lift.compute_partial(s)`).
181    pub fn secondary_product_lift(
182        &self,
183        x: &BidegreeElement,
184    ) -> Arc<SecondaryResolutionHomomorphism<CC, CC>> {
185        if let Some(map) = self.secondary_products.get(x) {
186            return Arc::clone(&map);
187        }
188
189        let name = format!("prod_{x}",);
190        let underlying = Arc::new(ResolutionHomomorphism::from_class(
191            name,
192            Arc::clone(self.alg.resolution()),
193            Arc::clone(self.alg.unit()),
194            x.degree(),
195            &x.vec().iter().collect::<Vec<_>>(),
196        ));
197        let lift = Arc::new(SecondaryResolutionHomomorphism::new(
198            Arc::clone(&self.res_lift),
199            Arc::clone(&self.unit_lift),
200            underlying,
201        ));
202
203        Arc::clone(
204            self.secondary_products
205                .entry(x.clone())
206                .or_insert(lift)
207                .value(),
208        )
209    }
210
211    /// The secondary product of `x` with every $E_3$-surviving class of the unit at bidegree `b`,
212    /// computed in $\Mod_{C\lambda^2}$.
213    ///
214    /// Returns one [`SecondaryProduct`] per surviving generator at `b`; the $\lambda$ part is
215    /// already reduced by the image of $d_2$. The caller must have run [`extend_all`](Self::extend_all)
216    /// and computed both resolutions far enough.
217    pub fn secondary_multiply_into(
218        &self,
219        x: &BidegreeElement,
220        b: Bidegree,
221    ) -> Vec<SecondaryProduct> {
222        let p = self.prime();
223        let shift = x.degree();
224        let res_sseq = Arc::clone(
225            self.res_sseq
226                .lock()
227                .unwrap()
228                .as_ref()
229                .expect("call extend_all() first"),
230        );
231
232        let ext_dim = self.alg.resolution().number_of_gens_in_bidegree(b + shift);
233        let lambda_dim = self
234            .alg
235            .resolution()
236            .number_of_gens_in_bidegree(b + shift + LAMBDA_BIDEGREE);
237
238        let page = self.unit_page_data(b);
239        let n = page.subspace_dimension();
240        if n == 0 {
241            return Vec::new();
242        }
243
244        let lift = self.secondary_product_lift(x);
245        lift.underlying().extend_all();
246        lift.extend_all();
247
248        let mut outputs = vec![FpVector::new(p, ext_dim + lambda_dim); n];
249        lift.hom_k(
250            Some(&res_sseq),
251            b,
252            page.subspace_gens(),
253            outputs.iter_mut().map(FpVector::as_slice_mut),
254        );
255
256        page.subspace_gens()
257            .zip(outputs)
258            .map(|(g, out)| SecondaryProduct {
259                source: BidegreeElement::new(b, g.to_owned()),
260                ext_part: out.slice(0, ext_dim).to_owned(),
261                lambda_part: out.slice(ext_dim, ext_dim + lambda_dim).to_owned(),
262            })
263            .collect()
264    }
265}

267#[cfg(test)]
268mod tests {
269    use sseq::coordinates::BidegreeGenerator;
270
271    use super::*;
272    use crate::utils::construct_standard;
273
274    #[test]
275    fn test_sphere_d2() {
276        let res = Arc::new(construct_standard::<false, _, _>("S_2", None).unwrap());
277        // Far enough to reach the first Adams differential d2(h4) = h0 h3^2 at (14, 3).
278        res.compute_through_stem(Bidegree::n_s(16, 6));
279        let e2 = Arc::new(ExtAlgebra::new(Arc::clone(&res), res));
280        let sec_e2 = SecondaryExtAlgebra::new(Arc::clone(&e2));
281        sec_e2.extend_all();
282
283        // h_0, h_1, h_2 are permanent cycles.
284        for (n, s) in [(0, 1), (1, 1), (3, 1)] {
285            let h = e2.generator(BidegreeGenerator::new(Bidegree::n_s(n, s), 0));
286            let h_survives = sec_e2
287                .survives(&h)
288                .unwrap_or_else(|| panic!("h at (n={n}, s={s}) should have a computed d2"));
289            assert!(h_survives, "h at (n={n}, s={s}) should survive d2");
290            let h_d2 = sec_e2
291                .d2(&h)
292                .unwrap_or_else(|| panic!("h at (n={n}, s={s}) should have a computed d2"));
293            assert!(
294                h_d2.vec().is_zero(),
295                "d2 of a permanent class should vanish"
296            );
297        }
298
299        // The first Adams differential: d2(h4) = h0 h3^2, the generator of Ext^{3,17} at (14, 3).
300        let h4 = e2.generator(BidegreeGenerator::new(Bidegree::n_s(15, 1), 0));
301        let d = sec_e2.d2(&h4).expect("d2(h4) target should be computed");
302        assert_eq!(d.degree(), Bidegree::n_s(14, 3));
303        assert_eq!(e2.dimension(Bidegree::n_s(14, 3)), 1);
304        assert!(!d.vec().is_zero(), "d2(h4) = h0 h3^2 should be nonzero");
305        let h4_survives = sec_e2.survives(&h4).expect("h4 should have a computed d2");
306        assert!(!h4_survives, "h4 should not survive d2");
307    }
