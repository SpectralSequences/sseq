1//! A bigraded-algebra view of a resolution.
2//!
3//! [`ExtAlgebra`] wraps a resolution of a module `M` together with the resolution of the base
4//! field `k` (the "unit"), and presents $\Ext(M, k)$ as a bigraded module over the bigraded
5//! algebra $\Ext(k, k)$. When `M == k` this is the algebra $\Ext(k, k)$ itself.
6//!
7//! The goal is ergonomics: computing a product of Ext classes is a single [`ExtAlgebra::multiply`]
8//! call instead of the manual [`ResolutionHomomorphism`] + `extend` + `hom_k` plumbing that the
9//! examples currently re-derive. This is the foundational layer; the secondary differential ($d_2$)
10//! and Massey products are planned follow-ups.
11//!
12//! # Conventions
13//! A product is realised by a [`ResolutionHomomorphism`] built from a fixed multiplier class living
14//! in $\Ext(M, k)$ (source = resolution of `M`, target = resolution of `k`). That single chain map
15//! computes the products of the multiplier with *all* classes of $\Ext(k, k)$. We cache one such
16//! map per *generator* of $\Ext(M, k)$ (keyed by [`BidegreeGenerator`]); a product by a general
17//! class is assembled at request time as the corresponding linear combination of generator maps.
18//!
19//! The secondary differential ($d_2$) and the $\Mod_{C\lambda^2}$ secondary product live in the
20//! [`secondary`] submodule ([`SecondaryExtAlgebra`]).
21

23pub mod secondary;

38/// $\Ext(M, k)$ as a bigraded module over the bigraded algebra $\Ext(k, k)$, backed by a
39/// resolution. See the [module-level documentation](self) for conventions.
40pub struct ExtAlgebra<CC: FreeChainComplex> {
41    /// Resolution of `M`; products land in its Ext.
42    resolution: Arc<CC>,
43    /// Resolution of the base field `k`. `Arc`-shared with `resolution` when `M == k`.
44    unit: Arc<CC>,
45    is_unit: bool,
46    /// One multiplication map per generator of $\Ext(M, k)$, built and extended on demand.
47    products: DashMap<BidegreeGenerator, Arc<ResolutionHomomorphism<CC, CC>>>,
48}

50impl ExtAlgebra<QueryModuleResolution> {
51    /// Build an [`ExtAlgebra`] from a resolution, deriving the unit via [`get_unit`].
52    ///
53    /// This may prompt for the unit's save directory when `M != k` (see [`get_unit`]); for a fully
54    /// non-interactive setup, use [`ExtAlgebra::new`] with an explicit unit instead.
55    pub fn from_resolution(resolution: Arc<QueryModuleResolution>) -> anyhow::Result<Self> {
56        let (_, unit) = get_unit(Arc::clone(&resolution))?;
57        Ok(Self::new(resolution, unit))
58    }
59
60    /// Ensure both the resolution and the unit are computed through the given stem.
61    pub fn compute_through_stem(&self, max: Bidegree) {
62        self.unit.compute_through_stem(max);
63        if !self.is_unit {
64            self.resolution.compute_through_stem(max);
65        }
66    }
67}

69impl<CC: FreeChainComplex> ExtAlgebra<CC> {
70    /// Build an [`ExtAlgebra`] from an explicit `(resolution, unit)` pair.
71    pub fn new(resolution: Arc<CC>, unit: Arc<CC>) -> Self {
72        assert_eq!(resolution.prime(), unit.prime());
73        Self {
74            is_unit: Arc::ptr_eq(&resolution, &unit),
75            resolution,
76            unit,
77            products: DashMap::new(),
78        }
79    }
80
81    /// Build an [`ExtAlgebra`] for resolution-*intrinsic* operations that do not involve products
82    /// (notably the secondary `d2` differential), using the resolution itself in place of a unit.
83    ///
84    /// This avoids the unit-resolution setup (and any associated prompt) that
85    /// [`from_resolution`](Self::from_resolution) performs. The product methods
86    /// ([`multiply`](Self::multiply) etc.) and the unit-side queries are only meaningful here when
87    /// `M == k`; for products with `M != k`, build with [`from_resolution`](Self::from_resolution)
88    /// or [`new`](Self::new) instead.
89    pub fn without_unit(resolution: Arc<CC>) -> Self {
90        Self::new(Arc::clone(&resolution), resolution)
91    }
92
93    pub fn resolution(&self) -> &Arc<CC> {
94        &self.resolution
95    }
96
97    pub fn unit(&self) -> &Arc<CC> {
98        &self.unit
99    }
100
101    pub fn is_unit(&self) -> bool {
102        self.is_unit
103    }
104
105    pub fn prime(&self) -> ValidPrime {
106        self.resolution.prime()
107    }
108
109    /// Ensure both the resolution and the unit are computed through the given bidegree.
110    pub fn compute_through_bidegree(&self, b: Bidegree) {
111        self.unit.compute_through_bidegree(b);
112        if !self.is_unit {
113            self.resolution.compute_through_bidegree(b);
114        }
115    }
116
117    /// The dimension of $\Ext^{s,t}(M, k)$ at the given bidegree.
118    pub fn dimension(&self, b: Bidegree) -> usize {
119        self.resolution.number_of_gens_in_bidegree(b)
120    }
121
122    /// The basis generators of $\Ext(M, k)$ at the given bidegree.
123    pub fn basis(&self, b: Bidegree) -> Vec<BidegreeGenerator> {
124        (0..self.dimension(b))
125            .map(|i| BidegreeGenerator::new(b, i))
126            .collect()
127    }
128
129    /// A class in $\Ext(M, k)$ from its coordinates in the generator basis at bidegree `b`.
130    pub fn element(&self, b: Bidegree, coords: &[u32]) -> BidegreeElement {
131        assert_eq!(self.dimension(b), coords.len());
132        BidegreeElement::new(b, FpVector::from_slice(self.prime(), coords))
133    }
134
135    /// A single generator of $\Ext(M, k)$ as a class.
136    pub fn generator(&self, g: BidegreeGenerator) -> BidegreeElement {
137        let ambient = self.dimension(g.degree());
138        assert!(ambient > g.idx());
139        g.into_element(self.prime(), self.dimension(g.degree()))
140    }
141
142    /// The dimension of $\Ext(k, k)$ at the given bidegree (the multiplicand/"scalar" side).
143    pub fn unit_dimension(&self, b: Bidegree) -> usize {
144        self.unit.number_of_gens_in_bidegree(b)
145    }
146
147    /// The basis generators of $\Ext(k, k)$ at the given bidegree.
148    pub fn unit_basis(&self, b: Bidegree) -> Vec<BidegreeGenerator> {
149        (0..self.unit_dimension(b))
150            .map(|i| BidegreeGenerator::new(b, i))
151            .collect()
152    }
153
154    /// A class in $\Ext(k, k)$ from its coordinates in the generator basis at bidegree `b`.
155    pub fn unit_element(&self, b: Bidegree, coords: &[u32]) -> BidegreeElement {
156        assert_eq!(self.unit_dimension(b), coords.len());
157        BidegreeElement::new(b, FpVector::from_slice(self.prime(), coords))
158    }
159
160    /// A single generator of $\Ext(k, k)$ as a class.
161    pub fn unit_generator(&self, g: BidegreeGenerator) -> BidegreeElement {
162        let ambient = self.unit_dimension(g.degree());
163        assert!(ambient > g.idx());
164        g.into_element(self.prime(), ambient)
165    }
166}

168impl<CC> ExtAlgebra<CC>
169where
170    CC: FreeChainComplex + AugmentedChainComplex,
171{
172    /// The multiplication map for a single generator `g` of $\Ext(M, k)$, built and cached on
173    /// first use. The returned map is *not* guaranteed to be extended; [`ExtAlgebra::multiply_into`]
174    /// extends it as needed.
175    pub fn generator_product_map(
176        &self,
177        g: BidegreeGenerator,
178    ) -> Arc<ResolutionHomomorphism<CC, CC>> {
179        if let Some(map) = self.products.get(&g) {
180            return Arc::clone(&map);
181        }
182
183        let dim = self.resolution.number_of_gens_in_bidegree(g.degree());
184        let mut class = vec![0u32; dim];
185        class[g.idx()] = 1;
186
187        let name = format!("prod_{}_{}_{}", g.n(), g.s(), g.idx());
188        let hom = Arc::new(ResolutionHomomorphism::from_class(
189            name,
190            Arc::clone(&self.resolution),
191            Arc::clone(&self.unit),
192            g.degree(),
193            &class,
194        ));
195
196        Arc::clone(self.products.entry(g).or_insert(hom).value())
197    }
198
199    /// Left-multiplication by the class `x` (in $\Ext(M, k)$), applied to every basis generator of
200    /// $\Ext(k, k)$ at bidegree `b`.
201    ///
202    /// Returns `None` when the product is out of the computed range — that is, when `b` or
203    /// `b + x.degree()` has not been resolved — so callers never mistake an uncomputed product for a
204    /// zero one. Otherwise returns a matrix with one row per generator of $\Ext(k, k)$ at `b`; row
205    /// `j` is the product `x · g_j` expressed in the generator basis of $\Ext(M, k)$ at bidegree
206    /// `b + x.degree()`. A computed-but-empty bidegree yields a valid zero-dimension matrix, not
207    /// `None`.
208    pub fn multiply_into(&self, x: &BidegreeElement, b: Bidegree) -> Option<Matrix> {
209        let shift = x.degree();
210        let target = b + shift;
211
212        if !self.unit.has_computed_bidegree(b) || !self.resolution.has_computed_bidegree(target) {
213            return None;
214        }
215
216        let unit_dim = self.unit.number_of_gens_in_bidegree(b);
217        let res_dim = self.resolution.number_of_gens_in_bidegree(target);
218        let mut matrix = Matrix::new(self.prime(), unit_dim, res_dim);
219
220        for (i, c) in x.vec().iter_nonzero() {
221            let map = self.generator_product_map(BidegreeGenerator::new(shift, i));
222            map.extend_all();
223
224            // `hom_k(b.t())[j][k]`: `j` indexes the multiplicand generator of the unit at `b`, `k`
225            // indexes the result generator of the resolution at `target`.
226            let hom_k = map.get_map(target.s()).hom_k(b.t());
227            for (j, row) in hom_k.iter().enumerate() {
228                for (k, &v) in row.iter().enumerate() {
229                    matrix.row_mut(j).add_basis_element(k, c * v);
230                }
231            }
232        }
233        Some(matrix)
234    }
235
236    /// The product `x · y` if it lies in the computed range, else `None`. See
237    /// [`multiply_into`](Self::multiply_into) for the operand conventions. The result lies in
238    /// bidegree `x.degree() + y.degree()`.
239    pub fn try_multiply(
240        &self,
241        x: &BidegreeElement,
242        y: &BidegreeElement,
243    ) -> Option<BidegreeElement> {
244        let target = x.degree() + y.degree();
245        let matrix = self.multiply_into(x, y.degree())?;
246        let mut out = FpVector::new(self.prime(), matrix.columns());
247        for (j, c) in y.vec().iter_nonzero() {
248            out.as_slice_mut().add(matrix.row(j), c);
249        }
250        Some(BidegreeElement::new(target, out))
251    }
252
253    /// The product `x · y`, where `x ∈ Ext(M, k)` and `y ∈ Ext(k, k)`. When `M == k` both operands
254    /// live in the same algebra $\Ext(k, k)$. The result lies in bidegree `x.degree() + y.degree()`.
255    ///
256    /// Panics if the product is out of the computed range; use
257    /// [`try_multiply`](Self::try_multiply) to handle that case.
258    pub fn multiply(&self, x: &BidegreeElement, y: &BidegreeElement) -> BidegreeElement {
259        self.try_multiply(x, y).expect(
260            "multiply: product is out of the computed range; compute further or use try_multiply",
261        )
262    }
263}

265#[cfg(test)]
266mod tests {
267    use super::*;
268    use crate::utils::construct_standard;
269
270    #[test]
271    fn test_sphere_products() {
272        let res = Arc::new(construct_standard::<false, _, _>("S_2", None).unwrap());
273        res.compute_through_stem(Bidegree::n_s(8, 8));
274        let alg = ExtAlgebra::new(Arc::clone(&res), res);
275
276        // h_i live in Ext^{1, *}: h_0 = (n=0, s=1), h_1 = (n=1, s=1), h_2 = (n=3, s=1).
277        let h0 = alg.generator(BidegreeGenerator::new(Bidegree::n_s(0, 1), 0));
278        let h1 = alg.generator(BidegreeGenerator::new(Bidegree::n_s(1, 1), 0));
279
280        // h_0^2 is the nonzero generator of Ext^{2,2} = (n=0, s=2).
281        let h0_sq = alg.multiply(&h0, &h0);
282        assert_eq!(h0_sq.degree(), Bidegree::n_s(0, 2));
283        assert_eq!(alg.dimension(Bidegree::n_s(0, 2)), 1);
284        assert!(!h0_sq.vec().is_zero(), "h_0^2 should be nonzero");
285
286        // The Adams relations h_0 h_1 = 0 = h_1 h_0.
287        assert!(
288            alg.multiply(&h0, &h1).vec().is_zero(),
289            "h_0 h_1 should vanish"
290        );
291        assert!(
292            alg.multiply(&h1, &h0).vec().is_zero(),
293            "h_1 h_0 should vanish"
294        );
295
296        // Cross-check `multiply` against a direct `hom_k` read for h_0 · h_1.
297        let rows = alg
298            .multiply_into(&h0, h1.degree())
299            .expect("h_0 · h_1 is in range");
300        let direct: u32 = rows.row(0).iter().sum();
301        assert_eq!(direct, 0);
302    }
