1//! Primary Massey products in $\Ext$.
2//!
3//! [`ExtAlgebra::massey`] computes a single triple Massey product $\langle a, b, c\rangle$, while
4//! [`ExtAlgebra::massey_iter_c`] and [`ExtAlgebra::massey_iter_a`] sweep a whole family at once:
5//! the former fixes $a, b$ and ranges over every valid third factor $\langle a, b, -\rangle$, the
6//! latter fixes $b, c$ and ranges over every valid first factor $\langle -, b, c\rangle$. The two
7//! directions differ in whether the `b ∘ c` null-homotopy is rebuilt per `c` or reused for fixed
8//! `b, c`.
9//!
10//! All three wrap [`ChainHomotopy`]: we lift the multiplication maps, build the null-homotopy of
11//! the composite `b ∘ c`, and read off the bracket by pairing against the first factor. The valid
12//! choices of `a` and `c` are the kernel of multiplication by `b`.
13//!
14//! The result is an [`AffineSubspace`]: a coset representative (the offset) together with the
15//! indeterminacy $a \cdot \Ext + \Ext \cdot c$ (the linear part). Both terms of the indeterminacy
16//! are the $\Ext(k, k)$-module action on $\Ext(M, k)$, so it is computed for any `M`. This matches
17//! (and reuses the logic of) the `massey` example, which computes the products up to a sign.
18

33/// The result of a Massey product computation
34pub struct MasseyResult {
35    /// The bidegree of the bracket, `a.degree() + b.degree() + c.degree() - Bidegree::s_t(1, 0)`.
36    pub degree: Bidegree,
37    /// The value of the bracket as a coset.
38    pub coset: AffineSubspace,
39}
40
41impl MasseyResult {
42    /// Returns a representative element of the Massey product.
43    pub fn representative(&self) -> BidegreeElement {
44        BidegreeElement::new(self.degree, self.coset.offset().clone())
45    }
46
47    /// Whether the Massey product contains zero.
48    pub fn contains_zero(&self) -> bool {
49        self.coset.contains_zero()
50    }
51}
52
53struct MasseyComputeDatum {
54    answers: Matrix,
55    kernel: Subspace,
56    tot: Bidegree,
57}
58
59impl<CC> ExtAlgebra<CC>
60where
61    CC: FreeChainComplex + AugmentedChainComplex,
62{
63    /// The bidegree shift of $\langle a, b, -\rangle$: a class `c` produces a bracket in bidegree
64    /// `c.degree() + a.degree() + b.degree() - (1, 0)`.
65    fn massey_shift(a: &BidegreeElement, b: &BidegreeElement) -> Bidegree {
66        a.degree() + b.degree() - Bidegree::s_t(1, 0)
67    }
68
69    /// The multiplication-by-`b` chain self-map of the unit, extended far enough for brackets
70    /// landing at `shift`.
71    fn massey_b_hom(
72        &self,
73        b: &BidegreeElement,
74        shift: Bidegree,
75    ) -> Arc<ResolutionHomomorphism<CC, CC>> {
76        let b_coords: Vec<u32> = b.vec().iter().collect();
77        let hom = Arc::new(ResolutionHomomorphism::from_class(
78            String::new(),
79            Arc::clone(self.unit()),
80            Arc::clone(self.unit()),
81            b.degree(),
82            &b_coords,
83        ));
84        hom.extend_through_stem(shift);
85        hom
86    }
87
88    /// Compute, for a single multiplicand bidegree `c_deg`, the per-generator bracket values and
89    /// the kernel of multiplication by `b` (the valid third factors). The bracket values form a
90    /// `num_gens × target_num_gens` matrix whose row `gen` is the bracket of the `gen`th generator
91    /// of `c_deg`. Returns `None` if the bracket bidegree is empty or uncomputed.
92    fn massey_at(
93        &self,
94        a: &BidegreeElement,
95        b: &BidegreeElement,
96        b_hom: &Arc<ResolutionHomomorphism<CC, CC>>,
97        shift: Bidegree,
98        offset_a: usize,
99        c_deg: Bidegree,
100    ) -> Option<MasseyComputeDatum> {
101        let p = self.prime();
102        let resolution = self.resolution();
103        let unit = self.unit();
104
105        if !resolution.has_computed_bidegree(c_deg + shift) {
106            return None;
107        }
108        let tot = c_deg + shift;
109
110        let num_gens = resolution.number_of_gens_in_bidegree(c_deg);
111        let product_num_gens = resolution.number_of_gens_in_bidegree(b.degree() + c_deg);
112        let target_num_gens = resolution.number_of_gens_in_bidegree(tot);
113        if target_num_gens == 0 {
114            return None;
115        }
116
117        let a_coords: Vec<u32> = a.vec().iter().collect();
118        let b_coords: Vec<u32> = b.vec().iter().collect();
119
120        let mut answers = Matrix::new(p, num_gens, target_num_gens);
121        let mut product = AugmentedMatrix::<2>::new(p, num_gens, [product_num_gens, num_gens]);
122        product.segment(1, 1).add_identity();
123
124        let mut matrix = Matrix::new(p, num_gens, 1);
125        for idx in 0..num_gens {
126            let hom = Arc::new(ResolutionHomomorphism::new(
127                String::new(),
128                Arc::clone(resolution),
129                Arc::clone(unit),
130                c_deg,
131            ));
132
133            matrix.row_mut(idx).set_entry(0, 1);
134            hom.extend_step(c_deg, Some(&matrix));
135            matrix.row_mut(idx).set_entry(0, 0);
136
137            hom.extend_through_stem(tot);
138
139            let homotopy = ChainHomotopy::new(Arc::clone(&hom), Arc::clone(b_hom));
140            homotopy.extend(tot);
141
142            let last = homotopy.homotopy(tot.s());
143            let mut answer_row = answers.row_mut(idx);
144            for i in 0..target_num_gens {
145                let output = last.output(tot.t(), i);
146                for (k, &val) in a_coords.iter().enumerate() {
147                    if val != 0 {
148                        answer_row.add_basis_element(i, val * output.entry(offset_a + k));
149                    }
150                }
151            }
152
153            for (k, &val) in b_coords.iter().enumerate() {
154                if val != 0 {
155                    let g = BidegreeGenerator::new(b.degree(), k);
156                    hom.act(product.row_mut(idx).slice_mut(0, product_num_gens), val, g);
157                }
158            }
159        }
160        product.row_reduce();
161        let kernel = product.compute_kernel();
162
163        Some(MasseyComputeDatum {
164            answers,
165            kernel,
166            tot,
167        })
168    }

170    /// Compute a representative of a Massey product evaluated at `row` using the data returned by
171    /// [`massey_at`](Self::massey_at).
172    fn massey_representative(&self, answers: &Matrix, row: FpSlice) -> FpVector {
173        let mut v = FpVector::new(self.prime(), answers.columns());
174        answers.apply(v.as_slice_mut(), 1, row);
175        v
176    }
177
178    /// Assemble a [`MasseyResult`] at the bracket bidegree `tot` from the per-generator bracket
179    /// values `answers` and the third factor `c` (with coordinates `row`).
180    fn massey_result(
181        &self,
182        a: &BidegreeElement,
183        c: &BidegreeElement,
184        answers: &Matrix,
185        row: FpSlice,
186        tot: Bidegree,
187    ) -> MasseyResult {
188        let representative = self.massey_representative(answers, row);
189        let indeterminacy = self.massey_indeterminacy(a, c, tot);
190        MasseyResult {
191            degree: tot,
192            coset: AffineSubspace::new(representative, indeterminacy),
193        }
194    }
195
196    /// The indeterminacy $a \cdot \Ext^{|b| + |c| - (1,0)} + \Ext^{|a| + |b| - (1,0)} \cdot c$ at
197    /// the bracket bidegree `tot`, as a subspace of $\Ext(M, k)$ at `tot`.
198    fn massey_indeterminacy(
199        &self,
200        a: &BidegreeElement,
201        c: &BidegreeElement,
202        tot: Bidegree,
203    ) -> Subspace {
204        let mut sub = Subspace::new(self.prime(), self.dimension(tot));
205
206        // a · Ext(M, k)^{tot - a.degree()}, computed as y · a (equal up to sign).
207        for y in self.basis(tot - a.degree()) {
208            if let Some(prod) = self.try_multiply(&self.generator(y), a) {
209                sub.add_vector(prod.vec());
210            }
211        }
212        // Ext(k, k)^{tot - c.degree()} · c, computed as c · x (equal up to sign).
213        for x in self.unit_basis(tot - c.degree()) {
214            if let Some(prod) = self.try_multiply(c, &self.unit_generator(x)) {
215                sub.add_vector(prod.vec());
216            }
217        }
218        sub
219    }

221    /// Compute the family of Massey products $\langle a, b, -\rangle$ for fixed `a` and `b` and
222    /// every valid third factor `c` across all computed bidegrees.
223    ///
224    /// `a` and `b` are taken in $\Ext(k, k)$; the third factor ranges over $\Ext(M, k)$. The caller
225    /// must have resolved `M` and the unit far enough. This assumes `a · b = 0` so that the bracket
226    /// is defined; it is not verified.
227    ///
228    /// Brackets that contain `0` are omitted.
229    ///
230    /// This iterates over the third factor, building a fresh null-homotopy of `b ∘ c` per `c`. To
231    /// vary the *first* factor with `b, c` fixed instead, use
232    /// [`massey_iter_a`](Self::massey_iter_a).
233    pub fn massey_iter_c(
234        &self,
235        a: &BidegreeElement,
236        b: &BidegreeElement,
237    ) -> Vec<(BidegreeElement, MasseyResult)> {
238        let shift = Self::massey_shift(a, b);
239        let offset_a =
240            self.unit()
241                .module(a.degree().s())
242                .generator_offset(a.degree().t(), a.degree().t(), 0);
243        let b_hom = self.massey_b_hom(b, shift);
244
245        let mut results = Vec::new();
246        for c_deg in self.resolution().iter_nonzero_stem() {
247            let Some(MasseyComputeDatum {
248                answers,
249                kernel,
250                tot,
251            }) = self.massey_at(a, b, &b_hom, shift, offset_a, c_deg)
252            else {
253                continue;
254            };
255            for row in kernel.iter() {
256                let c = BidegreeElement::new(c_deg, row.to_owned());
257                let result = self.massey_result(a, &c, &answers, row, tot);
258                if result.contains_zero() {
259                    continue;
260                }
261                results.push((c, result));
262            }
263        }
264        results
265    }
266
267    /// Compute the family of Massey products $\langle -, b, c\rangle$ for fixed `b` and `c` and
268    /// every valid first factor `a` across all computed bidegrees.
269    ///
270    /// Note the homotopy is read at the *first factor's* filtration, so the cost grows with how far
271    /// out the first factor ranges (and `f_b` must be extended over that range). For the dual
272    /// pattern — fixed small `a, b`, sweeping a large third factor —
273    /// [`massey_iter_c`](Self::massey_iter_c) is faster, since it reads at the small fixed `a.s`.
274    ///
275    /// `b` is taken in $\Ext(k, k)$ and `c` in $\Ext(M, k)$; the first factor ranges over $\Ext(k,
276    /// k)$. The caller must have resolved `M` and the unit far enough. This assumes `b · c = 0` so
277    /// that the bracket is defined; it is not verified. Brackets that contain `0` are omitted.
278    pub fn massey_iter_a(
279        &self,
280        b: &BidegreeElement,
281        c: &BidegreeElement,
282    ) -> Vec<(BidegreeElement, MasseyResult)> {
283        let p = self.prime();
284        let resolution = self.resolution();
285        let unit = self.unit();
286
287        // The bracket of a first factor `a` lands at `tot = a.degree() + bc_shift`.
288        let bc_shift = b.degree() + c.degree() - Bidegree::s_t(1, 0);
289
290        // `f_c` realises `c` (resolution of `M` → unit); `f_b` is multiplication by `b` (in the
291        // unit). The single null-homotopy `s_bc` of `b ∘ c` is reused for every first factor.
292        let c_coords: Vec<u32> = c.vec().iter().collect();
293        let f_c = Arc::new(ResolutionHomomorphism::from_class(
294            String::new(),
295            Arc::clone(resolution),
296            Arc::clone(unit),
297            c.degree(),
298            &c_coords,
299        ));
300        let b_coords: Vec<u32> = b.vec().iter().collect();
301        let f_b = Arc::new(ResolutionHomomorphism::from_class(
302            String::new(),
303            Arc::clone(unit),
304            Arc::clone(unit),
305            b.degree(),
306            &b_coords,
307        ));
308        let s_bc = ChainHomotopy::new(Arc::clone(&f_c), Arc::clone(&f_b));
309
310        let mut results = Vec::new();
311        for a_deg in unit.iter_nonzero_stem() {
312            let tot = a_deg + bc_shift;
313            if !resolution.has_computed_bidegree(tot) {
314                continue;
315            }
316            let target_num_gens = resolution.number_of_gens_in_bidegree(tot);
317            let unit_dim = unit.number_of_gens_in_bidegree(a_deg);
318            if target_num_gens == 0 || unit_dim == 0 {
319                continue;
320            }
321
322            // Extend the maps and the (single) homotopy far enough to read off this bracket.
323            f_c.extend_through_stem(tot);
324            f_b.extend_through_stem(tot);
325            s_bc.extend(tot);
326
327            // `answers[j][i]` is the bracket of the `j`th first-factor generator, read off the
328            // homotopy at filtration `a.s` (where the first factor lives).
329            let offset_a = unit
330                .module(a_deg.s())
331                .generator_offset(a_deg.t(), a_deg.t(), 0);
332            let last = s_bc.homotopy(tot.s());
333            let mut answers = Matrix::new(p, unit_dim, target_num_gens);
334            for i in 0..target_num_gens {
335                let output = last.output(tot.t(), i);
336                for j in 0..unit_dim {
337                    answers.row_mut(j).set_entry(i, output.entry(offset_a + j));
338                }
339            }
340
341            // Valid first factors are the kernel of `(· b)` on $\Ext(k, k)$, computed as `b · a`
342            // (equal up to sign, so the same kernel) via the fixed `f_b`.
343            let product_num_gens = unit.number_of_gens_in_bidegree(a_deg + b.degree());
344            let mut product = AugmentedMatrix::<2>::new(p, unit_dim, [product_num_gens, unit_dim]);
345            product.segment(1, 1).add_identity();
346            for j in 0..unit_dim {
347                let g = BidegreeGenerator::new(a_deg, j);
348                f_b.act(product.row_mut(j).slice_mut(0, product_num_gens), 1, g);
349            }
350            product.row_reduce();
351            let kernel = product.compute_kernel();
352
353            for row in kernel.iter() {
354                let a = BidegreeElement::new(a_deg, row.to_owned());
355                let result = self.massey_result(&a, c, &answers, row, tot);
356                if result.contains_zero() {
357                    continue;
358                }
359                results.push((a, result));
360            }
361        }
362        results
363    }

365    /// Compute the triple Massey product $\langle a, b, c\rangle$.
366    ///
367    /// `a` and `b` are taken in $\Ext(k, k)$ and `c` in $\Ext(M, k)$. Returns `None` if `a · b !=
368    /// 0` or `b · c != 0`.
369    pub fn massey(
370        &self,
371        a: &BidegreeElement,
372        b: &BidegreeElement,
373        c: &BidegreeElement,
374    ) -> Option<MasseyResult> {
375        let shift = Self::massey_shift(a, b);
376        let offset_a =
377            self.unit()
378                .module(a.degree().s())
379                .generator_offset(a.degree().t(), a.degree().t(), 0);
380        let b_hom = self.massey_b_hom(b, shift);
381
382        // The bracket is defined only when `a · b = 0`. Compute `b · a` (equal to `a · b` up to
383        // sign, so the same vanishing condition) via the multiplication-by-`b` self-map of the
384        // unit, `b_hom`. The product lands at `a.degree() + b.degree()`, one filtration above
385        // `shift`, so `b_hom` must be extended one step further than `massey_b_hom` built it.
386        let ab_deg = a.degree() + b.degree();
387        b_hom.extend_through_stem(ab_deg);
388        let mut ab = FpVector::new(self.prime(), self.unit().number_of_gens_in_bidegree(ab_deg));
389        for (j, coef) in a.vec().iter_nonzero() {
390            b_hom.act(
391                ab.as_slice_mut(),
392                coef,
393                BidegreeGenerator::new(a.degree(), j),
394            );
395        }
396        if !ab.is_zero() {
397            return None;
398        }
399
400        let MasseyComputeDatum {
401            answers,
402            kernel,
403            tot,
404        } = self.massey_at(a, b, &b_hom, shift, offset_a, c.degree())?;
405
406        let mut reduced = c.vec().to_owned();
407        kernel.reduce(reduced.as_slice_mut());
408        if !reduced.is_zero() {
409            return None;
410        }
411
412        Some(self.massey_result(a, c, &answers, c.vec(), tot))
413    }

423    #[test]
424    fn test_sphere_massey() {
425        let res = Arc::new(construct_standard::<false, _, _>("S_2", None).unwrap());
426        res.compute_through_stem(Bidegree::n_s(6, 5));
427        let alg = ExtAlgebra::new(Arc::clone(&res), res);
428
429        let h0 = alg.generator(BidegreeGenerator::new(Bidegree::n_s(0, 1), 0));
430        let h1 = alg.generator(BidegreeGenerator::new(Bidegree::n_s(1, 1), 0));
431
432        // The classic relation <h0, h1, h0> = h1^2, the generator of Ext^{2,4} at (2, 2).
433        let bracket = alg
434            .massey(&h0, &h1, &h0)
435            .expect("<h0, h1, h0> should be defined");
436        assert_eq!(bracket.degree, Bidegree::n_s(2, 2));
437        assert_eq!(alg.dimension(Bidegree::n_s(2, 2)), 1);
438        assert!(
439            !bracket.coset.offset().is_zero(),
440            "<h0, h1, h0> = h1^2 should be nonzero"
441        );
442
443        let h1_sq = alg.multiply(&h1, &h1);
444        assert_eq!(
445            bracket.coset.offset().iter().collect::<Vec<_>>(),
446            h1_sq.vec().iter().collect::<Vec<_>>(),
447            "<h0, h1, h0> should equal h1^2"
448        );
449        // The indeterminacy a·Ext + Ext·c vanishes here, so the bracket is a single class.
450        assert_eq!(bracket.coset.linear_part().dimension(), 0);
451
452        // <h0, h1, h1> is undefined: h1 · h1 = h1^2 != 0, so h1 is not a valid third factor.
453        assert!(
454            alg.massey(&h0, &h1, &h1).is_none(),
455            "<h0, h1, h1> should be undefined since h1^2 != 0"
456        );
457
458        // <h0, h0, h1> is undefined: h0 · h0 = h0^2 != 0, so the bracket's a · b = 0 fails.
459        assert!(
460            alg.massey(&h0, &h0, &h1).is_none(),
461            "<h0, h0, h1> should be undefined since h0^2 != 0"
462        );
463    }

465    /// For `M == k`, iterating the first factor must agree with iterating the third, via the
466    /// symmetry `<h0, h1, x> = ±<x, h1, h0>`. At `p = 2` the sign is trivial, so the cosets match
467    /// exactly. This pins that the single-homotopy `massey_iter_a` path is correct.
468    #[test]
469    fn test_iter_a_matches_iter_c() {
470        let res = Arc::new(construct_standard::<false, _, _>("S_2", None).unwrap());
471        res.compute_through_stem(Bidegree::n_s(6, 5));
472        let alg = ExtAlgebra::new(Arc::clone(&res), res);
473
474        let h0 = alg.generator(BidegreeGenerator::new(Bidegree::n_s(0, 1), 0));
475        let h1 = alg.generator(BidegreeGenerator::new(Bidegree::n_s(1, 1), 0));
476
477        // `<h0, h1, ->` over the third factor vs `<-, h1, h0>` over the first factor.
478        let by_c = alg.massey_iter_c(&h0, &h1);
479        let by_a = alg.massey_iter_a(&h1, &h0);
480        assert!(!by_c.is_empty(), "expected some defined brackets");
481
482        let normalize = |family: Vec<(BidegreeElement, MasseyResult)>| {
483            let mut keyed: Vec<(String, AffineSubspace)> = family
484                .into_iter()
485                .map(|(x, result)| (format!("{x}"), result.coset))
486                .collect();
487            keyed.sort_by(|l, r| l.0.cmp(&r.0));
488            keyed
489        };
490        assert_eq!(normalize(by_c), normalize(by_a));
491    }
