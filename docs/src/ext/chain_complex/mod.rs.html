32pub trait FreeChainComplex<const U: bool = false>:
33    ChainComplex<
34        Module = MuFreeModule<U, <Self as ChainComplex>::Algebra>,
35        Homomorphism = MuFreeModuleHomomorphism<
36            U,
37            MuFreeModule<U, <Self as ChainComplex>::Algebra>,
38        >,
39    >
40where
41    <Self as ChainComplex>::Algebra: MuAlgebra<U>,
42{
43    fn graded_dimension_string(&self) -> String {
44        let mut result = String::new();
45        let min_degree = self.min_degree();
46        for s in (0..self.next_homological_degree()).rev() {
47            let module = self.module(s);
48
49            for t in min_degree + s..=module.max_computed_degree() {
50                result.push(unicode_num(module.number_of_gens_in_degree(t)));
51                result.push(' ');
52            }
53            result.push('\n');
54            // If it is empty so far, don't print anything
55            if result.trim_start().is_empty() {
56                result.clear()
57            }
58        }
59        result
60    }
61
62    fn to_sseq(&self) -> sseq::Sseq<sseq::Adams> {
63        let p = self.prime();
64        let mut sseq = sseq::Sseq::new(p, Bidegree::n_s(self.min_degree(), 0));
65        for b in self.iter_stem() {
66            sseq.set_dimension(b, self.number_of_gens_in_bidegree(b));
67        }
68        sseq
69    }
70
71    fn filtration_one_products(&self, op_deg: i32, op_idx: usize) -> sseq::Product {
72        let p = self.prime();
73        let mut matrices = BiVec::new(self.min_degree());
74        let max_y = self.next_homological_degree() - 1;
75        matrices.extend_with(self.module(0).max_computed_degree() - op_deg + 2, |x| {
76            let mut entries = BiVec::with_capacity(0, max_y);
77            let mut b = Bidegree::n_s(x, 0);
78            while self.has_computed_bidegree(b + Bidegree::s_t(1, op_deg)) {
79                entries.push(
80                    self.filtration_one_product(op_deg, op_idx, b)
81                        .map(|m| Matrix::from_vec(p, &m)),
82                );
83                b = b + Bidegree::n_s(0, 1);
84            }
85            entries
86        });
87
88        sseq::Product {
89            b: Bidegree::x_y(op_deg - 1, 1),
90            left: true,
91            matrices,
92        }
93    }
94
95    /// Computes the filtration one product.
96    ///
97    /// # Returns
98    /// If the chain complex is stable, this always returns `Some`. If it is unstable, this returns
99    /// `None` if the product is not defined.
100    fn filtration_one_product(
101        &self,
102        op_deg: i32,
103        op_idx: usize,
104        source: Bidegree,
105    ) -> Option<Vec<Vec<u32>>> {
106        let target = source + Bidegree::s_t(1, op_deg);
107        if !self.has_computed_bidegree(target) {
108            return None;
109        }
110
111        let source_mod = self.module(target.s() - 1);
112        let target_mod = self.module(target.s());
113
114        if U && op_idx >= self.algebra().dimension_unstable(op_deg, source.t()) {
115            return None;
116        }
117
118        let source_dim = source_mod.number_of_gens_in_degree(source.t());
119        let target_dim = target_mod.number_of_gens_in_degree(target.t());
120
121        let d = self.differential(target.s());
122
123        let mut products = vec![Vec::with_capacity(target_dim); source_dim];
124        for i in 0..target_dim {
125            let dx = d.output(target.t(), i);
126
127            for (j, row) in products.iter_mut().enumerate() {
128                let idx = source_mod.operation_generator_to_index(op_deg, op_idx, source.t(), j);
129                row.push(dx.entry(idx));
130            }
131        }
132
133        Some(products)
134    }
135
136    fn number_of_gens_in_bidegree(&self, b: Bidegree) -> usize {
137        self.module(b.s()).number_of_gens_in_degree(b.t())
138    }
139
140    /// Iterate through all nonzero bidegrees in increasing order of stem.
141    fn iter_nonzero_stem(&self) -> impl Iterator<Item = Bidegree> + '_ {
142        self.iter_stem()
143            .filter(move |&b| self.number_of_gens_in_bidegree(b) > 0)
144    }
145
146    /// Get a string representation of d(gen), where d is the differential of the resolution.
147    fn boundary_string(&self, g: BidegreeGenerator) -> String {
148        let d = self.differential(g.s());
149        let target = d.target();
150        let result_vector = d.output(g.t(), g.idx());
151
152        target.element_to_string(g.t(), result_vector.as_slice())
153    }
154}

166/// A chain complex is defined to start in degree 0. The min_degree is the min_degree of the
167/// modules in the chain complex, all of which must be the same.
168pub trait ChainComplex: Send + Sync {
169    type Algebra: Algebra;
170    type Module: Module<Algebra = Self::Algebra>;
171    type Homomorphism: ModuleHomomorphism<Source = Self::Module, Target = Self::Module>;
172
173    fn prime(&self) -> ValidPrime {
174        self.algebra().prime()
175    }
176
177    fn algebra(&self) -> Arc<Self::Algebra>;
178    fn min_degree(&self) -> i32;
179    fn zero_module(&self) -> Arc<Self::Module>;
180    fn module(&self, homological_degree: i32) -> Arc<Self::Module>;
181
182    /// This returns the differential starting from the sth module.
183    fn differential(&self, s: i32) -> Arc<Self::Homomorphism>;
184
185    /// If the complex has been computed at bidegree (s, t). This means the module has been
186    /// computed at (s, t), and so has the differential at (s, t). In the case of a free module,
187    /// the target of the differential, namely the bidegree (s - 1, t), need not be computed, as
188    /// long as all the generators hit by the differential have already been computed.
189    fn has_computed_bidegree(&self, b: Bidegree) -> bool;
190
191    /// Ensure all bidegrees less than or equal to (s, t) have been computed
192    fn compute_through_bidegree(&self, b: Bidegree);
193
194    /// The first s such that `self.module(s)` is not defined.
195    fn next_homological_degree(&self) -> i32;
196
197    /// Iterate through all defined bidegrees in increasing order of stem.
198    fn iter_stem(&self) -> StemIterator<'_, Self> {
199        StemIterator {
200            cc: self,
201            current: Bidegree::n_s(self.min_degree(), 0),
202            max_s: self.next_homological_degree(),
203        }
204    }
205
206    /// Apply the quasi-inverse of the (s, t)th differential to the list of inputs and results.
207    /// This defaults to applying `self.differentials(s).quasi_inverse(t)`, but in some cases
208    /// the quasi-inverse might be stored separately on disk.
209    ///
210    /// This returns whether the application was successful
211    #[must_use]
212    fn apply_quasi_inverse<T, S>(&self, results: &mut [T], b: Bidegree, inputs: &[S]) -> bool
213    where
214        for<'a> &'a mut T: Into<FpSliceMut<'a>>,
215        for<'a> &'a S: Into<FpSlice<'a>>,
216    {
217        assert_eq!(results.len(), inputs.len());
218        if results.is_empty() {
219            return true;
220        }
221
222        let mut iter = inputs.iter().zip_eq(results);
223        let (input, result) = iter.next().unwrap();
224        let d = self.differential(b.s());
225        if d.apply_quasi_inverse(result.into(), b.t(), input.into()) {
226            for (input, result) in iter {
227                assert!(d.apply_quasi_inverse(result.into(), b.t(), input.into()));
228            }
229            true
230        } else {
231            false
232        }
233    }
234
235    /// A directory used to save information about the chain complex.
236    fn save_dir(&self) -> &SaveDirectory {
237        &SaveDirectory::None
238    }
239
240    /// Get the save file of a bidegree
241    fn save_file(
242        &self,
243        kind: crate::save::SaveKind,
244        b: Bidegree,
245    ) -> crate::save::SaveFile<Self::Algebra> {
246        crate::save::SaveFile {
247            algebra: self.algebra(),
248            kind,
249            b,
250            idx: None,
251        }
252    }
253}

255/// An iterator returned by [`ChainComplex::iter_stem`]
256pub struct StemIterator<'a, CC: ?Sized> {
257    cc: &'a CC,
258    current: Bidegree,
259    max_s: i32,
260}
261
262impl<CC: ChainComplex + ?Sized> Iterator for StemIterator<'_, CC> {
263    type Item = Bidegree;
264
265    fn next(&mut self) -> Option<Self::Item> {
266        if self.max_s == 0 {
267            return None;
268        }
269        let cur = self.current;
270
271        if cur.s() == self.max_s {
272            self.current = Bidegree::n_s(cur.n() + 1, 0);
273            return self.next();
274        }
275        if cur.t() > self.cc.module(cur.s()).max_computed_degree() {
276            if cur.s() == 0 {
277                return None;
278            } else {
279                self.current = Bidegree::n_s(cur.n() + 1, 0);
280                return self.next();
281            }
282        }
283        self.current = cur + Bidegree::n_s(0, 1);
284        Some(cur)
285    }
286}

288/// An augmented chain complex is a map of chain complexes C -> D that is a *quasi-isomorphism*. We
289/// usually think of C as a resolution of D. The chain map must be a map of degree shift 0.
290pub trait AugmentedChainComplex: ChainComplex {
291    type TargetComplex: ChainComplex<Algebra = Self::Algebra>;
292    type ChainMap: ModuleHomomorphism<
293            Source = Self::Module,
294            Target = <Self::TargetComplex as ChainComplex>::Module,
295        >;
296
297    fn target(&self) -> Arc<Self::TargetComplex>;
298    fn chain_map(&self, s: i32) -> Arc<Self::ChainMap>;
299}

301/// A bounded chain complex is a chain complex C for which C_s = 0 for all s >= max_s
302pub trait BoundedChainComplex: ChainComplex {
303    fn max_s(&self) -> i32;
304
305    fn euler_characteristic(&self, t: i32) -> isize {
306        (0..self.max_s())
307            .map(|s| (if s % 2 == 0 { 1 } else { -1 }) * self.module(s).dimension(t) as isize)
308            .sum()
309    }
310}
