32pub trait FreeChainComplex<const U: bool = false>:
33    ChainComplex<
34    Module = MuFreeModule<U, <Self as ChainComplex>::Algebra>,
35    Homomorphism = MuFreeModuleHomomorphism<U, MuFreeModule<U, <Self as ChainComplex>::Algebra>>,
36>
37where
38    <Self as ChainComplex>::Algebra: MuAlgebra<U>,
39{
40    fn graded_dimension_string(&self) -> String {
41        let mut result = String::new();
42        let min_degree = self.min_degree();
43        for s in (0..self.next_homological_degree()).rev() {
44            let module = self.module(s);
45
46            for t in min_degree + s..=module.max_computed_degree() {
47                result.push(unicode_num(module.number_of_gens_in_degree(t)));
48                result.push(' ');
49            }
50            result.push('\n');
51            // If it is empty so far, don't print anything
52            if result.trim_start().is_empty() {
53                result.clear()
54            }
55        }
56        result
57    }
58
59    fn to_sseq(&self) -> sseq::Sseq<sseq::Adams> {
60        let p = self.prime();
61        let mut sseq = sseq::Sseq::new(p, self.min_degree(), 0);
62        for b in self.iter_stem() {
63            sseq.set_dimension(b.n(), b.s(), self.number_of_gens_in_bidegree(b));
64        }
65        sseq
66    }
67
68    fn filtration_one_products(&self, op_deg: i32, op_idx: usize) -> sseq::Product {
69        let p = self.prime();
70        let mut matrices = BiVec::new(self.min_degree());
71        let max_y = self.next_homological_degree() - 1;
72        matrices.extend_with(self.module(0).max_computed_degree() - op_deg + 2, |x| {
73            let mut entries = BiVec::with_capacity(0, max_y);
74            let mut b = Bidegree::n_s(x, 0);
75            while self.has_computed_bidegree(b + Bidegree::s_t(1, op_deg)) {
76                entries.push(
77                    self.filtration_one_product(op_deg, op_idx, b)
78                        .map(|m| Matrix::from_vec(p, &m)),
79                );
80                b = b + Bidegree::n_s(0, 1);
81            }
82            entries
83        });
84
85        sseq::Product {
86            left: true,
87            x: op_deg - 1,
88            y: 1,
89            matrices,
90        }
91    }
92
93    /// Computes the filtration one product.
94    ///
95    /// # Returns
96    /// If the chain complex is stable, this always returns `Some`. If it is unstable, this returns
97    /// `None` if the product is not defined.
98    fn filtration_one_product(
99        &self,
100        op_deg: i32,
101        op_idx: usize,
102        source: Bidegree,
103    ) -> Option<Vec<Vec<u32>>> {
104        let target = source + Bidegree::s_t(1, op_deg);
105        if !self.has_computed_bidegree(target) {
106            return None;
107        }
108
109        let source_mod = self.module(target.s() - 1);
110        let target_mod = self.module(target.s());
111
112        if U && op_idx >= self.algebra().dimension_unstable(op_deg, source.t()) {
113            return None;
114        }
115
116        let source_dim = source_mod.number_of_gens_in_degree(source.t());
117        let target_dim = target_mod.number_of_gens_in_degree(target.t());
118
119        let d = self.differential(target.s());
120
121        let mut products = vec![Vec::with_capacity(target_dim); source_dim];
122        for i in 0..target_dim {
123            let dx = d.output(target.t(), i);
124
125            for (j, row) in products.iter_mut().enumerate() {
126                let idx = source_mod.operation_generator_to_index(op_deg, op_idx, source.t(), j);
127                row.push(dx.entry(idx));
128            }
129        }
130
131        Some(products)
132    }
133
134    fn number_of_gens_in_bidegree(&self, b: Bidegree) -> usize {
135        self.module(b.s()).number_of_gens_in_degree(b.t())
136    }
137
138    /// Iterate through all nonzero bidegrees in increasing order of stem.
139    fn iter_nonzero_stem(&self) -> impl Iterator<Item = Bidegree> + '_ {
140        self.iter_stem()
141            .filter(move |&b| self.number_of_gens_in_bidegree(b) > 0)
142    }
143
144    /// Get a string representation of d(gen), where d is the differential of the resolution.
145    fn boundary_string(&self, gen: BidegreeGenerator, compact: bool) -> String {
146        let d = self.differential(gen.s());
147        let target = d.target();
148        let result_vector = d.output(gen.t(), gen.idx());
149
150        BidegreeElement::new(gen.degree(), result_vector.clone())
151            .to_string_module(&*target, compact)
152    }
153}

165/// A chain complex is defined to start in degree 0. The min_degree is the min_degree of the
166/// modules in the chain complex, all of which must be the same.
167pub trait ChainComplex: Send + Sync {
168    type Algebra: Algebra;
169    type Module: Module<Algebra = Self::Algebra>;
170    type Homomorphism: ModuleHomomorphism<Source = Self::Module, Target = Self::Module>;
171
172    fn prime(&self) -> ValidPrime {
173        self.algebra().prime()
174    }
175
176    fn algebra(&self) -> Arc<Self::Algebra>;
177    fn min_degree(&self) -> i32;
178    fn zero_module(&self) -> Arc<Self::Module>;
179    fn module(&self, homological_degree: i32) -> Arc<Self::Module>;
180
181    /// This returns the differential starting from the sth module.
182    fn differential(&self, s: i32) -> Arc<Self::Homomorphism>;
183
184    /// If the complex has been computed at bidegree (s, t). This means the module has been
185    /// computed at (s, t), and so has the differential at (s, t). In the case of a free module,
186    /// the target of the differential, namely the bidegree (s - 1, t), need not be computed, as
187    /// long as all the generators hit by the differential have already been computed.
188    fn has_computed_bidegree(&self, b: Bidegree) -> bool;
189
190    /// Ensure all bidegrees less than or equal to (s, t) have been computed
191    fn compute_through_bidegree(&self, b: Bidegree);
192
193    /// The first s such that `self.module(s)` is not defined.
194    fn next_homological_degree(&self) -> i32;
195
196    /// Iterate through all defined bidegrees in increasing order of stem.
197    fn iter_stem(&self) -> StemIterator<'_, Self> {
198        StemIterator {
199            cc: self,
200            current: Bidegree::n_s(self.min_degree(), 0),
201            max_s: self.next_homological_degree(),
202        }
203    }
204
205    /// Apply the quasi-inverse of the (s, t)th differential to the list of inputs and results.
206    /// This defaults to applying `self.differentials(s).quasi_inverse(t)`, but in some cases
207    /// the quasi-inverse might be stored separately on disk.
208    ///
209    /// This returns whether the application was successful
210    #[must_use]
211    fn apply_quasi_inverse<T, S>(&self, results: &mut [T], b: Bidegree, inputs: &[S]) -> bool
212    where
213        for<'a> &'a mut T: Into<FpSliceMut<'a>>,
214        for<'a> &'a S: Into<FpSlice<'a>>,
215    {
216        assert_eq!(results.len(), inputs.len());
217        if results.is_empty() {
218            return true;
219        }
220
221        let mut iter = inputs.iter().zip_eq(results);
222        let (input, result) = iter.next().unwrap();
223        let d = self.differential(b.s());
224        if d.apply_quasi_inverse(result.into(), b.t(), input.into()) {
225            for (input, result) in iter {
226                assert!(d.apply_quasi_inverse(result.into(), b.t(), input.into()));
227            }
228            true
229        } else {
230            false
231        }
232    }
233
234    /// A directory used to save information about the chain complex.
235    fn save_dir(&self) -> &SaveDirectory {
236        &SaveDirectory::None
237    }
238
239    /// Get the save file of a bidegree
240    fn save_file(
241        &self,
242        kind: crate::save::SaveKind,
243        b: Bidegree,
244    ) -> crate::save::SaveFile<Self::Algebra> {
245        crate::save::SaveFile {
246            algebra: self.algebra(),
247            kind,
248            b,
249            idx: None,
250        }
251    }
252}

254/// An iterator returned by [`ChainComplex::iter_stem`]
255pub struct StemIterator<'a, CC: ?Sized> {
256    cc: &'a CC,
257    current: Bidegree,
258    max_s: i32,
259}
260
261impl<CC: ChainComplex + ?Sized> Iterator for StemIterator<'_, CC> {
262    type Item = Bidegree;
263
264    fn next(&mut self) -> Option<Self::Item> {
265        if self.max_s == 0 {
266            return None;
267        }
268        let cur = self.current;
269
270        if cur.s() == self.max_s {
271            self.current = Bidegree::n_s(cur.n() + 1, 0);
272            return self.next();
273        }
274        if cur.t() > self.cc.module(cur.s()).max_computed_degree() {
275            if cur.s() == 0 {
276                return None;
277            } else {
278                self.current = Bidegree::n_s(cur.n() + 1, 0);
279                return self.next();
280            }
281        }
282        self.current = cur + Bidegree::n_s(0, 1);
283        Some(cur)
284    }
285}

287/// An augmented chain complex is a map of chain complexes C -> D that is a *quasi-isomorphism*. We
288/// usually think of C as a resolution of D. The chain map must be a map of degree shift 0.
289pub trait AugmentedChainComplex: ChainComplex {
290    type TargetComplex: ChainComplex<Algebra = Self::Algebra>;
291    type ChainMap: ModuleHomomorphism<
292        Source = Self::Module,
293        Target = <Self::TargetComplex as ChainComplex>::Module,
294    >;
295
296    fn target(&self) -> Arc<Self::TargetComplex>;
297    fn chain_map(&self, s: i32) -> Arc<Self::ChainMap>;
298}

300/// A bounded chain complex is a chain complex C for which C_s = 0 for all s >= max_s
301pub trait BoundedChainComplex: ChainComplex {
302    fn max_s(&self) -> i32;
303
304    fn euler_characteristic(&self, t: i32) -> isize {
305        (0..self.max_s())
306            .map(|s| (if s % 2 == 0 { 1 } else { -1 }) * self.module(s).dimension(t) as isize)
307            .sum()
308    }
309}
