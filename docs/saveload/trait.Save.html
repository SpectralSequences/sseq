let v : Vec<u32> = vec![6, 3, 4, 2];

let mut cursor : Cursor<Vec<u8>> = Cursor::new(Vec::new());
v.save(&mut cursor)?;

let mut w : Vec<u32> = Load::load(&mut cursor, &())?;

assert_eq!(v, w);
assert_eq!(0, cursor.bytes().count());
